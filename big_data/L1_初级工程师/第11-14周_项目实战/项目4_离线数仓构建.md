---

## 二、项目目标

1. **掌握数仓分层架构设计**：理解每层的作用、设计原则和数据流转
2. **实践ETL全流程**：从MySQL数据采集到最终报表输出
3. **综合运用Hadoop/Spark/Hive技术栈**：Sqoop采集 + Hive建模 + Spark计算
4. **产出可交付的数仓项目**：完整DDL + ETL脚本 + 数据质量检查 + 文档

### 4.3 数据生成脚本

```python
"""
generate_data.py — 生成模拟电商数据

生成内容:
  1. 用户表: 10,000条
  2. 商品表: 5,000条（10个品类）
  3. 订单表: 50,000条（时间跨度3个月）
  4. 订单明细: 200,000条
  5. 用户行为日志: 1,000,000条/天 × 90天

输出格式:
  - MySQL表通过SQL文件导入
  - 日志文件直接写入HDFS
"""
import random
import csv
from datetime import datetime, timedelta

# 配置
NUM_USERS = 10000
NUM_SKUS = 5000
NUM_CATEGORIES = 10
NUM_ORDERS = 50000
LOG_PER_DAY = 1000000
DATE_START = datetime(2024, 1, 1)
DATE_END = datetime(2024, 3, 31)

# 品类列表
CATEGORIES = [
    (1, "手机数码"), (2, "家用电器"), (3, "服装鞋帽"),
    (4, "食品饮料"), (5, "美妆个护"), (6, "运动户外"),
    (7, "图书文娱"), (8, "家具家居"), (9, "母婴用品"),
    (10, "汽车用品")
]

# 行为类型
BEHAVIORS = ['pv', 'pv', 'pv', 'pv', 'buy', 'cart', 'fav']

def generate_users():
    """生成用户数据"""
    users = []
    provinces = ["北京", "上海", "广东", "浙江", "江苏", "四川", "湖北", "山东",
                 "福建", "河南", "湖南", "重庆", "陕西", "辽宁", "天津"]
    cities_map = {
        "北京": ["朝阳区", "海淀区", "丰台区", "东城区"],
        "上海": ["浦东新区", "黄浦区", "徐汇区", "静安区"],
        "广东": ["广州", "深圳", "东莞", "佛山"],
        "浙江": ["杭州", "宁波", "温州", "嘉兴"],
        "江苏": ["南京", "苏州", "无锡", "常州"],
    }

    for i in range(1, NUM_USERS + 1):
        province = random.choice(provinces)
        city = random.choice(cities_map.get(province, ["未知"]))
        gender = random.choice(["男", "女"])
        birthday = datetime.now() - timedelta(days=random.randint(365*18, 365*65))
        register_date = DATE_START + timedelta(days=random.randint(-180, 0))
        level = random.choices(["普通", "白银", "黄金", "VIP"], weights=[60, 25, 10, 5])[0]

        users.append({
            "user_id": i,
            "username": f"user_{i:06d}",
            "gender": gender,
            "birthday": birthday.strftime("%Y-%m-%d"),
            "province": province,
            "city": city,
            "register_date": register_date.strftime("%Y-%m-%d"),
            "user_level": level,
        })
    return users

def generate_skus():
    """生成商品数据"""
    skus = []
    brands = ["品牌A", "品牌B", "品牌C", "品牌D", "品牌E",
              "品牌F", "品牌G", "品牌H"]

    for i in range(1, NUM_SKUS + 1):
        cat_id, cat_name = random.choice(CATEGORIES)
        price = round(random.uniform(10, 10000), 2)
        skus.append({
            "sku_id": i,
            "sku_name": f"{cat_name}_商品_{i:05d}",
            "category_id": cat_id,
            "category_name": cat_name,
            "brand": random.choice(brands),
            "price": price,
            "stock": random.randint(0, 10000),
            "status": random.choices(["上架", "下架"], weights=[90, 10])[0],
        })
    return skus

def generate_orders(users, skus):
    """生成订单数据"""
    orders = []
    order_details = []

    for i in range(1, NUM_ORDERS + 1):
        user = random.choice(users)
        create_time = DATE_START + timedelta(
            days=random.randint(0, (DATE_END - DATE_START).days),
            hours=random.randint(0, 23),
            minutes=random.randint(0, 59)
        )

        num_items = random.choices([1, 2, 3, 4, 5], weights=[40, 30, 15, 10, 5])[0]
        total = 0

        for j in range(num_items):
            sku = random.choice(skus)
            qty = random.randint(1, 3)
            price = sku["price"]
            total += price * qty
            order_details.append({
                "detail_id": len(order_details) + 1,
                "order_id": i,
                "sku_id": sku["sku_id"],
                "sku_name": sku["sku_name"],
                "order_price": price,
                "sku_num": qty,
                "create_time": create_time.strftime("%Y-%m-%d %H:%M:%S"),
            })

        status = random.choices(
            ["已完成", "已取消", "待支付", "已退款"],
            weights=[75, 10, 10, 5]
        )[0]

        orders.append({
            "order_id": i,
            "user_id": user["user_id"],
            "order_status": status,
            "total_amount": round(total, 2),
            "payment_amount": round(total * random.uniform(0.9, 1.0), 2),
            "shipping_province": user["province"],
            "shipping_city": user["city"],
            "create_time": create_time.strftime("%Y-%m-%d %H:%M:%S"),
        })
    return orders, order_details

def generate_user_logs(users, skus, date_str):
    """生成单日用户行为日志"""
    logs = []
    num_records = random.randint(int(LOG_PER_DAY * 0.8), int(LOG_PER_DAY * 1.2))

    for _ in range(num_records):
        user = random.choice(users)
        sku = random.choice(skus)
        behavior = random.choice(BEHAVIORS)
        hour = random.choices(
            range(24),
            weights=[1, 1, 1, 1, 1, 2, 5, 10, 12, 12,
                     12, 10, 10, 10, 10, 10, 12, 12, 15, 15,
                     15, 10, 5, 2]
        )[0]
        minute = random.randint(0, 59)
        second = random.randint(0, 59)
        ts = datetime.strptime(
            f"{date_str} {hour:02d}:{minute:02d}:{second:02d}",
            "%Y-%m-%d %H:%M:%S"
        ).strftime("%Y-%m-%d %H:%M:%S")

        logs.append(f"{user['user_id']},{sku['sku_id']},{sku['category_id']},{behavior},{ts}")

    return logs

# 执行生成
if __name__ == "__main__":
    print("生成用户数据...")
    users = generate_users()

    print("生成商品数据...")
    skus = generate_skus()

    print("生成订单数据...")
    orders, order_details = generate_orders(users, skus)

    print(f"生成完成: 用户{len(users)} 商品{len(skus)} "
          f"订单{len(orders)} 明细{len(order_details)}")
```

### 5.2 ODS层ETL脚本

```python
"""
ods_etl.py — ODS层数据加载脚本

功能:
  1. Sqoop从MySQL抽取全量数据
  2. 加载到Hive ODS层分区表
  3. 记录加载日志
"""
import subprocess
import os
from datetime import datetime, timedelta

MYSQL_HOST = "mysql"
MYSQL_PORT = "3306"
MYSQL_USER = "root"
MYSQL_PASS = "root123"
MYSQL_DB = "ecommerce"

YESTERDAY = (datetime.now() - timedelta(days=1)).strftime("%Y-%m-%d")

def sqoop_import(table_name, target_dir, columns=None):
    """使用Sqoop从MySQL导入数据到HDFS"""
    cmd = f"""
    sqoop import \\
        --connect jdbc:mysql://{MYSQL_HOST}:{MYSQL_PORT}/{MYSQL_DB} \\
        --username {MYSQL_USER} \\
        --password {MYSQL_PASS} \\
        --table {table_name} \\
        --target-dir {target_dir}/{YESTERDAY}/ \\
        --delete-target-dir \\
        --num-mappers 4 \\
        --fields-terminated-by ',' \\
        --null-string '\\\\N' \\
        --null-non-string '\\\\N'
    """
    if columns:
        cmd += f" --columns {columns}"

    print(f"[Sqoop] 开始导入 {table_name} → {target_dir}/{YESTERDAY}/")
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    if result.returncode == 0:
        print(f"[Sqoop] {table_name} 导入成功")
    else:
        print(f"[Sqoop] {table_name} 导入失败: {result.stderr}")
    return result.returncode

def hive_load_ods(table_name, data_dir):
    """将Sqoop导入的数据加载到Hive ODS表"""
    from pyhive import hive
    conn = hive.Connection(host="hive-server", port=10000, database="ods")
    cursor = conn.cursor()

    # 添加分区
    sql = f"""
    ALTER TABLE {table_name} ADD IF NOT EXISTS 
    PARTITION (dt='{YESTERDAY}')
    LOCATION '{data_dir}'
    """
    cursor.execute(sql)
    print(f"[Hive] {table_name} 分区 dt={YESTERDAY} 已加载")
    cursor.close()
    conn.close()

def run_ods_etl():
    """执行ODS层完整ETL流程"""
    print(f"=== ODS层ETL开始 === 日期: {YESTERDAY}")

    # 1. 抽取MySQL数据到HDFS
    tables_config = {
        "user_info": {
            "table": "user_info",
            "target_dir": "/data/ods/user_info"
        },
        "sku_info": {
            "table": "sku_info",
            "target_dir": "/data/ods/sku_info"
        },
        "order_info": {
            "table": "order_info",
            "target_dir": "/data/ods/order_info"
        },
        "order_detail": {
            "table": "order_detail",
            "target_dir": "/data/ods/order_detail"
        }
    }

    for name, config in tables_config.items():
        ret = sqoop_import(config["table"], config["target_dir"])
        if ret == 0:
            hive_load_ods(f"ods_{name}", config["target_dir"])

    # 2. 用户行为日志（直接生成到HDFS）
    print("[Log] 生成用户行为日志...")
    # 此处调用generate_user_logs并上传到HDFS

    print(f"=== ODS层ETL完成 === 日期: {YESTERDAY}")

if __name__ == "__main__":
    run_ods_etl()
```

### 6.2 DWD层ETL脚本（Spark）

```python
"""
dwd_etl.py — DWD层清洗和宽表构建脚本

功能:
  1. 读取ODS层数据
  2. 数据清洗（去空、去重、格式化）
  3. 关联维度表生成宽表
  4. 添加日期维度字段
"""
from pyspark.sql import SparkSession
from pyspark.sql.functions import *
from pyspark.sql.types import *

spark = SparkSession.builder \
    .appName("DWD_ETL") \
    .enableHiveSupport() \
    .config("spark.sql.adaptive.enabled", "true") \
    .getOrCreate()

YESTERDAY = "2024-01-01"  # 应由参数传入

def etl_dwd_order_detail():
    """构建订单明细宽表"""
    print(f"[DWD] 构建订单明细宽表 dt={YESTERDAY}")

    # 读取ODS层数据
    ods_order_detail = spark.table(f"ods.ods_order_detail").filter(col("dt") == YESTERDAY)
    ods_order_info = spark.table(f"ods.ods_order_info").filter(col("dt") == YESTERDAY)
    ods_user_info = spark.table(f"ods.ods_user_info").filter(col("dt") == YESTERDAY)
    ods_sku_info = spark.table(f"ods.ods_sku_info").filter(col("dt") == YESTERDAY)

    # 关联生成宽表
    dwd = ods_order_detail \
        .join(ods_order_info.select("order_id", "user_id", "total_amount",
                                     "payment_amount", "order_status"),
              "order_id", "left") \
        .join(ods_user_info.select("user_id", col("username").alias("user_name"),
                                    col("gender").alias("user_gender"),
                                    col("user_level"),
                                    col("province").alias("user_province"),
                                    col("city").alias("user_city"),
                                    col("birthday")),
              "user_id", "left") \
        .join(ods_sku_info.select("sku_id", col("category_name"), col("brand")),
              "sku_id", "left") \
        .withColumn("user_age",
            year(current_date()) - year(to_date(col("birthday")))) \
        .withColumn("order_year", year(to_timestamp(col("create_time")))) \
        .withColumn("order_month", month(to_timestamp(col("create_time")))) \
        .withColumn("order_day", dayofmonth(to_timestamp(col("create_time")))) \
        .withColumn("order_week", weekofyear(to_timestamp(col("create_time")))) \
        .withColumn("order_hour", hour(to_timestamp(col("create_time")))) \
        .select(
            "detail_id", "order_id", "user_id", "user_name", "user_gender",
            "user_age", "user_level", "user_province", "user_city",
            "sku_id", "sku_name", "category_id", "category_name", "brand",
            "order_price", "sku_num", "total_amount", "payment_amount",
            "order_status", "create_time",
            "order_year", "order_month", "order_day", "order_week", "order_hour"
        )

    # 写入DWD层
    dwd.write \
        .mode("overwrite") \
        .insertInto("dwd.dwd_order_detail", overwrite=True)

    print(f"[DWD] 订单明细宽表完成，行数: {dwd.count()}")

def etl_dwd_user_log():
    """清洗用户行为日志"""
    print(f"[DWD] 清洗用户行为日志 dt={YESTERDAY}")

    ods_user_log = spark.table(f"ods.ods_user_log").filter(col("dt") == YESTERDAY)

    dwd = ods_user_log \
        .filter(col("user_id").isNotNull()) \
        .filter(col("behavior").isin("pv", "buy", "cart", "fav")) \
        .withColumn("behavior_date", substring(col("ts"), 1, 10)) \
        .withColumn("behavior_hour", hour(to_timestamp(col("ts")))) \
        .withColumn("behavior_minute", minute(to_timestamp(col("ts")))) \
        .withColumn("is_weekend",
            when(dayofweek(to_date(substring(col("ts"), 1, 10))).isin(1, 7), 1)
            .otherwise(0))

    dwd.write.mode("overwrite").insertInto("dwd.dwd_user_log", overwrite=True)
    print(f"[DWD] 用户行为日志完成，行数: {dwd.count()}")

if __name__ == "__main__":
    etl_dwd_order_detail()
    etl_dwd_user_log()
    spark.stop()
```

```sql
-- ===== ADS-日交易统计 =====
INSERT OVERWRITE TABLE ads.ads_trade_stats_day PARTITION (dt='${yesterday}')
SELECT
    a.order_count,
    a.order_amount,
    a.user_count,
    a.avg_order_amount,
    a.max_order_amount,
    a.cancel_rate,
    COALESCE(b.pv, 0) AS pv,
    COALESCE(b.uv, 0) AS uv,
    COALESCE(a.payment_amount, 0) AS gmv
FROM (
    SELECT
        COUNT(DISTINCT order_id) AS order_count,
        SUM(total_amount) AS order_amount,
        SUM(payment_amount) AS payment_amount,
        COUNT(DISTINCT user_id) AS user_count,
        ROUND(SUM(total_amount) / COUNT(DISTINCT order_id), 2) AS avg_order_amount,
        MAX(total_amount) AS max_order_amount,
        ROUND(SUM(CASE WHEN order_status = '已取消' THEN 1 ELSE 0 END) * 1.0 / COUNT(*), 4) AS cancel_rate
    FROM dwd.dwd_order_detail WHERE dt = '${yesterday}'
) a
LEFT JOIN (
    SELECT
        COUNT(*) AS pv,
        COUNT(DISTINCT user_id) AS uv
    FROM dwd.dwd_user_log WHERE dt = '${yesterday}'
) b ON 1=1;


-- ===== ADS-商品销量TOP10 =====
INSERT OVERWRITE TABLE ads.ads_sku_sales_top10 PARTITION (dt='${yesterday}')
SELECT sku_id, sku_name, category_name, order_count, order_amount, rank_num
FROM (
    SELECT
        sku_id,
        MAX(sku_name) AS sku_name,
        MAX(category_name) AS category_name,
        SUM(sku_num) AS order_count,
        SUM(order_price * sku_num) AS order_amount,
        ROW_NUMBER() OVER (ORDER BY SUM(sku_num) DESC) AS rank_num
    FROM dwd.dwd_order_detail
    WHERE dt = '${yesterday}'
    GROUP BY sku_id
) t
WHERE rank_num <= 10;


-- ===== ADS-行为转化漏斗 =====
INSERT OVERWRITE TABLE ads.ads_user_action_conversion PARTITION (dt='${yesterday}')
SELECT
    pv_users,
    cart_users,
    fav_users,
    buy_users,
    ROUND(cart_users * 1.0 / NULLIF(pv_users, 0), 4) AS pv_to_cart_rate,
    ROUND(buy_users * 1.0 / NULLIF(cart_users, 0), 4) AS cart_to_buy_rate,
    ROUND(buy_users * 1.0 / NULLIF(pv_users, 0), 4) AS overall_conversion
FROM (
    SELECT
        COUNT(DISTINCT CASE WHEN pv_count > 0 THEN user_id END) AS pv_users,
        COUNT(DISTINCT CASE WHEN cart_count > 0 THEN user_id END) AS cart_users,
        COUNT(DISTINCT CASE WHEN fav_count > 0 THEN user_id END) AS fav_users,
```

## 九、数据质量检查

```python
"""
dq_check.py — 数据质量检查脚本
"""
def check_row_count(table, partition, expected_min):
    """行数检查：确保数据量在合理范围"""
    result = spark.sql(f"SELECT COUNT(*) as cnt FROM {table} WHERE dt='{partition}'")
    cnt = result.collect()[0]['cnt']
    assert cnt >= expected_min, f"{table}: 行数 {cnt} < 预期最小值 {expected_min}"
    print(f"[DQ] {table}: 行数 {cnt} ✓")

def check_null_rate(table, partition, column, max_null_rate):
    """空值检查"""
    total = spark.sql(f"SELECT COUNT(*) as cnt FROM {table} WHERE dt='{partition}'").collect()[0]['cnt']
    nulls = spark.sql(
        f"SELECT COUNT(*) as cnt FROM {table} WHERE dt='{partition}' AND {column} IS NULL"
    ).collect()[0]['cnt']
    null_rate = nulls / total if total > 0 else 0
    assert null_rate <= max_null_rate, f"{table}.{column}: 空值率 {null_rate:.2%}"
    print(f"[DQ] {table}.{column}: 空值率 {null_rate:.2%} ✓")

def check_duplicate(table, partition, keys):
    """重复检查"""
    dup = spark.sql(f"""
        SELECT {keys}, COUNT(*) as cnt 
        FROM {table} WHERE dt='{partition}'
        GROUP BY {keys} HAVING COUNT(*) > 1
    """).count()
    assert dup == 0, f"{table}: 发现 {dup} 条重复数据"
    print(f"[DQ] {table}: 无重复 ✓")

# 执行检查
check_row_count("ods.ods_order_info", YESTERDAY, 100)
check_row_count("dwd.dwd_order_detail", YESTERDAY, 100)
check_null_rate("dwd.dwd_order_detail", YESTERDAY, "user_id", 0.05)
check_duplicate("dwd.dwd_order_detail", YESTERDAY, "detail_id")
```
