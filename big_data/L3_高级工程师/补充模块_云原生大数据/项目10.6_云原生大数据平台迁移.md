## 一、项目描述

将L1-L2阶段基于Docker Compose构建的大数据平台迁移到Kubernetes，实现云原生部署。项目涵盖K8s集群搭建、存储层迁移、计算引擎迁移、服务层迁移、监控运维全流程，完成从"容器化"到"云原生"的架构升级。

### 迁移价值

```
从Docker Compose到Kubernetes的演进价值：

  痛点1: 单点故障无法自愈
    → Docker Compose: 容器挂了需要手动重启
    → K8s: 自动检测故障并重启Pod，保证SLA

  痛点2: 资源无法弹性伸缩
    → Docker Compose: 固定资源分配，闲时浪费、忙时不足
    → K8s: HPA/VPA自动扩缩容，按需使用资源

  痛点3: 配置管理混乱
    → Docker Compose: 环境变量散落在docker-compose.yml和.env
    → K8s: ConfigMap/Secret统一管理，环境隔离

  痛点4: 滚动升级困难
    → Docker Compose: docker-compose up -d 全量重启
    → K8s: RollingUpdate零停机升级，一键回滚

  痛点5: 多租户隔离缺失
    → Docker Compose: 所有服务共享网络和资源
    → K8s: Namespace + ResourceQuota + NetworkPolicy多租户隔离

  痛点6: 可观测性不足
    → Docker Compose: 依赖docker logs手动排查
    → K8s: Prometheus + Grafana + Loki全链路可观测
```

#### 4.1.3 迁移HDFS数据到MinIO

```python
# hdfs_to_minio_migration.py
import os
from hdfs import InsecureClient
from minio import Minio

hdfs_client = InsecureClient("http://namenode:9870", user="hdfs")
minio_client = Minio(
    "minio.data-platform.svc.cluster.local:9000",
    access_key="admin",
    secret_key="admin123456",
    secure=False
)

MIGRATION_MAP = {
    "/warehouse/ods": "warehouse/ods",
    "/warehouse/dwd": "warehouse/dwd",
    "/warehouse/dws": "warehouse/dws",
    "/warehouse/ads": "warehouse/ads",
    "/data/iceberg": "iceberg",
}

LOCAL_TMP = "/tmp/hdfs_migration"

def migrate_directory(hdfs_path, minio_prefix):
    files = hdfs_client.list(hdfs_path, recursive=True)
    migrated = 0
    for f in files:
        if f["type"] == "FILE":
            local_path = os.path.join(LOCAL_TMP, f["name"])
            hdfs_client.download(f["path"], local_path)
            object_name = os.path.join(minio_prefix, f["path"].replace(hdfs_path, "").lstrip("/"))
            bucket = minio_prefix.split("/")[0]
            minio_client.fput_object(bucket, object_name, local_path)
            os.remove(local_path)
            migrated += 1
            if migrated % 100 == 0:
                print(f"已迁移 {migrated} 个文件: {hdfs_path} → {minio_prefix}")
    print(f"完成: {hdfs_path} → {minio_prefix}, 共 {migrated} 个文件")

for hdfs_path, minio_prefix in MIGRATION_MAP.items():
    bucket = minio_prefix.split("/")[0]
    if not minio_client.bucket_exists(bucket):
        minio_client.make_bucket(bucket)
    migrate_directory(hdfs_path, minio_prefix)

print("HDFS → MinIO 迁移完成")
```

```python
# kafka_migration.py
from kafka import KafkaAdminClient, KafkaConsumer
from kafka.admin import NewTopic
import json

SOURCE_BOOTSTRAP = "localhost:9092"
TARGET_BOOTSTRAP = "kafka-cluster-kafka-bootstrap.kafka.svc.cluster.local:9092"

source_admin = KafkaAdminClient(
    bootstrap_servers=SOURCE_BOOTSTRAP,
    client_id="migration-source"
)

target_admin = KafkaAdminClient(
    bootstrap_servers=TARGET_BOOTSTRAP,
    client_id="migration-target"
)

source_topics = source_admin.list_topics()
target_topics = target_admin.list_topics()

topics_to_migrate = []
for topic in source_topics:
    if not topic.startswith("__"):
        config = source_admin.describe_configs(
            resources=[("topic", topic)]
        )
        partitions = len(source_admin.describe_topic(topic)["partitions"])
        rf = source_admin.describe_topic(topic)["partitions"][0]["replicas"]
        topics_to_migrate.append(NewTopic(
            name=topic,
            num_partitions=partitions,
            replication_factor=len(rf)
        ))
        print(f"待迁移Topic: {topic}, Partitions={partitions}, RF={len(rf)}")

new_topics = [t for t in topics_to_migrate if t.name not in target_topics]
if new_topics:
    target_admin.create_topics(new_topics)
    print(f"已创建 {len(new_topics)} 个Topic")

source_admin.close()
target_admin.close()

print("Kafka Topic迁移完成")
print(f"源集群Topic数: {len(source_topics)}")
print(f"目标集群Topic数: {len(target_topics) + len(new_topics)}")
```

```yaml
# rolling-upgrade-sop.yaml
#
# 滚动升级标准操作流程
#
# 原则:
#   1. 先升级无状态服务，再升级有状态服务
#   2. 有状态服务先做数据备份/Savepoint
#   3. 逐个Pod升级，验证后再继续
#   4. 保留回滚能力

# 步骤1: 升级Trino（无状态，风险最低）
# kubectl set image deployment/trino-coordinator \
#   trino=trinodb/trino:436 -n data-platform
# kubectl rollout status deployment/trino-coordinator -n data-platform
# kubectl set image deployment/trino-worker \
#   trino=trinodb/trino:436 -n data-platform
# kubectl rollout status deployment/trino-worker -n data-platform

# 步骤2: 升级Spark Operator
# helm upgrade spark-operator spark-operator/spark-operator \
#   --namespace data-app \
#   --set image.tag=v1beta2-1.4.3-3.5.0

# 步骤3: 升级Flink（需要Savepoint）
# kubectl annotate flinkdeployment realtime-trade-monitor \
#   savepoint.triggerNonce=1 -n data-app
# 等待Savepoint完成
# kubectl patch flinkdeployment realtime-trade-monitor -n data-app \
#   --type merge -p '{"spec":{"image":"flink:1.19.0-java8"}}'

# 步骤4: 升级Kafka（Strimzi滚动升级）
# kubectl patch kafka kafka-cluster -n kafka --type merge \
#   -p '{"spec":{"kafka":{"version":"3.7.0","image":"quay.io/strimzi/kafka:latest-kafka-3.7.0"}}}'
# Strimzi会自动滚动升级每个Broker

# 步骤5: 升级MinIO（有状态，最谨慎）
# helm upgrade minio minio/minio -n data-platform \
#   -f values-minio.yaml \
#   --set image.tag=RELEASE.2024-02-06T01-26-11Z

# 回滚操作
```

#### 7.3.3 故障排查手册

```
故障排查手册
=============

1. Pod无法启动
   症状: Pod状态为ImagePullBackOff / CrashLoopBackOff
   排查:
     kubectl describe pod <pod-name> -n <namespace>
     kubectl logs <pod-name> -n <namespace> --previous
   常见原因:
     - 镜像不存在或网络不通 → 检查image和imagePullPolicy
     - 资源不足 → kubectl describe node <node> | grep -A5 "Allocated"
     - ConfigMap/Secret缺失 → kubectl get configmap,secret -n <namespace>
     - PVC Pending → kubectl describe pvc <pvc-name>

2. Spark任务失败
   症状: SparkApplication状态为FAILED
   排查:
     kubectl -n data-app describe sparkapplication <name>
     kubectl -n data-app logs <name>-driver
   常见原因:
     - OOM → 增大executor.memory或driver.memory
     - 镜像缺少依赖 → 检查Dockerfile中ADD的jar包
     - MinIO连接失败 → 检查s3a endpoint和access key
     - 动态分配超时 → 检查shuffleTracking.enabled

3. Flink Checkpoint失败
   症状: Checkpoint持续失败或超时
   排查:
     kubectl -n data-app logs <flink-pod> | grep -i checkpoint
     检查MinIO中checkpoint目录是否可写
   常见原因:
     - MinIO不可达 → 检查s3 endpoint配置
     - Checkpoint超时 → 增大execution.checkpointing.timeout
     - 状态过大 → 考虑RocksDB增量Checkpoint

4. Kafka消费延迟
   症状: Consumer Lag持续增长
   排查:
     kubectl -n kafka exec -it kafka-cluster-kafka-0 -- \
       bin/kafka-consumer-groups.sh --bootstrap-server localhost:9092 \
       --describe --group <group-id>
   常见原因:
     - Consumer处理慢 → 增加Consumer实例或优化逻辑
     - 分区不均衡 → 检查分区分配策略
     - Broker磁盘IO高 → 检查磁盘使用和IO等待

5. PVC磁盘不足
   症状: Pod写入失败，日志报No space left
   排查:
     kubectl get pvc -A
     kubectl describe pvc <pvc-name>
   解决:
     - 扩容PVC: kubectl edit pvc <pvc-name> 修改spec.resources.requests.storage
     - 清理旧数据: 根据业务TTL策略清理过期数据
     - 增加新节点: 扩展集群存储容量

6. 网络策略问题
   症状: 跨命名空间服务无法通信
   排查:
     kubectl get networkpolicy -A
     kubectl exec -it <pod> -- curl -v <service>.<namespace>.svc.cluster.local:<port>
   解决:
     - 检查NetworkPolicy是否允许跨命名空间流量
     - 临时禁用NetworkPolicy进行验证
     - 添加允许规则

7. 节点NotReady
   症状: kubectl get nodes显示某节点NotReady
   排查:
     kubectl describe node <node-name>
     journalctl -u kubelet | tail -50
   常见原因:
     - kubelet服务异常 → systemctl restart kubelet
     - 磁盘满 → 清理Docker镜像和日志
     - 内存不足 → 检查节点上Pod资源使用
```

## 八、迁移脚本（Docker Compose → K8s）

```python
# compose_to_k8s_migrator.py
import yaml
import os
import re

COMPOSE_FILE = "docker-compose.yml"
OUTPUT_DIR = "k8s-manifests"

SERVICE_MIGRATION_MAP = {
    "minio": {"namespace": "data-platform", "type": "helm", "chart": "minio/minio"},
    "hive-metastore": {"namespace": "data-platform", "type": "deployment"},
    "spark-master": {"namespace": "data-app", "type": "spark-operator"},
    "spark-worker": {"namespace": "data-app", "type": "spark-operator"},
    "flink-jobmanager": {"namespace": "data-app", "type": "flink-deployment"},
    "flink-taskmanager": {"namespace": "data-app", "type": "flink-deployment"},
    "kafka": {"namespace": "kafka", "type": "strimzi"},
    "trino-coordinator": {"namespace": "data-platform", "type": "deployment"},
    "trino-worker": {"namespace": "data-platform", "type": "deployment"},
    "airflow-webserver": {"namespace": "data-app", "type": "helm", "chart": "apache-airflow/airflow"},
    "airflow-scheduler": {"namespace": "data-app", "type": "helm", "chart": "apache-airflow/airflow"},
    "clickhouse": {"namespace": "data-platform", "type": "statefulset"},
    "mysql": {"namespace": "data-platform", "type": "statefulset"},
    "grafana": {"namespace": "monitoring", "type": "helm", "chart": "prometheus-community/kube-prometheus-stack"},
    "prometheus": {"namespace": "monitoring", "type": "helm", "chart": "prometheus-community/kube-prometheus-stack"},
}

def parse_compose_env(env_list):
    env_dict = {}
    if env_list:
        for item in env_list:
            if isinstance(item, dict):
                for k, v in item.items():
                    env_dict[k] = str(v)
            elif isinstance(item, str) and "=" in item:
                k, v = item.split("=", 1)
                env_dict[k] = v
    return env_dict

def parse_resource_limit(limit_str):
    if not limit_str:
        return {"cpu": "500m", "memory": "1Gi"}
    result = {}
    if "cpus" in limit_str:
        result["cpu"] = str(limit_str["cpus"])
    if "memory" in limit_str:
        mem = limit_str["memory"]
        if mem.endswith("m"):
            result["memory"] = mem.replace("m", "Mi")
        elif mem.endswith("g"):
            result["memory"] = mem.replace("g", "Gi")
    return result

def generate_deployment(service_name, service_config, namespace):
    env_dict = parse_compose_env(service_config.get("environment", []))
    ports = service_config.get("ports", [])
    container_ports = []
    for p in ports:
        if isinstance(p, str):
            target = int(p.split(":")[-1])
            container_ports.append({"containerPort": target})
        elif isinstance(p, dict):
            container_ports.append({"containerPort": p.get("target_port", p.get("target", 80))})

    resources = parse_resource_limit(service_config.get("deploy", {}).get("resources", {}).get("limits", {}))

    env_list = [{"name": k, "value": v} for k, v in env_dict.items()]

    manifest = {
        "apiVersion": "apps/v1",
        "kind": "Deployment",
        "metadata": {
            "name": service_name,
            "namespace": namespace,
            "labels": {"app": service_name}
        },
        "spec": {
            "replicas": service_config.get("deploy", {}).get("replicas", 1),
            "selector": {
                "matchLabels": {"app": service_name}
            },
            "template": {
                "metadata": {
                    "labels": {"app": service_name}
                },
                "spec": {
                    "containers": [{
                        "name": service_name,
                        "image": service_config.get("image", f"{service_name}:latest"),
                        "ports": container_ports,
                        "env": env_list,
                        "resources": {
                            "requests": resources,
                            "limits": resources
                        }
                    }]
                }
            }
        }
    }
    return manifest

def generate_statefulset(service_name, service_config, namespace):
    deployment = generate_deployment(service_name, service_config, namespace)
    deployment["kind"] = "StatefulSet"
    deployment["spec"]["serviceName"] = service_name
    deployment["spec"]["volumeClaimTemplates"] = [{
        "metadata": {"name": "data"},
        "spec": {
            "accessModes": ["ReadWriteOnce"],
            "storageClassName": "local-path",
            "resources": {"requests": {"storage": "20Gi"}}
        }
    }]
    container = deployment["spec"]["template"]["spec"]["containers"][0]
    container["volumeMounts"] = [{"name": "data", "mountPath": "/data"}]
    return deployment

def generate_service(service_name, service_config, namespace):
    ports = service_config.get("ports", [])
    svc_ports = []
    for p in ports:
        if isinstance(p, str):
            parts = p.split(":")
            port = int(parts[-1])
            svc_ports.append({"port": port, "targetPort": port})
        elif isinstance(p, dict):
            port = p.get("published", p.get("target", 80))
            target = p.get("target_port", p.get("target", 80))
            svc_ports.append({"port": port, "targetPort": target})

    return {
        "apiVersion": "v1",
        "kind": "Service",
        "metadata": {
            "name": service_name,
            "namespace": namespace
        },
        "spec": {
            "selector": {"app": service_name},
            "ports": svc_ports,
            "type": "ClusterIP"
        }
    }

def migrate():
    os.makedirs(OUTPUT_DIR, exist_ok=True)

    with open(COMPOSE_FILE, "r") as f:
        compose = yaml.safe_load(f)

    services = compose.get("services", {})
    migration_report = []

    for svc_name, svc_config in services.items():
        migration_info = SERVICE_MIGRATION_MAP.get(svc_name, {
            "namespace": "default", "type": "deployment"
        })
        ns = migration_info["namespace"]
        svc_type = migration_info["type"]

        ns_dir = os.path.join(OUTPUT_DIR, ns)
        os.makedirs(ns_dir, exist_ok=True)

        if svc_type == "helm":
            migration_report.append({
                "service": svc_name,
                "type": "Helm Chart",
                "chart": migration_info.get("chart", ""),
                "namespace": ns,
                "note": "使用Helm部署，需手动编写values.yaml"
            })
            continue

        if svc_type == "strimzi":
            migration_report.append({
                "service": svc_name,
                "type": "Strimzi Kafka",
                "namespace": ns,
                "note": "使用Strimzi Operator部署Kafka集群"
            })
            continue

        if svc_type == "spark-operator":
            migration_report.append({
                "service": svc_name,
                "type": "Spark Operator",
                "namespace": ns,
                "note": "迁移为SparkApplication CRD"
            })
            continue

        if svc_type == "flink-deployment":
            migration_report.append({
                "service": svc_name,
                "type": "Flink Operator",
                "namespace": ns,
                "note": "迁移为FlinkDeployment CRD"
            })
            continue

        manifests = []
        if svc_type == "statefulset":
            manifests.append(generate_statefulset(svc_name, svc_config, ns))
        else:
            manifests.append(generate_deployment(svc_name, svc_config, ns))
        manifests.append(generate_service(svc_name, svc_config, ns))

        output_file = os.path.join(ns_dir, f"{svc_name}.yaml")
        with open(output_file, "w") as f:
            yaml.dump_all(manifests, f, default_flow_style=False)

        migration_report.append({
            "service": svc_name,
            "type": svc_type,
            "namespace": ns,
            "output": output_file
        })

    report_file = os.path.join(OUTPUT_DIR, "migration_report.yaml")
    with open(report_file, "w") as f:
        yaml.dump(migration_report, f, default_flow_style=False, allow_unicode=True)

    print(f"迁移报告已生成: {report_file}")
    print(f"共处理 {len(services)} 个服务")
    for r in migration_report:
        print(f"  {r['service']}: {r['type']} → {r['namespace']}")

if __name__ == "__main__":
    migrate()
```
