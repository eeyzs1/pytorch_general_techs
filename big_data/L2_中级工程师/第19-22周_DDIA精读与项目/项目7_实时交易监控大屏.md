# 项目7：实时交易监控大屏

> **所属阶段**：L2 中级工程师 | **预计时长**：40h | **难度**：★★★★☆
>
> **技术栈**：MySQL → Debezium + Kafka → Flink → ClickHouse → Grafana

---

## 一、项目描述

构建从CDC采集到实时大屏展示的端到端实时数据处理管道。模拟电商公司的实时交易监控需求，搭建完整的生产级数据流水线。

### 业务价值

```
实时交易监控大屏解决的问题：

  问题1: "今天的GMV是多少？"
    → 以前: 等T+1报表，第二天才知道
    → 现在: 实时大屏，秒级更新

  问题2: "大促期间有没有异常？"
    → 以前: 事后复盘发现异常
    → 现在: 实时告警，立即响应

  问题3: "各品类的销售趋势？"
    → 以前: Excel手工出图，每周一次
    → 现在: 实时趋势图，自动刷新
```

### 第2层：流式ETL层（Flink）

**引擎**：Apache Flink

**任务定义**：

```yaml
任务1 - 订单宽表生成:
  输入: mysql.ecommerce.orders, mysql.ecommerce.users
  逻辑: 订单JOIN用户信息, 生成宽表
  输出: Kafka Topic 'order-wide-table'
  延迟: < 3秒

任务2 - 实时统计指标:
  输入: order-wide-table
  逻辑: 
    - 每分钟交易额(按品类分组)
    - 每分钟订单数
    - 实时UV(独立用户数)
  输出: ClickHouse 'trade_stats_minute'
  窗口: 1分钟滚动窗口

任务3 - 异常检测:
  输入: order-wide-table
  规则:
    - 单用户1分钟下单>3次 → 高频交易告警
    - 单笔金额>10000 → 大额交易告警
  输出: Kafka Topic 'fraud-alerts'
```

**Flink作业Java代码（完整）**：

```java
package com.example.realtimetrade;

import org.apache.flink.api.common.eventtime.WatermarkStrategy;
import org.apache.flink.api.common.functions.AggregateFunction;
import org.apache.flink.api.common.serialization.SimpleStringSchema;
import org.apache.flink.api.java.tuple.Tuple2;
import org.apache.flink.streaming.api.datastream.DataStream;
import org.apache.flink.streaming.api.environment.StreamExecutionEnvironment;
import org.apache.flink.streaming.api.windowing.assigners.TumblingEventTimeWindows;
import org.apache.flink.streaming.api.windowing.time.Time;
import org.apache.flink.streaming.connectors.kafka.FlinkKafkaConsumer;
import org.apache.flink.streaming.connectors.kafka.FlinkKafkaProducer;
import org.apache.flink.table.api.bridge.java.StreamTableEnvironment;
import com.alibaba.fastjson.JSON;
import com.alibaba.fastjson.JSONObject;

import java.time.Duration;
import java.util.HashSet;
import java.util.Properties;

public class RealTimeTradeMonitor {

    public static void main(String[] args) throws Exception {
        StreamExecutionEnvironment env = 
            StreamExecutionEnvironment.getExecutionEnvironment();
        StreamTableEnvironment tEnv = StreamTableEnvironment.create(env);

        // Checkpoint配置（Exactly-Once保证）
        env.enableCheckpointing(60000);
        env.getCheckpointConfig().setCheckpointTimeout(120000);
        env.setStateBackend(new EmbeddedRocksDBStateBackend(true));
        env.getCheckpointConfig().setCheckpointStorage(
            "file:///tmp/flink/checkpoints/trade-monitor");

        // ====== Kafka Consumer配置 ======
        Properties kafkaProps = new Properties();
        kafkaProps.setProperty("bootstrap.servers", "localhost:9092");
        kafkaProps.setProperty("group.id", "trade-monitor-group");
        kafkaProps.setProperty("auto.offset.reset", "earliest");

        // ====== 源1: 订单CDC流 ======
        FlinkKafkaConsumer<String> orderSource = new FlinkKafkaConsumer<>(
            "mysql.ecommerce.orders",
            new SimpleStringSchema(),
            kafkaProps
        );

        DataStream<OrderEvent> orderStream = env.addSource(orderSource)
            .map(line -> parseOrderEvent(JSON.parseObject(line)))
            .assignTimestampsAndWatermarks(
                WatermarkStrategy
                    .<OrderEvent>forBoundedOutOfOrderness(Duration.ofSeconds(5))
                    .withTimestampAssigner((event, ts) -> event.createTime)
            );

        // ====== 源2: 用户CDC流 ======
        FlinkKafkaConsumer<String> userSource = new FlinkKafkaConsumer<>(
            "mysql.ecommerce.users",
            new SimpleStringSchema(),
            kafkaProps
        );

        DataStream<UserInfo> userStream = env.addSource(userSource)
            .map(line -> parseUserInfo(JSON.parseObject(line)));

        // ====== 任务1: 订单宽表生成 ======
        DataStream<OrderWide> orderWideStream = orderStream
            .keyBy(OrderEvent::getUserId)
            .connect(userStream.keyBy(UserInfo::getUserId))
            .process(new OrderUserJoinFunction())
            .uid("order-user-join");

        // ====== 宽表写入Kafka ======
        FlinkKafkaProducer<String> wideTableSink = new FlinkKafkaProducer<>(
            "order-wide-table",
            new SimpleStringSchema(),
            kafkaProps
        );

        orderWideStream
            .map(wide -> JSON.toJSONString(wide))
            .addSink(wideTableSink)
            .uid("wide-table-sink");

        // ====== 任务2: 实时统计指标（1分钟窗口）=====
        DataStream<TradeStats> tradeStatsStream = orderWideStream
            .keyBy(OrderWide::getCategory)
            .window(TumblingEventTimeWindows.of(Time.minutes(1)))
            .aggregate(new TradeStatsAggregator(),
                (key, window, input, out) -> {
                    TradeStats stats = input.iterator().next();
                    stats.setWindowStart(window.getStart());
                    stats.setWindowEnd(window.getEnd());
                    stats.setCategory(key);
                    out.collect(stats);
                })
            .uid("trade-stats-window");

        // ====== 统计结果写入ClickHouse ======
        tradeStatsStream.addSink(new ClickHouseTradeStatsSink())
            .uid("clickhouse-sink");

        // ====== 任务3: 异常检测 ======
        // 高频交易检测（1分钟窗口）
        DataStream<FraudAlert> highFreqAlerts = orderWideStream
            .keyBy(OrderWide::getUserId)
            .window(TumblingEventTimeWindows.of(Time.minutes(1)))
            .aggregate(new HighFrequencyDetector())
            .filter(alert -> alert.getOrderCount() > 3)
            .map(alert -> {
                alert.setAlertType("HIGH_FREQUENCY");
                return alert;
            })
            .uid("high-freq-detector");

        // 大额交易检测
        DataStream<FraudAlert> largeAmountAlerts = orderWideStream
            .filter(order -> order.getAmount().compareTo(
                new java.math.BigDecimal("10000")) > 0)
            .map(order -> {
                FraudAlert alert = new FraudAlert();
                alert.setUserId(order.getUserId());
                alert.setOrderId(order.getOrderId());
                alert.setAmount(order.getAmount());
                alert.setAlertType("LARGE_AMOUNT");
                alert.setAlertTime(System.currentTimeMillis());
                alert.setDetail("订单金额超过10000元");
                return alert;
            })
            .uid("large-amount-detector");

        // 告警写入Kafka
        FlinkKafkaProducer<String> alertSink = new FlinkKafkaProducer<>(
            "fraud-alerts",
            new SimpleStringSchema(),
            kafkaProps
        );

        highFreqAlerts
            .union(largeAmountAlerts)
            .map(alert -> JSON.toJSONString(alert))
            .addSink(alertSink)
            .uid("alert-sink");

        env.execute("Real-Time Trade Monitor");
    }

    // ====== 数据模型 ======
    
    static class OrderEvent {
        private Long orderId;
        private Long userId;
        private Long productId;
        private String category;
        private java.math.BigDecimal amount;
        private String status;
        private Long createTime;
        
        // getters and setters...
    }

    static class UserInfo {
        private Long userId;
        private String username;
        private String city;
        private Integer age;
        // getters and setters...
    }

    static class OrderWide {
        private Long orderId;
        private Long userId;
        private String username;
        private String city;
        private String category;
        private java.math.BigDecimal amount;
        private String status;
        private Long createTime;
        // getters and setters...
    }

    static class TradeStats {
        private String category;
        private Long windowStart;
        private Long windowEnd;
        private Long orderCount;
        private java.math.BigDecimal totalAmount;
        private Long userCount;
        // getters and setters...
    }

    static class FraudAlert {
        private Long userId;
        private Long orderId;
        private java.math.BigDecimal amount;
        private Long orderCount;
        private String alertType;
        private Long alertTime;
        private String detail;
        // getters and setters...
    }
}
```

### 第3层：OLAP存储层（ClickHouse）

**ClickHouse建表DDL**：

```sql
-- =====================================================
-- ClickHouse 表设计
-- 原则：宽表 + ReplicatedMergeTree + 按天分区 + TTL
-- =====================================================

-- 1. 订单宽表（实时写入）
CREATE TABLE order_wide_realtime ON CLUSTER default (
    order_id UInt64,
    user_id UInt64,
    username String,
    city String,
    age UInt8,
    product_id UInt64,
    product_name String,
    category String,
    amount Decimal(10, 2),
    status String,
    create_time DateTime64(3),
    build_time DateTime64(3) DEFAULT now64(),
    INDEX idx_user user_id TYPE bloom_filter GRANULARITY 1,
    INDEX idx_category category TYPE set(0) GRANULARITY 1
) ENGINE = ReplicatedMergeTree(
    '/clickhouse/tables/{shard}/order_wide_realtime',
    '{replica}'
)
PARTITION BY toYYYYMMDD(create_time)
ORDER BY (create_time, order_id)
TTL create_time + INTERVAL 72 HOUR
SETTINGS index_granularity = 8192;

-- 2. 分钟级交易统计
CREATE TABLE trade_stats_minute ON CLUSTER default (
    window_start DateTime64(3),
    window_end DateTime64(3),
    category String,
    city String,
    order_count UInt64,
    total_amount Decimal(20, 2),
    avg_amount Decimal(10, 2),
    user_count UInt64
) ENGINE = ReplicatedSummingMergeTree(
    '/clickhouse/tables/{shard}/trade_stats_minute',
    '{replica}',
    (order_count, total_amount, user_count)
)
PARTITION BY toYYYYMM(window_start)
ORDER BY (window_start, category, city)
TTL window_start + INTERVAL 30 DAY
SETTINGS index_granularity = 8192;

-- 3. 实时总览（物化视图，自动聚合）
CREATE MATERIALIZED VIEW trade_overview_realtime ON CLUSTER default
ENGINE = SummingMergeTree()
PARTITION BY toYYYYMM(window_start)
ORDER BY (window_start, category)
AS SELECT
    window_start,
    category,
    count() AS order_count,
    sum(amount) AS total_amount,
    uniq(user_id) AS user_count
FROM order_wide_realtime
GROUP BY window_start, category;

-- 4. 异常告警记录表
CREATE TABLE fraud_alerts ON CLUSTER default (
    user_id UInt64,
    alert_type String,
    alert_time DateTime64(3),
    order_count UInt64,
    amount Decimal(20, 2),
    detail String,
    create_time DateTime64(3) DEFAULT now64()
) ENGINE = MergeTree()
PARTITION BY toYYYYMM(alert_time)
ORDER BY (alert_time, alert_type, user_id)
TTL alert_time + INTERVAL 7 DAY;

-- 5. 端到端延迟监控表
CREATE TABLE pipeline_latency ON CLUSTER default (
    event_time DateTime64(3),
    source_type String,
    stage String,
    latency_ms UInt64,
    record_count UInt64
) ENGINE = MergeTree()
PARTITION BY toYYYYMM(event_time)
ORDER BY (event_time, source_type, stage);
```

## 四、数据生成器

```python
"""
模拟交易数据生成器
持续生成交易数据到MySQL，触发CDC采集
"""
import pymysql
import random
import time
import threading
from datetime import datetime

DB_CONFIG = {
    'host': 'localhost',
    'port': 3306,
    'user': 'app_user',
    'password': 'password',
    'database': 'ecommerce',
    'charset': 'utf8mb4'
}

CATEGORIES = ['电子', '服装', '食品', '图书', '家居', '美妆', '运动', '母婴']
CITIES = ['北京', '上海', '广州', '深圳', '杭州', '成都', '武汉', '南京']
PRODUCTS = list(range(1001, 2001))
USERS = list(range(1, 10001))


class OrderGenerator:
    def __init__(self, rate_per_second=50):
        self.rate = rate_per_second
        self.conn = pymysql.connect(**DB_CONFIG)
        self.cursor = self.conn.cursor()
        self.order_count = 0

    def generate_order(self):
        user_id = random.choice(USERS)
        product_id = random.choice(PRODUCTS)
        amount = round(random.uniform(10, 5000), 2)
        status = random.choices(
            ['created', 'paid', 'completed'],
            weights=[20, 30, 50]
        )[0]
        now = datetime.now()

        try:
            self.cursor.execute(
                """INSERT INTO orders (user_id, product_id, amount, status, create_time, update_time)
                   VALUES (%s, %s, %s, %s, %s, %s)""",
                (user_id, product_id, amount, status, now, now)
            )
            self.conn.commit()
            self.order_count += 1

            if self.order_count % 100 == 0:
                print(f"[{datetime.now()}] 已生成 {self.order_count} 条订单")

        except Exception as e:
            self.conn.rollback()
            print(f"生成订单失败: {e}")

    def run(self, duration_seconds=3600):
        end_time = time.time() + duration_seconds
        interval = 1.0 / self.rate

        print(f"开始生成交易数据: rate={self.rate}/s, duration={duration_seconds}s")

        while time.time() < end_time:
            self.generate_order()
            time.sleep(interval * random.uniform(0.5, 1.5))

        self.cursor.close()
        self.conn.close()
        print(f"数据生成完成: 共 {self.order_count} 条")


if __name__ == '__main__':
    generator = OrderGenerator(rate_per_second=100)
    generator.run(duration_seconds=3600)
```
