## 一、项目描述

将L1项目4构建的Hive离线数仓迁移到基于Apache Iceberg的湖仓架构，体验从传统数仓到现代湖仓的完整演进。项目涵盖环境搭建、数据迁移、湖仓特有能力验证、多引擎访问、性能对比全流程。

### 业务价值

```
从Hive数仓到Iceberg湖仓的演进价值：

  痛点1: 订单状态无法更新
    → Hive: 只能INSERT OVERWRITE整分区，延迟T+1
    → Iceberg: 行级UPDATE，实时反映订单状态变化

  痛点2: 加字段需要重建表
    → Hive: ALTER TABLE ADD COLUMN只对新分区生效
    → Iceberg: Schema演化，加删改列无需重写数据

  痛点3: 分区策略无法调整
    → Hive: 改分区需要重建整张表
    → Iceberg: 分区演化，从按天分区改为按月分区无需重写

  痛点4: 无法回溯历史数据
    → Hive: 数据被OVERWRITE后无法恢复
    → Iceberg: 时间旅行，查询任意历史快照

  痛点5: 只能用Spark/Hive查询
    → Hive: 绑定Hive生态
    → Iceberg: Spark/Flink/Trino/Presto多引擎访问
```

### 5.2 编写Spark迁移脚本

```python
"""
hive_to_iceberg_migration.py

Hive数仓 → Iceberg湖仓 全量数据迁移脚本
"""
from pyspark.sql import SparkSession
from pyspark.sql.functions import col, to_timestamp, to_date, year, month, dayofmonth, hour

spark = SparkSession.builder \
    .appName("Hive_to_Iceberg_Migration") \
    .config("spark.sql.catalog.iceberg", "org.apache.iceberg.spark.SparkCatalog") \
    .config("spark.sql.catalog.iceberg.type", "hive") \
    .config("spark.sql.catalog.iceberg.uri", "thrift://hive-metastore:9083") \
    .config("spark.sql.catalog.iceberg.warehouse", "s3a://warehouse/iceberg") \
    .config("spark.sql.extensions", "org.apache.iceberg.spark.extensions.IcebergSparkSessionExtensions") \
    .config("spark.hadoop.fs.s3a.endpoint", "http://minio:9000") \
    .config("spark.hadoop.fs.s3a.access.key", "admin") \
    .config("spark.hadoop.fs.s3a.secret.key", "admin123456") \
    .config("spark.hadoop.fs.s3a.path.style.access", "true") \
    .enableHiveSupport() \
    .getOrCreate()

MIGRATION_DATE = "2024-01-15"

def migrate_ods_layer():
    print("=== 迁移ODS层 ===")

    tables = [
        ("ods.ods_order_info", "iceberg_ods.ods_order_info"),
        ("ods.ods_order_detail", "iceberg_ods.ods_order_detail"),
        ("ods.ods_user_info", "iceberg_ods.ods_user_info"),
        ("ods.ods_sku_info", "iceberg_ods.ods_sku_info"),
        ("ods.ods_user_log", "iceberg_ods.ods_user_log"),
    ]

    for hive_table, iceberg_table in tables:
        print(f"迁移 {hive_table} → {iceberg_table}")
        try:
            hive_df = spark.table(hive_table).filter(col("dt") == MIGRATION_DATE)
            row_count = hive_df.count()
            print(f"  读取 {row_count} 行")

            hive_df.writeTo(iceberg_table).append()
            print(f"  写入完成")
        except Exception as e:
            print(f"  迁移失败: {e}")

def migrate_dwd_layer():
    print("=== 迁移DWD层 ===")

    print("构建DWD订单明细宽表...")
    ods_order_detail = spark.table("iceberg_ods.ods_order_detail")
    ods_order_info = spark.table("iceberg_ods.ods_order_info")
    ods_user_info = spark.table("iceberg_ods.ods_user_info")
    ods_sku_info = spark.table("iceberg_ods.ods_sku_info")

    dwd_order_detail = ods_order_detail \
        .join(ods_order_info.select("order_id", "user_id", "total_amount",
                                     "payment_amount", "order_status"),
              "order_id", "left") \
        .join(ods_user_info.select("user_id",
                                    col("username").alias("user_name"),
                                    col("gender").alias("user_gender"),
                                    col("user_level"),
                                    col("province").alias("user_province"),
                                    col("city").alias("user_city"),
                                    col("birthday")),
              "user_id", "left") \
        .join(ods_sku_info.select("sku_id", "category_id", "category_name", "brand"),
              "sku_id", "left") \
        .withColumn("user_age",
            year(spark.current_date()) - year(to_date(col("birthday")))) \
        .withColumn("order_year", year(col("create_time"))) \
        .withColumn("order_month", month(col("create_time"))) \
        .withColumn("order_day", dayofmonth(col("create_time"))) \
        .withColumn("order_hour", hour(col("create_time")))

    dwd_order_detail.writeTo("iceberg_dwd.dwd_order_detail").append()
    print(f"  DWD订单明细宽表写入完成，行数: {dwd_order_detail.count()}")

    print("构建DWD用户行为明细...")
    ods_user_log = spark.table("iceberg_ods.ods_user_log")

    dwd_user_log = ods_user_log \
        .filter(col("user_id").isNotNull()) \
        .filter(col("behavior").isin("pv", "buy", "cart", "fav")) \
        .withColumn("behavior_date", col("ts").cast("date").cast("string")) \
        .withColumn("behavior_hour", hour(col("ts"))) \
        .withColumn("is_weekend",
            when(dayofweek(to_date(col("behavior_date"))).isin(1, 7), 1)
            .otherwise(0))

    dwd_user_log.writeTo("iceberg_dwd.dwd_user_log").append()
    print(f"  DWD用户行为明细写入完成，行数: {dwd_user_log.count()}")

def migrate_dws_layer():
    print("=== 迁移DWS层 ===")

    print("构建DWS用户日行为汇总...")
    dwd_user_log = spark.table("iceberg_dwd.dwd_user_log")

    dws_user_action = dwd_user_log.groupBy("user_id") \
        .agg(
            spark.sql.functions.sum(spark.sql.functions.when(col("behavior") == "pv", 1).otherwise(0)).alias("pv_count"),
            spark.sql.functions.sum(spark.sql.functions.when(col("behavior") == "cart", 1).otherwise(0)).alias("cart_count"),
            spark.sql.functions.sum(spark.sql.functions.when(col("behavior") == "fav", 1).otherwise(0)).alias("fav_count"),
            spark.sql.functions.sum(spark.sql.functions.when(col("behavior") == "buy", 1).otherwise(0)).alias("buy_count"),
            spark.sql.functions.count("*").alias("total_actions"),
            spark.sql.functions.countDistinct("behavior_hour").alias("active_hours"),
            spark.sql.functions.lit(0).alias("is_new_user"),
            spark.sql.functions.lit(MIGRATION_DATE).cast("date").alias("stat_date")
        )

    dws_user_action.writeTo("iceberg_dws.dws_user_action_day").append()
    print(f"  DWS用户日行为汇总写入完成")

    print("构建DWS用户日订单汇总...")
    dwd_order_detail = spark.table("iceberg_dwd.dwd_order_detail")

    dws_trade_user = dwd_order_detail.groupBy("user_id") \
        .agg(
            spark.sql.functions.countDistinct("order_id").alias("order_count"),
            spark.sql.functions.sum("total_amount").alias("order_amount"),
            spark.sql.functions.sum("payment_amount").alias("payment_amount"),
            spark.sql.functions.sum("sku_num").alias("sku_count"),
            spark.sql.functions.sum(spark.sql.functions.when(col("order_status") == "已取消", 1).otherwise(0)).alias("cancel_count"),
            spark.sql.functions.sum(spark.sql.functions.when(col("order_status") == "已退款", 1).otherwise(0)).alias("refund_count"),
            spark.sql.functions.lit(MIGRATION_DATE).cast("date").alias("stat_date")
        )

    dws_trade_user.writeTo("iceberg_dws.dws_trade_user_order_day").append()
    print(f"  DWS用户日订单汇总写入完成")

def migrate_ads_layer():
    print("=== 迁移ADS层 ===")

    print("构建ADS日交易统计...")
    dwd_order_detail = spark.table("iceberg_dwd.dwd_order_detail")
    dwd_user_log = spark.table("iceberg_dwd.dwd_user_log")

    order_stats = dwd_order_detail.agg(
        spark.sql.functions.countDistinct("order_id").alias("order_count"),
        spark.sql.functions.sum("total_amount").alias("order_amount"),
        spark.sql.functions.sum("payment_amount").alias("gmv"),
        spark.sql.functions.countDistinct("user_id").alias("user_count"),
        spark.sql.functions.avg("total_amount").alias("avg_order_amount"),
        spark.sql.functions.max("total_amount").alias("max_order_amount"),
        spark.sql.functions.lit(0).alias("cancel_rate")
    )

    log_stats = dwd_user_log.agg(
        spark.sql.functions.count("*").alias("pv"),
        spark.sql.functions.countDistinct("user_id").alias("uv")
    )

    from pyspark.sql import functions as F
    ads_trade = order_stats.crossJoin(log_stats) \
        .withColumn("stat_date", F.lit(MIGRATION_DATE).cast("date"))

    ads_trade.writeTo("iceberg_ads.ads_trade_stats_day").append()
    print(f"  ADS日交易统计写入完成")

    print("构建ADS商品销量TOP10...")
    ads_sku_top10 = dwd_order_detail.groupBy("sku_id", "sku_name", "category_name") \
        .agg(
            spark.sql.functions.sum("sku_num").alias("order_count"),
            spark.sql.functions.sum("order_price" * col("sku_num")).alias("order_amount")
        ) \
        .withColumn("rank_num", spark.sql.functions.row_number().over(
            spark.sql.Window.orderBy(col("order_count").desc())
        )) \
        .filter(col("rank_num") <= 10) \
        .withColumn("stat_date", F.lit(MIGRATION_DATE).cast("date"))

    ads_sku_top10.writeTo("iceberg_ads.ads_sku_sales_top10").append()
    print(f"  ADS商品销量TOP10写入完成")

def verify_migration():
```

```sql
-- 场景: 订单状态从"待支付"变为"已支付"

-- Hive做法: 无法实现行级更新，只能INSERT OVERWRITE整分区
-- Iceberg做法: 直接UPDATE指定行

-- 更新单条订单状态
UPDATE iceberg_ods.ods_order_info
SET status = '已支付', pay_time = CURRENT_TIMESTAMP()
WHERE order_id = 100001;

-- 批量更新（所有超过7天未支付的订单标记为"已取消"）
UPDATE iceberg_ods.ods_order_info
SET status = '已取消'
WHERE status = '待支付'
  AND create_time < CURRENT_TIMESTAMP() - INTERVAL 7 DAYS;

-- MERGE INTO（增量Upsert）
MERGE INTO iceberg_ods.ods_order_info t
USING (
    SELECT order_id, user_id, status, total_amount,
           payment_amount, shipping_province, shipping_city,
           create_time, pay_time, finish_time
    FROM incremental_orders
) s
ON t.order_id = s.order_id
WHEN MATCHED AND s.status != t.status THEN
    UPDATE SET status = s.status, pay_time = s.pay_time, finish_time = s.finish_time
WHEN NOT MATCHED THEN INSERT *;
```

```python
"""
```

```python
import subprocess

def get_hdfs_dir_size(path):
    cmd = f"hdfs dfs -du -s {path}"
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    if result.returncode == 0:
        parts = result.stdout.strip().split()
        return int(parts[0]) if parts else 0
    return 0

def get_s3_dir_size(bucket, prefix):
    cmd = f"mc du myminio/{bucket}/{prefix}"
    result = subprocess.run(cmd, shell=True, capture_output=True, text=True)
    if result.returncode == 0:
        return result.stdout.strip()
    return "N/A"

def compare_storage():
    tables = [
        ("ods_order_info", "/warehouse/ods/ods_order_info", "warehouse/iceberg/iceberg_ods/ods_order_info"),
        ("ods_order_detail", "/warehouse/ods/ods_order_detail", "warehouse/iceberg/iceberg_ods/ods_order_detail"),
        ("dwd_order_detail", "/warehouse/dwd/dwd_order_detail", "warehouse/iceberg/iceberg_dwd/dwd_order_detail"),
        ("dws_user_action_day", "/warehouse/dws/dws_user_action_day", "warehouse/iceberg/iceberg_dws/dws_user_action_day"),
        ("ads_trade_stats_day", "/warehouse/ads/ads_trade_stats_day", "warehouse/iceberg/iceberg_ads/ads_trade_stats_day"),
    ]

    print(f"{'表名':<30} {'Hive(HDFS)':<20} {'Iceberg(S3)':<20}")
```

```yaml
    - 定期过期快照: 每天执行expire_snapshots，保留7天
```

### 8.3 常见问题与解决方案

```
Q1: 迁移后查询变慢了？
A: 检查以下几点:
   1. 是否有小文件问题 → 执行rewrite_data_files
   2. 是否缺少分区裁剪 → 检查分区字段和查询条件
   3. 是否数据文件过大 → 调整target-file-size-bytes
   4. 是否快照过多 → 执行expire_snapshots

Q2: Flink写入Iceberg产生大量小文件？
A: 解决方案:
   1. 增大Flink Checkpoint间隔（60s→120s）
   2. 设置write.target-file-size-bytes
   3. 定期执行rewrite_data_files合并小文件
   4. 使用RewriteDataFilesAction批量合并

Q3: Schema演化后下游Spark Job报错？
A: 原因和解决:
   1. 下游Job硬编码了列顺序 → 改用列名引用
   2. 删列后下游仍引用 → 使用RENAME替代DELETE
   3. 改类型后精度丢失 → 只做安全类型拓宽

Q4: 时间旅行查询很慢？
A: 原因和解决:
   1. 旧快照的数据文件可能已被清理 → 增大保留时间
   2. 增量查询扫描范围太大 → 缩小snapshot范围
   3. 快照元数据文件过多 → 定期做checkpoint

Q5: MinIO存储空间持续增长？
A: 原因和解决:
   1. 快照未过期 → 定期执行expire_snapshots
   2. 孤立文件未清理 → 定期执行remove_orphan_files
   3. 旧数据文件未删除 → 检查deletedFileRetentionDuration
```
