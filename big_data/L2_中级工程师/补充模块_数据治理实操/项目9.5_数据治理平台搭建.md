# 项目9.5：数据治理平台搭建

> **所属阶段**：L2 中级工程师 | **预计时长**：20h | **难度**：★★★★☆
>
> **技术栈**：Great Expectations + DataHub + OpenLineage + Marquez + Apache Ranger + Grafana

---

## 一、项目描述

为L1项目4的离线数仓和L2项目7的实时管道搭建完整的数据治理体系，覆盖数据质量、元数据管理、数据血缘追踪和数据安全四大领域。通过本项目，学员将掌握生产级数据治理平台的搭建方法，建立"可观测、可追踪、可管控"的数据治理能力。

### 业务价值

```
数据治理平台解决的问题:

  问题1: "数据质量出了问题，什么时候发现的？"
    → 以前: 下游报表异常才发现，影响已扩散
    → 现在: ETL后自动质量检查，分钟级告警

  问题2: "这张表的字段什么含义？负责人是谁？"
    → 以前: 靠口口相传，Wiki文档过时
    → 现在: DataHub元数据平台，实时搜索

  问题3: "这个指标的上游数据源是什么？"
    → 以前: 翻代码、问同事，耗时数小时
    → 现在: 血缘图一键追踪，秒级定位

  问题4: "谁访问了敏感数据？"
    → 以前: 无记录，无法审计
    → 现在: Ranger权限管控 + 审计日志
```

### 3.5 与Airflow集成：ETL后自动运行质量检查

from airflow import DAG
from airflow.operators.python import PythonOperator
from airflow.providers.apache.hive.operators.hive import HiveOperator
from datetime import datetime, timedelta
import great_expectations as gx
import requests
import json

default_args = {
    'owner': 'data-governance',
    'depends_on_past': False,
    'retries': 1,
    'retry_delay': timedelta(minutes=5),
}

dag = DAG(
    'data_warehouse_etl_with_dq',
    default_args=default_args,
    schedule_interval='0 6 * * *',
    start_date=datetime(2024, 1, 1),
    catchup=False,
    max_active_runs=1,
)

def run_ge_checkpoint(suite_name, table_name, **context):
    ds = context['ds']
    ctx = gx.get_context()
    checkpoint = ctx.get_checkpoint(f"{table_name}_checkpoint")
    result = checkpoint.run()
    if not result['success']:
        raise ValueError(f"数据质量检查失败: table={table_name}, suite={suite_name}, dt={ds}")
    return result['success']

def send_wechat_alert(suite_name, table_name, **context):
    webhook_url = "https://qyapi.weixin.qq.com/cgi-bin/webhook/send?key=YOUR_KEY"
    ds = context['ds']
    payload = {
        "msgtype": "markdown",
        "markdown": {
            "content": (
                f"**数据质量告警**\n"
                f"> 日期: {ds}\n"
                f"> 表: {table_name}\n"
                f"> 规则集: {suite_name}\n"
                f"> 状态: **检查失败**\n"
                f"> 请及时排查"
            )
        }
    }
    requests.post(webhook_url, json=payload)

def send_email_alert(suite_name, table_name, **context):
    import smtplib
    from email.mime.text import MIMEText
    ds = context['ds']
    msg = MIMEText(
        f"数据质量检查失败\n日期: {ds}\n表: {table_name}\n规则集: {suite_name}\n请及时排查"
    )
    msg['Subject'] = f'[数据质量告警] {table_name} 检查失败 - {ds}'
    msg['From'] = 'dq-alert@company.com'
    msg['To'] = 'data-team@company.com'
    with smtplib.SMTP('smtp.company.com', 25) as server:
        server.send_message(msg)

### 3.6 质量检查结果写入MySQL（供Grafana查询）

```python
import great_expectations as gx
import pymysql
from datetime import datetime

def save_dq_result_to_mysql(suite_name, table_name, result, dt):
    conn = pymysql.connect(
        host='mysql', port=3306, user='root',
        password='root123', database='dq_monitor'
    )
    cursor = conn.cursor()

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS dq_check_result (
            id INT AUTO_INCREMENT PRIMARY KEY,
            check_time DATETIME,
            suite_name VARCHAR(200),
            table_name VARCHAR(200),
            dt VARCHAR(20),
            success TINYINT,
            total_expectations INT,
            successful_expectations INT,
            failed_expectations INT,
            success_percent DECIMAL(5,2)
        )
    """)

    stats = result['run_results']
    total = 0
    success = 0
    failed = 0

    for run_result in stats.values():
        vr = run_result['validation_result']
        results = vr['results']
        total += len(results)
        for r in results:
            if r['success']:
                success += 1
            else:
                failed += 1

    success_pct = round(success / total * 100, 2) if total > 0 else 0

    cursor.execute("""
        INSERT INTO dq_check_result
        (check_time, suite_name, table_name, dt, success,
         total_expectations, successful_expectations, failed_expectations, success_percent)
        VALUES (%s, %s, %s, %s, %s, %s, %s, %s, %s)
    """, (
        datetime.now(), suite_name, table_name, dt,
        1 if result['success'] else 0,
        total, success, failed, success_pct
    ))

    conn.commit()
    cursor.close()
    conn.close()
```

```python
# dags/warehouse_etl_with_lineage.py
from airflow import DAG
from airflow.operators.python import PythonOperator
from airflow.providers.apache.hive.operators.hive import HiveOperator
from datetime import datetime, timedelta

default_args = {
    'owner': 'data-engineering',
    'depends_on_past': False,
    'retries': 2,
    'retry_delay': timedelta(minutes=5),
}

with DAG(
    'warehouse_etl_with_lineage',
    default_args=default_args,
    schedule_interval='0 6 * * *',
    start_date=datetime(2024, 1, 1),
    catchup=False,
) as dag:

    load_ods_order = HiveOperator(
        task_id='load_ods_order_info',
        hql="""
            INSERT OVERWRITE TABLE ods.ods_order_info PARTITION (dt='{{ ds }}')
            SELECT order_id, user_id, order_status, total_amount,
                   payment_amount, shipping_province, shipping_city,
                   create_time, pay_time, finish_time
            FROM staging.stg_order_info WHERE dt='{{ ds }}';
        """,
    )

    load_ods_user = HiveOperator(
        task_id='load_ods_user_info',
        hql="""
            INSERT OVERWRITE TABLE ods.ods_user_info PARTITION (dt='{{ ds }}')
            SELECT user_id, username, gender, birthday, province, city,
                   register_date, user_level
            FROM staging.stg_user_info WHERE dt='{{ ds }}';
        """,
    )

    load_ods_sku = HiveOperator(
        task_id='load_ods_sku_info',
        hql="""
            INSERT OVERWRITE TABLE ods.ods_sku_info PARTITION (dt='{{ ds }}')
            SELECT sku_id, sku_name, category_id, category_name, brand,
                   price, stock, status
            FROM staging.stg_sku_info WHERE dt='{{ ds }}';
        """,
    )

    build_dwd = HiveOperator(
        task_id='build_dwd_order_detail',
        hql="""
            INSERT OVERWRITE TABLE dwd.dwd_order_detail PARTITION (dt='{{ ds }}')
            SELECT
                d.detail_id, d.order_id, o.user_id,
                u.username AS user_name, u.gender AS user_gender,
                u.user_level, u.province AS user_province, u.city AS user_city,
                d.sku_id, d.sku_name, s.category_id, s.category_name, s.brand,
                d.order_price, d.sku_num, o.total_amount, o.payment_amount,
                o.order_status, d.create_time,
                YEAR(d.create_time) AS order_year,
                MONTH(d.create_time) AS order_month,
                DAY(d.create_time) AS order_day,
                WEEKOFYEAR(d.create_time) AS order_week,
                HOUR(d.create_time) AS order_hour
            FROM ods.ods_order_detail d
            JOIN ods.ods_order_info o ON d.order_id = o.order_id
            JOIN ods.ods_user_info u ON o.user_id = u.user_id
            JOIN ods.ods_sku_info s ON d.sku_id = s.sku_id
            WHERE d.dt='{{ ds }}' AND o.dt='{{ ds }}'
              AND u.dt='{{ ds }}' AND s.dt='{{ ds }}';
        """,
    )

    build_dws = HiveOperator(
        task_id='build_dws_trade_user_order_day',
        hql="""
            INSERT OVERWRITE TABLE dws.dws_trade_user_order_day PARTITION (dt='{{ ds }}')
            SELECT
                user_id, COUNT(DISTINCT order_id), SUM(total_amount),
                SUM(payment_amount), SUM(sku_num), 0, 0
            FROM dwd.dwd_order_detail WHERE dt='{{ ds }}'
            GROUP BY user_id;
        """,
    )

    build_ads = HiveOperator(
        task_id='build_ads_trade_stats_day',
        hql="""
            INSERT OVERWRITE TABLE ads.ads_trade_stats_day PARTITION (dt='{{ ds }}')
            SELECT
                COUNT(DISTINCT order_id), SUM(total_amount),
                COUNT(DISTINCT user_id),
                ROUND(SUM(total_amount)/COUNT(DISTINCT order_id),2),
                MAX(total_amount), 0, 0, 0, SUM(payment_amount)
            FROM dwd.dwd_order_detail WHERE dt='{{ ds }}';
        """,
    )

    [load_ods_order, load_ods_user, load_ods_sku] >> build_dwd >> build_dws >> build_ads
```

### 6.4 数据脱敏规则

```python
from pyspark.sql import SparkSession
from pyspark.sql.functions import udf, col, regexp_replace, concat, lit, substring
from pyspark.sql.types import StringType

spark = SparkSession.builder.appName("DataMasking").enableHiveSupport().getOrCreate()

df = spark.sql("SELECT * FROM dwd.dwd_order_detail WHERE dt='2024-01-15'")

@udf(StringType())
def mask_phone(phone):
    if phone and len(phone) == 11:
        return phone[:3] + "****" + phone[7:]
    return phone

@udf(StringType())
def mask_id_card(id_card):
    if id_card and len(id_card) == 18:
        return id_card[:6] + "********" + id_card[14:]
    return id_card

@udf(StringType())
def mask_bank_card(card):
    if card and len(card) >= 13:
        return card[:4] + "****" + card[-4:]
    return card

df_masked = df \
    .withColumn("user_phone_masked", mask_phone(col("user_phone"))) \
    .withColumn("user_id_card_masked", mask_id_card(col("user_id_card"))) \
    .withColumn("bank_card_masked", mask_bank_card(col("bank_card")))

df_masked.write.mode("overwrite").insertInto("dwd.dwd_order_detail_masked")
```

## 八、数据治理SOP文档模板

```
数据治理标准操作流程 (SOP)

1. 数据质量检查SOP
   1.1 每日ETL完成后自动触发GE Checkpoint
   1.2 检查失败→企业微信告警→值班工程师排查
   1.3 排查步骤: 查看Data Docs → 定位失败规则 → 检查源数据 → 修复数据 → 重跑Checkpoint
   1.4 修复后验证: 重新运行Checkpoint确认通过率100%

2. 元数据变更SOP
   2.1 表结构变更: 先更新DataHub数据字典 → 再执行DDL变更
   2.2 新增数据源: 配置DataHub摄取Recipe → 执行摄取 → 验证元数据完整性
   2.3 字段含义变更: 更新DataHub字段描述 → 通知下游使用方

3. 血缘追踪SOP
   3.1 新增ETL任务: 确保Airflow/Spark/Flink配置OpenLineage
   3.2 血缘异常排查: Marquez UI查看 → 对比代码逻辑 → 修复血缘配置
   3.3 影响分析: 在Marquez中搜索数据集 → 查看下游依赖 → 通知相关方

4. 数据安全SOP
   4.1 新增用户权限: 提交工单 → 审批 → Ranger配置策略 → 验证权限
   4.2 敏感数据访问: 审计日志审查 → 异常访问告警 → 安全团队介入
   4.3 数据脱敏: 新增敏感字段 → 配置脱敏规则 → 验证脱敏效果
```
