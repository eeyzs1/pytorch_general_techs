```python
# ========== 函数基础 ==========
def greet(name, greeting="Hello"):
    """向某人问好（这是docstring）"""
    return f"{greeting}, {name}!"

print(greet("Alice"))            # "Hello, Alice!"
print(greet("Bob", "Hi"))        # "Hi, Bob!"
print(greet(greeting="Hey", name="Charlie"))  # 关键字参数

# ========== 参数类型 ==========
# 位置参数
def add(a, b):
    return a + b

# 默认参数（默认值必须是不可变类型！）
def power(base, exp=2):
    return base ** exp

# 可变参数 *args（接收任意数量的位置参数，打包为元组）
def sum_all(*numbers):
    """对任意数量的数求和"""
    return sum(numbers)

print(sum_all(1, 2, 3, 4, 5))  # 15
print(sum_all(10, 20))         # 30

# 关键字参数 **kwargs（接收任意数量的关键字参数，打包为字典）
def log_event(**info):
    """记录事件信息"""
    for key, value in info.items():
        print(f"[{key}]: {value}")

log_event(level="ERROR", message="连接超时", timestamp="2024-01-01 10:00:00")

# 参数顺序：位置参数 → *args → 关键字参数 → **kwargs
def complex_func(a, b, *args, c=10, **kwargs):
    print(f"a={a}, b={b}, args={args}, c={c}, kwargs={kwargs}")

complex_func(1, 2, 3, 4, 5, c=20, x=100, y=200)
# a=1, b=2, args=(3,4,5), c=20, kwargs={'x':100, 'y':200}

# ========== Lambda表达式（匿名函数）==========
# 语法：lambda 参数: 表达式
square = lambda x: x ** 2
print(square(5))  # 25

# 常用于排序的key参数
students = [
    {"name": "Alice", "score": 85},
    {"name": "Bob", "score": 92},
    {"name": "Charlie", "score": 78},
]

# 按分数排序
sorted_students = sorted(students, key=lambda s: s["score"], reverse=True)
for s in sorted_students:
    print(f"{s['name']}: {s['score']}")

# ========== 装饰器（了解）==========
# 装饰器是Python中一种强大的函数增强机制
import time

def timer(func):
    """计时装饰器——测量函数执行时间"""
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"{func.__name__} 执行耗时: {elapsed:.4f}秒")
        return result
    return wrapper

@timer
def slow_function():
    """模拟耗时操作"""
    time.sleep(0.5)
    return "完成"

result = slow_function()
print(result)

# ========== 函数的类型提示 ==========
def process_data(
    data: list[dict],
    max_items: int = 100
) -> dict[str, int]:
    """处理数据并返回统计信息"""
    return {"count": len(data[:max_items]), "processed": min(len(data), max_items)}

# 虽然Python不强制执行类型提示，但在团队协作中非常重要
# VS Code + Pylance会根据类型提示提供更好的自动补全
```

```python
"""
增强版日志分析器
"""
def enhanced_log_analysis(logs):
    """
    在基础分析之上增加：
    1. 识别高频访问IP（访问次数 > 3的标记为可疑）
    2. 统计每个URL的访问次数
    3. 计算错误率（非200状态码占比）
    4. 识别连续失败的IP（连续2次以上非200）
    """
    # TODO: 1. 高频IP识别
    ip_counts = {}
    for line in logs:
        ip = line.split()[0]
        ip_counts[ip] = ip_counts.get(ip, 0) + 1

    suspicious_ips = {ip: count for ip, count in ip_counts.items() if count > 3}
    print(f"可疑IP（访问>3次）: {suspicious_ips}")

    # TODO: 2. URL访问统计
    url_counts = {}
    for line in logs:
        parts = line.split()
        url = parts[3] if len(parts) > 3 else "unknown"
        url_counts[url] = url_counts.get(url, 0) + 1

    sorted_urls = sorted(url_counts.items(), key=lambda x: x[1], reverse=True)
    print("\nURL访问排名:")
    for url, count in sorted_urls:
        print(f"  {url}: {count}次")

    # TODO: 3. 错误率
    total = len(logs)
    errors = sum(1 for line in logs if not line.endswith("200"))
    error_rate = errors / total * 100 if total > 0 else 0
```

**3. 简单猜数字游戏**

```python
"""
猜数字游戏需求：
- 系统随机生成1-100之间的数字
- 玩家有10次机会
- 每次猜测后提示"大了"或"小了"
- 记录历史猜测
- 如果猜对，显示用了多少次
- 如果10次用完，显示正确答案
"""
import random

def guess_number_game():
    target = random.randint(1, 100)
    max_attempts = 10
    attempts = 0
    history = []

    print("=" * 40)
    print("     猜数字游戏 (1-100)")
    print(f"     你有{max_attempts}次机会")
    print("=" * 40)

    while attempts < max_attempts:
        try:
            guess = int(input(f"\n第{attempts+1}次猜测: "))
        except ValueError:
            print("请输入有效的数字！")
            continue

        attempts += 1
        history.append(guess)

        if guess == target:
            print(f"\n🎉 恭喜！你猜对了！用了{attempts}次")
            print(f"答案就是: {target}")
            print(f"猜测历史: {history}")
            return attempts
        elif guess < target:
            print("📈 太小了，再大一点")
        else:
            print("📉 太大了，再小一点")

        print(f"剩余次数: {max_attempts - attempts}")

    print(f"\n😢 游戏结束！正确答案是: {target}")
    print(f"猜测历史: {history}")
    return -1

if __name__ == "__main__":
    guess_number_game()
```
