### 2.2 CSV文件处理（25分钟）

```python
import csv
import os

# ========== 写入CSV ==========
def write_sample_csv(filepath):
    """生成示例CSV文件"""
    header = ["id", "name", "age", "city", "salary"]

    data = [
        ["001", "张三", "28", "北京", "15000"],
        ["002", "李四", "32", "上海", "20000"],
        ["", "", "", "", ""],           # 空行
        ["003", "王五", "25", "深圳", "18000"],
        ["003", "王五", "25", "深圳", "18000"],  # 重复
        ["004", " 赵六 ", "35 ", " 广州 ", "25000"],  # 含空格
        ["005", "孙七", "29", "杭州", "22000"],
    ]

    with open(filepath, "w", encoding="utf-8", newline="") as f:
        writer = csv.writer(f)
        writer.writerow(header)
        writer.writerows(data)

    print(f"已写入 {len(data)} 行数据到 {filepath}")

# ========== 读取CSV（基础方式）==========
def read_csv_basic(filepath):
    """基础CSV读取"""
    with open(filepath, "r", encoding="utf-8") as f:
        reader = csv.reader(f)
        header = next(reader)  # 读取表头
        print(f"表头: {header}")

        for row in reader:
            print(f"行数据: {row}")

# ========== 读取CSV（DictReader方式）==========
def read_csv_as_dict(filepath):
    """将CSV读取为字典列表"""
    records = []
    with open(filepath, "r", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        for row in reader:
            records.append(dict(row))
    return records

# ========== 数据清洗与写入 ==========
def clean_csv(input_file, output_file):
    """
    清洗CSV文件：
    1. 去除空行
    2. 去除重复行（以id为准）
    3. 标准化格式（去除两端空格）
    4. 同时输出CSV和JSON两种格式
    """
    seen_ids = set()
    cleaned = []
    skipped_empty = 0
    skipped_duplicate = 0

    with open(input_file, 'r', encoding='utf-8') as f:
        reader = csv.DictReader(f)
        for row in reader:
            # 跳过空行（所有值都为空的行）
            if not any(row.values()):
                skipped_empty += 1
                continue

            # 去重（以id为准）
            row_id = row.get('id', '').strip()
            if row_id in seen_ids:
                skipped_duplicate += 1
                continue
            seen_ids.add(row_id)

            # 标准化：去除两端空格
            row = {k: v.strip() for k, v in row.items()}
            cleaned.append(row)

    # 写入CSV
    with open(output_file, 'w', encoding='utf-8', newline='') as f:
        if cleaned:
            writer = csv.DictWriter(f, fieldnames=cleaned[0].keys())
            writer.writeheader()
            writer.writerows(cleaned)

    # 同时输出JSON版本（大数据生态常用格式）
    import json
    json_file = output_file.replace('.csv', '.json')
    with open(json_file, 'w', encoding='utf-8') as f:
        json.dump(cleaned, f, ensure_ascii=False, indent=2)

    # 输出清洗统计
    print(f"清洗报告:")
    print(f"  总行数: {len(cleaned) + skipped_empty + skipped_duplicate}")
    print(f"  有效行数: {len(cleaned)}")
    print(f"  跳过空行: {skipped_empty}")
    print(f"  跳过重复: {skipped_duplicate}")
    print(f"  输出CSV: {output_file}")
    print(f"  输出JSON: {json_file}")

    return len(cleaned)

# 运行示例
if __name__ == "__main__":
    write_sample_csv("sample.csv")
    clean_csv("sample.csv", "cleaned.csv")
```

```python
    """写入JSON Lines格式（大数据常用）"""

        for record in records:
            f.write(json.dumps(record, ensure_ascii=False) + "\n")

    """读取JSON Lines格式"""

                record = json.loads(line.strip())
                records.append(record)

    return records
```

```python
# ========== 基本异常处理 ==========

# try-except 基本结构
try:
    result = 10 / 0
except ZeroDivisionError:
    print("错误：不能除以零！")

# 捕获多种异常
try:
    num = int(input("请输入一个数字: "))
    result = 100 / num
    print(f"100 / {num} = {result}")
except ValueError:
    print("错误：请输入有效的数字")
except ZeroDivisionError:
    print("错误：不能除以零")
except Exception as e:
    print(f"未知错误: {e}")

# try-except-else-finally 完整结构
def safe_read_file(filepath):
    try:
        f = open(filepath, "r", encoding="utf-8")
    except FileNotFoundError:
        print(f"文件不存在: {filepath}")
        return None
    except PermissionError:
        print(f"没有权限读取: {filepath}")
        return None
    else:
        # try块成功执行后运行
        content = f.read()
        print(f"成功读取文件，共 {len(content)} 字符")
        return content
    finally:
        # 无论是否发生异常都会执行
        try:
            f.close()
        except NameError:
            pass  # f未定义

# ========== 自定义异常 ==========
class DataValidationError(Exception):
    """数据验证异常"""
    def __init__(self, message, row_data=None):
        super().__init__(message)
        self.row_data = row_data

class DataFormatError(Exception):
    """数据格式异常"""
    pass

# ========== 实际应用：健壮的数据处理函数 ==========
import logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s [%(levelname)s] %(message)s')
logger = logging.getLogger(__name__)

def robust_csv_reader(filepath):
    """
    健壮的CSV读取器：
    - 处理文件不存在
    - 处理编码问题
    - 处理格式错误
    - 记录日志
    """
    logger.info(f"开始读取文件: {filepath}")

    try:
        with open(filepath, "r", encoding="utf-8") as f:
            reader = csv.DictReader(f)
            records = []
            error_count = 0

            for line_num, row in enumerate(reader, 1):
                try:
                    # 验证必填字段
                    if not row.get("id"):
                        raise DataValidationError(
                            f"第{line_num}行缺少id字段",
                            row_data=row
                        )
                    records.append(row)
                except DataValidationError as e:
                    error_count += 1
                    logger.warning(f"数据验证失败: {e}")

            logger.info(f"读取完成: 成功{len(records)}条, 失败{error_count}条")
            return records

    except FileNotFoundError:
        logger.error(f"文件不存在: {filepath}")
        return []
    except UnicodeDecodeError:
        # 尝试其他编码
        logger.warning("UTF-8解码失败，尝试GBK编码")
        try:
            with open(filepath, "r", encoding="gbk") as f:
                reader = csv.DictReader(f)
                return list(reader)
        except Exception as e2:
            logger.error(f"GBK解码也失败: {e2}")
            return []
    except Exception as e:
        logger.error(f"读取文件时发生未知错误: {e}", exc_info=True)
        return []


# ========== 异常处理最佳实践 ==========
# 1. 具体捕获：捕获具体的异常类型，不要笼统地捕获Exception
# 2. 不要吞异常：至少记录日志，不要用 pass 默默忽略
# 3. 清理资源：使用with语句或finally确保资源释放
# 4. 不要用异常控制流程：异常用于处理意外情况，不要用于正常逻辑

# 不好的做法（用异常控制流程）:
def bad_example(data):
    try:
        return data["key"]
    except KeyError:
        return None

# 好的做法：
def good_example(data):
    return data.get("key")  # 使用dict.get()方法

# 异常链：保留原始异常信息
```

### 2.5 上下文管理器原理（15分钟）

```python
# ========== 上下文管理器（Context Manager）==========

# with语句的实际等价代码：
# with EXPR as VAR:
#     BLOCK
#
# 等价于：
# manager = EXPR
# VAR = manager.__enter__()
# try:
#     BLOCK
# finally:
#     manager.__exit__(exc_type, exc_val, exc_tb)

# ========== 基于类的上下文管理器 ==========
class Timer:
    """计时器上下文管理器"""
    import time

    def __enter__(self):
        self.start = time.perf_counter()
        return self

    def __exit__(self, exc_type, exc_val, exc_tb):
        self.end = time.perf_counter()
        self.elapsed = self.end - self.start
        print(f"代码块执行耗时: {self.elapsed:.4f}秒")
        return False  # 不抑制异常

with Timer():
    # 模拟耗时操作
    total = sum(range(1_000_000))
    print(f"计算结果: {total}")

# ========== 基于生成器的上下文管理器 ==========
from contextlib import contextmanager

@contextmanager
def temporary_file(filename, content=""):
    """创建临时文件，代码块结束后自动删除"""
    import os
    print(f"创建临时文件: {filename}")
    with open(filename, "w", encoding="utf-8") as f:
        f.write(content)
    try:
        yield filename  # 将控制权交给with块
    finally:
        os.remove(filename)
        print(f"已删除临时文件: {filename}")

with temporary_file("temp.txt", "Hello, World!") as tmp:
    with open(tmp, "r") as f:
        print(f"文件内容: {f.read()}")
# exit后 temp.txt 已被自动删除

# ========== 实用上下文管理器示例 ==========

@contextmanager
def change_directory(path):
    """临时切换工作目录"""
    import os
    old_dir = os.getcwd()
    os.chdir(path)
    try:
        yield
    finally:
        os.chdir(old_dir)

@contextmanager
def ignored(*exceptions):
    """忽略指定的异常（谨慎使用！）"""
    try:
        yield
    except exceptions:
        pass

# 使用示例
# with ignored(FileNotFoundError):
#     os.remove("some_file.txt")  # 文件不存在也不会报错

@contextmanager
def log_execution(operation_name):
    """记录操作执行的上下文管理器"""
    import logging
    logger = logging.getLogger(__name__)
    logger.info(f"[开始] {operation_name}")
    try:
        yield
        logger.info(f"[完成] {operation_name}")
    except Exception as e:
        logger.error(f"[失败] {operation_name}: {e}")
        raise
```

```python
"""
生成包含1000行的模拟订单CSV文件
要求包含：空行、重复数据、格式不规整的数据
"""
import csv
import random
from datetime import datetime, timedelta

def generate_orders_csv(filepath, num_rows=1000):
    """生成模拟订单数据"""
    cities = ["北京", "上海", "深圳", "杭州", "广州", "成都", "武汉", "南京"]
    categories = ["电子产品", "服装", "食品", "图书", "家居", "运动", "美妆", "母婴"]
    statuses = ["completed", "pending", "cancelled", "refunded"]

    header = ["order_id", "user_name", "city", "category",
              "amount", "status", "order_date"]

    with open(filepath, "w", encoding="utf-8", newline="") as f:
        writer = csv.writer(f)
        writer.writerow(header)

        used_ids = set()
        for i in range(num_rows):
            # 约10%的概率生成空行
            if random.random() < 0.1:
                writer.writerow(["", "", "", "", "", "", ""])
                continue

            order_id = f"ORD{random.randint(1, 800):04d}"

            # 约5%的概率生成重复ID
            if random.random() < 0.05 and len(used_ids) > 10:
                order_id = random.choice(list(used_ids))
            else:
                used_ids.add(order_id)

            # 约8%的概率在字符串中添加多余空格
            add_space = random.random() < 0.08

            user_name = f"用户{random.randint(1, 200):03d}"
            city = random.choice(cities)
            category = random.choice(categories)
            amount = round(random.uniform(10, 5000), 2)
            status = random.choice(statuses)
            days_ago = random.randint(0, 90)
            order_date = (datetime.now() - timedelta(days=days_ago)).strftime("%Y-%m-%d")

            if add_space:
                user_name = f"  {user_name}  "
                city = f" {city} "

            writer.writerow([order_id, user_name, city, category,
                           amount, status, order_date])

    print(f"已生成 {num_rows} 行模拟订单数据到 {filepath}")

# 生成文件

print("数据生成完成！")
```

```python
"""
清洗上面生成的订单CSV文件，并输出分析报告
"""
import csv
import json
from collections import defaultdict

def clean_and_analyze_orders(input_file, output_file):
    """清洗订单数据并输出统计"""
    seen_ids = set()
    cleaned = []
    stats = {
        "total": 0,
        "empty_skipped": 0,
        "duplicate_skipped": 0,
        "by_city": defaultdict(int),
        "by_category": defaultdict(int),
        "by_status": defaultdict(int),
        "total_amount": 0,
        "by_month": defaultdict(lambda: {"orders": 0, "amount": 0.0}),
    }

    with open(input_file, "r", encoding="utf-8") as f:
        reader = csv.DictReader(f)
        for row in reader:
            stats["total"] += 1

            # 跳过空行
            if not any(row.values()):
                stats["empty_skipped"] += 1
                continue

            # 去重
            oid = row.get("order_id", "").strip()
            if oid in seen_ids:
                stats["duplicate_skipped"] += 1
                continue
            seen_ids.add(oid)

            # 标准化
            row = {k: v.strip() for k, v in row.items()}

            # 统计
            amount = float(row.get("amount", 0))
            stats["by_city"][row.get("city", "未知")] += 1
            stats["by_category"][row.get("category", "未知")] += 1
            stats["by_status"][row.get("status", "未知")] += 1
            stats["total_amount"] += amount

            month = row.get("order_date", "未知")[:7]
            stats["by_month"][month]["orders"] += 1
            stats["by_month"][month]["amount"] += amount

            cleaned.append(row)

    # 写入清洗后的数据
    if cleaned:
        with open(output_file, "w", encoding="utf-8", newline="") as f:
            writer = csv.DictWriter(f, fieldnames=cleaned[0].keys())
            writer.writeheader()
            writer.writerows(cleaned)

        json_file = output_file.replace(".csv", ".json")
        with open(json_file, "w", encoding="utf-8") as f:
            json.dump(cleaned, f, ensure_ascii=False, indent=2)

    # 输出报告
    print("=" * 50)
    print("         订单数据清洗与分析报告")
    print("=" * 50)
    print(f"\n总行数: {stats['total']}")
    print(f"有效行数: {len(cleaned)}")
    print(f"跳过空行: {stats['empty_skipped']}")
    print(f"跳过重复: {stats['duplicate_skipped']}")

    print("\n【城市分布 TOP 5】")
    for city, count in sorted(stats["by_city"].items(),
                               key=lambda x: x[1], reverse=True)[:5]:
        print(f"  {city}: {count}单")

    print("\n【品类分布】")
    for cat, count in sorted(stats["by_category"].items(),
                              key=lambda x: x[1], reverse=True):

    print("\n【状态分布】")
    for status, count in stats["by_status"].items():
        pct = count / len(cleaned) * 100 if cleaned else 0
        print(f"  {status}: {count}单 ({pct:.1f}%)")

    print(f"\n总交易额: ¥{stats['total_amount']:,.2f}")
    if cleaned:
        print(f"平均客单价: ¥{stats['total_amount']/len(cleaned):,.2f}")

    print("\n【月度趋势】")
    for month in sorted(stats["by_month"].keys()):
        m = stats["by_month"][month]
        print(f"  {month}: {m['orders']}单, ¥{m['amount']:,.2f}")
```

**编写一个完整的文件格式转换器**，支持以下转换：

```python
"""
文件格式转换器：支持 CSV ↔ JSON ↔ JSONL 互转
"""
import csv
import json
import argparse
from pathlib import Path

class FileConverter:
    """通用的文件格式转换器"""

    SUPPORTED_FORMATS = {".csv", ".json", ".jsonl"}

    @staticmethod
    def detect_format(filepath):
        """自动检测文件格式"""
        suffix = Path(filepath).suffix.lower()
        if suffix == ".csv":
            return "csv"
        elif suffix == ".json":
            return "json"
        elif suffix == ".jsonl":
            return "jsonl"
        else:
            raise ValueError(f"不支持的文件格式: {suffix}")

    def convert(self, input_file, output_file):
        """自动转换文件格式"""
        in_fmt = self.detect_format(input_file)
        out_fmt = self.detect_format(output_file)

        print(f"转换: {in_fmt} → {out_fmt}")

        data = self._read(input_file, in_fmt)
        self._write(output_file, data, out_fmt)

        print(f"完成: {len(data)} 条记录已写入 {output_file}")

    def _read(self, filepath, fmt):
        if fmt == "csv":
            return self._read_csv(filepath)
        elif fmt == "json":
            return self._read_json(filepath)
        elif fmt == "jsonl":
            return self._read_jsonl(filepath)

    def _write(self, filepath, data, fmt):
        if fmt == "csv":
            self._write_csv(filepath, data)
        elif fmt == "json":
            self._write_json(filepath, data)
        elif fmt == "jsonl":
            self._write_jsonl(filepath, data)

    def _read_csv(self, filepath):
        with open(filepath, "r", encoding="utf-8") as f:
            return list(csv.DictReader(f))

    def _read_json(self, filepath):
        with open(filepath, "r", encoding="utf-8") as f:
            return json.load(f)

    def _read_jsonl(self, filepath):
        records = []
        with open(filepath, "r", encoding="utf-8") as f:
            for line in f:
                if line.strip():
                    records.append(json.loads(line))
        return records

    def _write_csv(self, filepath, data):
        if not data:
            return
        with open(filepath, "w", encoding="utf-8", newline="") as f:
            writer = csv.DictWriter(f, fieldnames=data[0].keys())
            writer.writeheader()
            writer.writerows(data)

    def _write_json(self, filepath, data):
        with open(filepath, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)

    def _write_jsonl(self, filepath, data):
        with open(filepath, "w", encoding="utf-8") as f:
            for record in data:
                f.write(json.dumps(record, ensure_ascii=False) + "\n")


if __name__ == "__main__":
    converter = FileConverter()

    # 示例：CSV → JSON
    converter.convert("orders.csv", "output.json")

    # 示例：JSON → CSV
    converter.convert("output.json", "output.csv")

    # 示例：CSV → JSONL
    converter.convert("orders.csv", "output.jsonl")
```
