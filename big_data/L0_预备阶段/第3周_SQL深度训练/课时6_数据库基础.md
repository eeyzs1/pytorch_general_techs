### 2.1 数据库核心概念（20分钟）

```
关系型数据库（RDBMS）的核心：一切皆"表"

┌──────────────────────────────────────────┐
│                   数据库                   │
│  ┌──────────┐  ┌──────────┐  ┌──────────┐│
│  │  users   │  │ orders   │  │ products ││
│  │  (表)    │  │  (表)    │  │  (表)    ││
│  └──────────┘  └──────────┘  └──────────┘│
│       │             │             │        │
│       └──────┬──────┘             │        │
│              │ 外键(user_id)       │        │
│              └────────────────────┘        │
└──────────────────────────────────────────┘
```

**核心术语**：

| 术语 | 说明 | 类比 |
|------|------|------|
| 表（Table） | 数据的二维结构，有行和列 | Excel工作表 |
| 行（Row/Record） | 表中的一条数据 | Excel的一行 |
| 列（Column/Field） | 表中的一个属性 | Excel的一列 |
| 主键（Primary Key） | 唯一标识一行的列 | 身份证号 |
| 外键（Foreign Key） | 关联其他表的列 | 订单中的用户ID |
| 索引（Index） | 加速查询的数据结构 | 书的目录 |
| 约束（Constraint） | 数据完整性规则 | NOT NULL、UNIQUE |

- Spark SQL 使用标准SQL

### 2.3 数据类型详解（20分钟）

```sql
-- ========== 数值类型 ==========

-- INT: 整数（-2^31 到 2^31-1，约±21亿）
-- BIGINT: 大整数（-2^63 到 2^63-1）
-- TINYINT: 小整数（-128到127）
-- DECIMAL(M,D): 精确小数（M总位数，D小数位）
-- FLOAT/DOUBLE: 近似浮点数

CREATE TABLE numeric_demo (
    id INT,
    age TINYINT,
    population BIGINT,
    price DECIMAL(10, 2),    -- 总共10位，2位小数
    rating FLOAT,
    score DOUBLE
);

-- DECIMAL vs FLOAT
-- DECIMAL: 精确，适合金额计算
-- FLOAT/DOUBLE: 近似，适合科学计算

SELECT 0.1 + 0.2;             -- 浮点数的不精确性
SELECT CAST(0.1 AS DECIMAL(10,2)) + CAST(0.2 AS DECIMAL(10,2));


-- ========== 字符串类型 ==========

-- CHAR(N): 定长字符串（最多255字符），空格填充
-- VARCHAR(N): 变长字符串（最多65535字符），节省空间
-- TEXT: 长文本（最多65535字符）
-- LONGTEXT: 超长文本（最多4GB）

CREATE TABLE string_demo (
    code CHAR(6),             -- 固定6位编码，如"CN1001"
    name VARCHAR(100),        -- 可变长度名称
    description TEXT,         -- 长描述
    content LONGTEXT          -- 超长内容
);


-- ========== 日期时间类型 ==========

-- DATE: 日期（'YYYY-MM-DD'）
-- TIME: 时间（'HH:MM:SS'）
-- DATETIME: 日期时间（'YYYY-MM-DD HH:MM:SS'）
-- TIMESTAMP: 时间戳（范围1970-2038，自动时区转换）

CREATE TABLE datetime_demo (
    birth_date DATE,
    login_time TIME,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);


-- ========== 其他类型 ==========

-- BOOLEAN: 实际是TINYINT(1)，0为false，非0为true
-- ENUM: 枚举类型，选一个值
-- JSON: JSON数据类型（MySQL 5.7+）

CREATE TABLE other_demo (
    is_active BOOLEAN DEFAULT TRUE,
    gender ENUM('male', 'female', 'other'),
    extra_info JSON
);
```

### 2.4 CRUD操作（30分钟）

```sql
-- ========== 创建表（CREATE）==========

CREATE TABLE students (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(50) NOT NULL,
    age TINYINT CHECK (age >= 0 AND age <= 150),
    gender ENUM('male', 'female') DEFAULT 'male',
    email VARCHAR(100) UNIQUE,
    city VARCHAR(50) DEFAULT '北京',
    score DECIMAL(5,2) DEFAULT 0.00,
    enrolled_date DATE DEFAULT (CURRENT_DATE),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);


-- ========== 插入数据（INSERT）==========

-- 插入单行
INSERT INTO students (name, age, gender, email, city, score)
VALUES ('张三', 22, 'male', 'zhangsan@test.com', '北京', 85.5);

-- 插入多行
INSERT INTO students (name, age, gender, email, city, score) VALUES
('李四', 25, 'female', 'lisi@test.com', '上海', 92.0),
('王五', 21, 'male', 'wangwu@test.com', '深圳', 78.5),
('赵六', 23, 'female', 'zhaoliu@test.com', '广州', 88.0),
('孙七', 24, 'male', 'sunqi@test.com', '杭州', 95.5);

-- 插入时使用默认值
INSERT INTO students (name, age, email) VALUES
('周八', 26, 'zhouba@test.com');


-- ========== 查询数据（SELECT）==========

-- 查询所有列
SELECT * FROM students;

-- 查询指定列
SELECT name, age, city, score FROM students;

-- 列别名
SELECT
    name AS 姓名,
    age AS 年龄,
    city AS 城市,
    score AS 分数
FROM students;

-- 计算列
SELECT
    name,
    score,
    score * 1.1 AS 加分后,       -- 加10%
    ROUND(score * 1.1, 1) AS 加分后_四舍五入
FROM students;

-- DISTINCT去重
SELECT DISTINCT city FROM students;
SELECT COUNT(DISTINCT city) AS 城市数量 FROM students;


-- ========== WHERE 条件筛选 ==========

-- 比较运算符: =, <>, !=, >, <, >=, <=
SELECT * FROM students WHERE score >= 85;

-- 逻辑运算符: AND, OR, NOT
SELECT * FROM students WHERE score >= 85 AND city = '北京';
SELECT * FROM students WHERE city = '北京' OR city = '上海';

-- IN
SELECT * FROM students WHERE city IN ('北京', '上海', '深圳');

-- BETWEEN
SELECT * FROM students WHERE score BETWEEN 80 AND 90;
SELECT * FROM students WHERE age BETWEEN 20 AND 25;

-- LIKE（模糊查询）
-- % 匹配任意字符序列
-- _ 匹配任意单个字符
SELECT * FROM students WHERE name LIKE '张%';    -- 姓张的
SELECT * FROM students WHERE email LIKE '%@test.com';
SELECT * FROM students WHERE name LIKE '___';    -- 3个字的名字

-- IS NULL / IS NOT NULL
SELECT * FROM students WHERE city IS NULL;
SELECT * FROM students WHERE city IS NOT NULL;


-- ========== ORDER BY 排序 ==========

SELECT name, score FROM students ORDER BY score DESC;       -- 降序
SELECT name, score FROM students ORDER BY score ASC;        -- 升序
SELECT name, city, score FROM students
ORDER BY city ASC, score DESC;  -- 先按城市升序，再按分数降序


-- ========== LIMIT 分页 ==========

-- 前5条
SELECT * FROM students ORDER BY score DESC LIMIT 5;

-- 分页（跳过0条取3条 = 第1页）
SELECT * FROM students LIMIT 3 OFFSET 0;

-- 分页（跳过3条取3条 = 第2页）
SELECT * FROM students LIMIT 3 OFFSET 3;

-- 通用分页公式: LIMIT page_size OFFSET (page - 1) * page_size


-- ========== 更新数据（UPDATE）==========

-- 更新单行
UPDATE students SET score = 90 WHERE id = 1;

-- 更新多行
UPDATE students SET score = score + 5 WHERE score < 80;

-- 更新多列
UPDATE students SET
    city = '成都',
    score = 82.5
WHERE name = '王五';

-- ⚠️ 不加WHERE会更新所有行！
-- UPDATE students SET score = 0;  -- 危险操作！


-- ========== 删除数据（DELETE）==========

-- 删除指定行
DELETE FROM students WHERE id = 1;

-- 条件删除
DELETE FROM students WHERE score < 60;

-- ⚠️ 不加WHERE会删除所有数据！
-- DELETE FROM students;  -- 危险操作！

-- 清空表（比DELETE快，重置自增计数器）
-- TRUNCATE TABLE students;


-- ========== 聚合函数 ==========

SELECT
    COUNT(*) AS 总人数,
    MIN(score) AS 最低分,
    MAX(score) AS 最高分,
    AVG(score) AS 平均分,
    SUM(score) AS 总分,
    ROUND(AVG(score), 2) AS 平均分_四舍五入
FROM students;

-- 按城市分组统计
SELECT
    city,
    COUNT(*) AS 人数,
    ROUND(AVG(score), 1) AS 平均分,
    MAX(score) AS 最高分
FROM students
GROUP BY city
ORDER BY 平均分 DESC;
```

### 2.5 创建电商数据库（35分钟）

```sql
-- ========== 创建电商数据库 ==========
DROP DATABASE IF EXISTS ecommerce;
CREATE DATABASE ecommerce
    CHARACTER SET utf8mb4
    COLLATE utf8mb4_unicode_ci;

USE ecommerce;

-- ========== 用户表 ==========
CREATE TABLE users (
    user_id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) NOT NULL COMMENT '用户名',
    email VARCHAR(100) UNIQUE NOT NULL COMMENT '邮箱',
    password_hash VARCHAR(255) NOT NULL COMMENT '密码哈希',
    phone VARCHAR(20) COMMENT '手机号',
    city VARCHAR(50) COMMENT '所在城市',
    gender ENUM('male', 'female', 'other') DEFAULT 'other',
    age TINYINT COMMENT '年龄',
    registered_at DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '注册时间',
    last_login_at DATETIME COMMENT '最后登录',
    is_active BOOLEAN DEFAULT TRUE COMMENT '是否激活',
    INDEX idx_city (city),
    INDEX idx_registered (registered_at)
) COMMENT '用户表';


-- ========== 商品分类表 ==========
CREATE TABLE categories (
    category_id INT PRIMARY KEY AUTO_INCREMENT,
    category_name VARCHAR(50) NOT NULL COMMENT '分类名称',
    parent_id INT DEFAULT NULL COMMENT '父分类ID（支持多级分类）',
    sort_order INT DEFAULT 0 COMMENT '排序',
    FOREIGN KEY (parent_id) REFERENCES categories(category_id)
) COMMENT '商品分类表';


-- ========== 商品表 ==========
CREATE TABLE products (
    product_id INT PRIMARY KEY AUTO_INCREMENT,
    product_name VARCHAR(200) NOT NULL COMMENT '商品名称',
    category_id INT COMMENT '分类ID',
    price DECIMAL(10, 2) NOT NULL COMMENT '价格',
    cost DECIMAL(10, 2) DEFAULT 0 COMMENT '成本',
    stock INT DEFAULT 0 COMMENT '库存',
    sales INT DEFAULT 0 COMMENT '销量',
    rating DECIMAL(3, 2) DEFAULT 0 COMMENT '评分(0-5)',
    description TEXT COMMENT '商品描述',
    is_on_sale BOOLEAN DEFAULT TRUE COMMENT '是否在售',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_category (category_id),
    INDEX idx_price (price),
    INDEX idx_sales (sales),
    FOREIGN KEY (category_id) REFERENCES categories(category_id)
) COMMENT '商品表';


-- ========== 订单表 ==========
CREATE TABLE orders (
    order_id INT PRIMARY KEY AUTO_INCREMENT,
    order_number VARCHAR(20) UNIQUE NOT NULL COMMENT '订单号',
    user_id INT NOT NULL COMMENT '用户ID',
    order_date DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '下单时间',
    total_amount DECIMAL(10, 2) NOT NULL COMMENT '订单总额',
    discount DECIMAL(10, 2) DEFAULT 0 COMMENT '折扣',
    actual_amount DECIMAL(10, 2) NOT NULL COMMENT '实付金额',
    status ENUM('pending', 'paid', 'shipped', 'completed', 'cancelled', 'refunded')
        DEFAULT 'pending' COMMENT '订单状态',
    shipping_city VARCHAR(50) COMMENT '收货城市',
    payment_method VARCHAR(20) COMMENT '支付方式',
    INDEX idx_user (user_id),
    INDEX idx_date (order_date),
    INDEX idx_status (status),
    INDEX idx_city (shipping_city),
    FOREIGN KEY (user_id) REFERENCES users(user_id)
) COMMENT '订单表';


-- ========== 订单明细表 ==========
CREATE TABLE order_items (
    item_id INT PRIMARY KEY AUTO_INCREMENT,
    order_id INT NOT NULL COMMENT '订单ID',
    product_id INT NOT NULL COMMENT '商品ID',
    quantity INT NOT NULL DEFAULT 1 COMMENT '数量',
    unit_price DECIMAL(10, 2) NOT NULL COMMENT '下单时单价',
    subtotal DECIMAL(10, 2) GENERATED ALWAYS AS (quantity * unit_price) STORED,
    INDEX idx_order (order_id),
    INDEX idx_product (product_id),
    FOREIGN KEY (order_id) REFERENCES orders(order_id),
    FOREIGN KEY (product_id) REFERENCES products(product_id)
) COMMENT '订单明细表';


-- ========== 评价表 ==========
CREATE TABLE reviews (
    review_id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    product_id INT NOT NULL,
    order_id INT NOT NULL,
    rating TINYINT NOT NULL CHECK (rating >= 1 AND rating <= 5),
    content TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_product (product_id),
    INDEX idx_user (user_id),
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (product_id) REFERENCES products(product_id),
    FOREIGN KEY (order_id) REFERENCES orders(order_id)
) COMMENT '评价表';


-- ========== 用户行为日志表 ==========
CREATE TABLE user_logs (
    log_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    product_id INT NOT NULL,
    action_type ENUM('view', 'cart', 'favorite', 'buy') NOT NULL,
    action_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    session_id VARCHAR(100),
    INDEX idx_user_time (user_id, action_time),
    INDEX idx_product (product_id),
    INDEX idx_action (action_type, action_time),
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (product_id) REFERENCES products(product_id)
) COMMENT '用户行为日志表';

-- 查看建表结果
SHOW TABLES;
DESCRIBE users;
DESCRIBE orders;
DESCRIBE order_items;
```

### 练习1：用Python批量生成INSERT语句（25分钟）

```python
"""
用Python脚本为电商数据库生成1000条模拟数据
"""
import random
import mysql.connector
from datetime import datetime, timedelta

# ========== 数据配置 ==========
CITIES = ["北京", "上海", "深圳", "广州", "杭州", "成都", "武汉", "南京", "西安", "重庆"]
CATEGORIES = [
    (1, "电子产品", None), (2, "服装", None), (3, "食品", None),
    (4, "图书", None), (5, "家居", None), (6, "运动", None),
    (7, "美妆", None), (8, "母婴", None), (9, "数码配件", 1),
    (10, "手机", 1),
]
PRODUCT_NAMES = {
    "电子产品": ["智能手机", "笔记本电脑", "平板电脑", "智能手表", "蓝牙耳机"],
    "服装": ["T恤", "牛仔裤", "连衣裙", "羽绒服", "运动鞋"],
    "食品": ["坚果礼盒", "巧克力", "茶叶", "咖啡豆", "蜂蜜"],
    "图书": ["Python编程", "数据科学", "机器学习", "算法导论", "SQL必知必会"],
    "家居": ["沙发", "台灯", "收纳盒", "床垫", "窗帘"],
    "运动": ["瑜伽垫", "哑铃", "跑步鞋", "篮球", "跳绳"],
    "美妆": ["口红", "面膜", "防晒霜", "粉底液", "眼影盘"],
    "母婴": ["纸尿裤", "奶粉", "婴儿车", "奶瓶", "玩具"],
}
STATUSES = ["pending", "paid", "shipped", "completed", "cancelled", "refunded"]
STATUS_WEIGHTS = [0.05, 0.10, 0.15, 0.60, 0.05, 0.05]

def generate_users(num=100):
    """生成用户INSERT语句"""
    inserts = []
    for i in range(1, num + 1):
        city = random.choice(CITIES)
        days_ago = random.randint(0, 365)
        reg_date = (datetime.now() - timedelta(days=days_ago)).strftime('%Y-%m-%d %H:%M:%S')
        gender = random.choice(['male', 'female'])
        age = random.randint(18, 60)

        sql = (f"INSERT INTO users (username, email, password_hash, phone, city, gender, age, registered_at) VALUES "
               f"('用户{i:04d}', 'user{i:04d}@example.com', 'hash_{i}', "
               f"'138{random.randint(10000000, 99999999)}', '{city}', '{gender}', {age}, '{reg_date}');")
        inserts.append(sql)
    return inserts

def generate_categories():
    """生成分类INSERT语句"""
    inserts = []
    for cat_id, name, parent_id in CATEGORIES:
        parent = str(parent_id) if parent_id else "NULL"
        inserts.append(
            f"INSERT INTO categories VALUES ({cat_id}, '{name}', {parent}, 0);"
        )
    return inserts

def generate_products(num=200):
    """生成商品INSERT语句"""
    inserts = []
    for i in range(1, num + 1):
        parent_cat = random.choice(CATEGORIES[:8])
        cat_id = parent_cat[0]
        cat_name = parent_cat[1]
        name = random.choice(PRODUCT_NAMES[cat_name]) + f" {random.choice(['基础款','升级款','Pro','Plus','Mini'])}"

        price = round(random.uniform(10, 5000), 2)
        cost = round(price * random.uniform(0.3, 0.7), 2)
        stock = random.randint(0, 1000)
        sales = random.randint(0, 500)
        rating = round(random.uniform(2.0, 5.0), 1)

        sql = (f"INSERT INTO products (product_name, category_id, price, cost, stock, sales, rating) VALUES "
               f"('{name}', {cat_id}, {price}, {cost}, {stock}, {sales}, {rating});")
        inserts.append(sql)
    return inserts

def generate_orders(users_count=100, num=300):
    """生成订单INSERT语句"""
    inserts = []
    order_item_inserts = []

    products_pool = list(range(1, 201))  # 200个商品

    for i in range(1, num + 1):
        user_id = random.randint(1, users_count)
        days_ago = random.randint(0, 90)
        order_date = (datetime.now() - timedelta(days=days_ago)).strftime('%Y-%m-%d %H:%M:%S')
        order_number = f"ORD{datetime.now().strftime('%Y%m%d')}{i:06d}"
        status = random.choices(STATUSES, weights=STATUS_WEIGHTS, k=1)[0]
        shipping_city = random.choice(CITIES)
        payment = random.choice(['alipay', 'wechat', 'credit_card'])

        # 随机1-5个商品
        item_count = random.randint(1, 5)
        selected_products = random.sample(products_pool, min(item_count, len(products_pool)))

        total = 0
        for product_id in selected_products:
            quantity = random.randint(1, 3)
            unit_price = round(random.uniform(10, 5000), 2)
            subtotal = round(quantity * unit_price, 2)
            total += subtotal

            order_item_inserts.append(
                f"INSERT INTO order_items (order_id, product_id, quantity, unit_price) VALUES "
                f"({i}, {product_id}, {quantity}, {unit_price});"
            )

        discount = round(random.uniform(0, total * 0.2), 2) if random.random() < 0.3 else 0
        actual = round(total - discount, 2)

        sql = (f"INSERT INTO orders (order_number, user_id, order_date, total_amount, discount, actual_amount, status, shipping_city, payment_method) VALUES "
               f"('{order_number}', {user_id}, '{order_date}', {total}, {discount}, {actual}, '{status}', '{shipping_city}', '{payment}');")
        inserts.append(sql)

    # 先返回订单，再返回order_items（因为有外键依赖）
    return inserts + order_item_inserts

def save_to_file(filename, statements):
    """保存SQL语句到文件"""
    with open(filename, "w", encoding="utf-8") as f:
        f.write("USE ecommerce;\n\n")
        for stmt in statements:
            f.write(stmt + "\n")
    print(f"已保存 {len(statements)} 条SQL到 {filename}")

if __name__ == "__main__":
    all_inserts = []

    # 先插入分类（无外键依赖）
    all_inserts.extend(generate_categories())
    # 再插入用户（无外键依赖）
    all_inserts.extend(generate_users(100))
    # 再插入商品（依赖分类表）
    all_inserts.extend(generate_products(200))
    # 最后插入订单和明细（依赖用户和商品表）
    all_inserts.extend(generate_orders(100, 300))

    save_to_file("seed_data.sql", all_inserts)

    print("\n生成统计:")
    print(f"  分类: 10条")
    print(f"  用户: 100条")
    print(f"  商品: 200条")
    print(f"  订单+明细: 约{300}条订单 + 对应的明细记录")

    print("\n导入方式:")
    print("  mysql -u root -p < seed_data.sql")
    print("  或在MySQL命令行中: source seed_data.sql;")
```

### 练习2：基础查询练习（20分钟）

```sql
-- 1. 查询所有用户（限制10条）
SELECT * FROM users LIMIT 10;

-- 2. 查询北京的用户数
SELECT COUNT(*) FROM users WHERE city = '北京';

-- 3. 查询价格最高的5个商品
SELECT product_name, price FROM products ORDER BY price DESC LIMIT 5;

-- 4. 查询销量超过100的商品
SELECT product_name, sales FROM products WHERE sales > 100 ORDER BY sales DESC;

-- 5. 查询本月注册的用户数
SELECT COUNT(*) FROM users WHERE registered_at >= DATE_FORMAT(NOW(), '%Y-%m-01');

-- 6. 查询各城市的用户分布
SELECT city, COUNT(*) AS cnt FROM users GROUP BY city ORDER BY cnt DESC;

-- 7. 查询商品价格区间分布
SELECT
    CASE
        WHEN price < 100 THEN '100以下'
        WHEN price BETWEEN 100 AND 500 THEN '100-500'
        WHEN price BETWEEN 501 AND 1000 THEN '501-1000'
        WHEN price BETWEEN 1001 AND 2000 THEN '1001-2000'
        ELSE '2000以上'
    END AS price_range,
    COUNT(*) AS cnt
FROM products
GROUP BY price_range
ORDER BY MIN(price);

-- 8. 查询每个订单的商品数量
SELECT o.order_id, o.order_number, COUNT(oi.item_id) AS item_count
FROM orders o
JOIN order_items oi ON o.order_id = oi.order_id
GROUP BY o.order_id, o.order_number
ORDER BY item_count DESC
LIMIT 10;

-- 9. 查询没有评价的商品（LEFT JOIN + IS NULL）
SELECT p.product_id, p.product_name
FROM products p
LEFT JOIN reviews r ON p.product_id = r.product_id
WHERE r.review_id IS NULL
LIMIT 10;

-- 10. 查询消费总额最高的5位用户
SELECT u.username, SUM(o.actual_amount) AS total_spent
FROM users u
JOIN orders o ON u.user_id = o.user_id
WHERE o.status = 'completed'
GROUP BY u.user_id, u.username
ORDER BY total_spent DESC
LIMIT 5;
```
