### 2.1 窗口函数概念（20分钟）

```sql
-- ========== 什么是窗口函数？==========

-- 普通聚合（GROUP BY）：每组返回一行
-- 窗口函数：每行都保留，在"窗口"内计算

/*
概念对比：

GROUP BY:
  原始10行 → GROUP BY city → 5行（每个城市一行）
  信息丢失：你不知道每行具体是谁

窗口函数:
  原始10行 → 窗口计算 → 仍然10行（每行多了一个计算结果）
  信息保留：每行都多了聚合统计信息
*/


-- ========== 窗口函数基本语法 ==========

/*
SELECT
    列1,
    列2,
    窗口函数() OVER (
        PARTITION BY 分区列     -- 按什么分组（可选）
        ORDER BY 排序列        -- 按什么排序（可选）
        ROWS/RANGE 窗口范围    -- 窗口大小（可选）
    ) AS 别名
FROM 表名;
*/


-- 示例：给每个商品加上品类内价格排名
SELECT
    product_name,
    category_id,
    price,
    ROW_NUMBER() OVER (PARTITION BY category_id ORDER BY price DESC) AS rank_in_category
FROM products
ORDER BY category_id, rank_in_category;


-- ========== 窗口函数的执行顺序 ==========
/*
SQL执行顺序:
FROM → WHERE → GROUP BY → HAVING → SELECT → ORDER BY
                              ↑
                    窗口函数在这里执行

所以WHERE中不能用窗口函数的结果！
需要用子查询或CTE：
SELECT * FROM (
    SELECT *, ROW_NUMBER() OVER (...) AS rn FROM t
) sub WHERE rn <= 10;
*/
```

### 2.2 排名函数（30分钟）

```sql
-- ========== ROW_NUMBER：连续排名（不重复）==========

-- 每个品类价格排名前3的商品
SELECT * FROM (
    SELECT
        product_name,
        category_id,
        price,
        ROW_NUMBER() OVER (
            PARTITION BY category_id
            ORDER BY price DESC
        ) AS rank_in_category
    FROM products
) ranked
WHERE rank_in_category <= 3
ORDER BY category_id, rank_in_category;

-- 按price DESC排名结果:
-- 价格: 5000, 3000, 3000, 1000
-- ROW_NUMBER: 1, 2, 3, 4  (即使价格相同也不重复)


-- ========== RANK：跳跃排名（同值同名次，后续跳过）==========

SELECT
    product_name,
    price,
    RANK() OVER (ORDER BY price DESC) AS rank_price,
    ROW_NUMBER() OVER (ORDER BY price DESC) AS row_num
FROM products
ORDER BY price DESC
LIMIT 10;

-- 价格: 5000, 3000, 3000, 1000
-- RANK:    1,    2,    2,    4  (同值同名次，第3名被跳过)
-- ROW_NUMBER: 1, 2, 3, 4


-- ========== DENSE_RANK：密集排名（同值同名次，不跳过）==========

SELECT
    product_name,
    price,
    RANK() OVER (ORDER BY price DESC) AS rank_price,
    DENSE_RANK() OVER (ORDER BY price DESC) AS dense_rank_price,
    ROW_NUMBER() OVER (ORDER BY price DESC) AS row_num
FROM products
ORDER BY price DESC
LIMIT 10;

-- 价格: 5000, 3000, 3000, 1000
-- RANK:       1,    2,    2,    4
-- DENSE_RANK: 1,    2,    2,    3  (第3名不被跳过)
-- ROW_NUMBER: 1,    2,    3,    4


-- ========== NTILE：分桶 ==========

-- 将商品按价格分为4等份
SELECT
    product_name,
    price,
    NTILE(4) OVER (ORDER BY price DESC) AS price_quartile
FROM products
ORDER BY price DESC;

-- 应用：将用户按消费金额分为4组（VIP/高级/中级/普通）
SELECT
    u.username,
    ROUND(SUM(o.actual_amount), 2) AS total_spent,
    NTILE(4) OVER (ORDER BY SUM(o.actual_amount) DESC) AS spending_level
FROM users u
INNER JOIN orders o ON u.user_id = o.user_id
WHERE o.status = 'completed'
GROUP BY u.user_id, u.username
ORDER BY total_spent DESC;

/*
NTILE与CASE对照：
  NTILE(4)结果:
  1 → VIP客户（前25%）
  2 → 高级客户（25%-50%）
  3 → 中级客户（50%-75%）
  4 → 普通客户（后25%）
*/
```

### 2.3 偏移函数：LAG和LEAD（25分钟）

```sql
-- ========== LAG：获取前面行的值 ==========
-- 用于计算环比、时间差等

-- 示例1：查询每日GMV与前一天对比
WITH daily_gmv AS (
    SELECT
        DATE(order_date) AS order_day,
        SUM(actual_amount) AS gmv
    FROM orders
    WHERE status = 'completed'
    GROUP BY DATE(order_date)
)
SELECT
    order_day,
    ROUND(gmv, 2) AS today_gmv,
    ROUND(LAG(gmv, 1) OVER (ORDER BY order_day), 2) AS yesterday_gmv,
    ROUND(gmv - LAG(gmv, 1) OVER (ORDER BY order_day), 2) AS change_amount,
    ROUND(
        (gmv - LAG(gmv, 1) OVER (ORDER BY order_day)) /
        NULLIF(LAG(gmv, 1) OVER (ORDER BY order_day), 0) * 100,
        2
    ) AS change_pct
FROM daily_gmv
ORDER BY order_day;


-- 示例2：用户连续下单天数分析
WITH daily_orders AS (
    SELECT DISTINCT
        user_id,
        DATE(order_date) AS order_day
    FROM orders
)
SELECT
    user_id,
    order_day,
    LAG(order_day, 1) OVER (
        PARTITION BY user_id
        ORDER BY order_day
    ) AS prev_order_day,
    DATEDIFF(
        order_day,
        LAG(order_day, 1) OVER (
            PARTITION BY user_id
            ORDER BY order_day
        )
    ) AS days_since_last_order
FROM daily_orders
ORDER BY user_id, order_day;


-- 示例3：每月销售额环比增长率
WITH monthly_revenue AS (
    SELECT
        DATE_FORMAT(order_date, '%Y-%m') AS month,
        SUM(actual_amount) AS revenue
    FROM orders
    WHERE status = 'completed'
    GROUP BY DATE_FORMAT(order_date, '%Y-%m')
)
SELECT
    month,
    ROUND(revenue, 2) AS revenue,
    ROUND(LAG(revenue) OVER (ORDER BY month), 2) AS prev_month_revenue,
    ROUND(
        (revenue - LAG(revenue) OVER (ORDER BY month)) /
        NULLIF(LAG(revenue) OVER (ORDER BY month), 0) * 100,
        2
    ) AS growth_rate
FROM monthly_revenue
ORDER BY month;


-- ========== LEAD：获取后面行的值 ==========

-- 示例：查找每个商品下一次被购买的时间
WITH product_purchase_dates AS (
    SELECT DISTINCT
        p.product_name,
        DATE(o.order_date) AS purchase_date
    FROM products p
    INNER JOIN order_items oi ON p.product_id = oi.product_id
    INNER JOIN orders o ON oi.order_id = o.order_id
)
SELECT
    product_name,
    purchase_date,
    LEAD(purchase_date, 1) OVER (
        PARTITION BY product_name
        ORDER BY purchase_date
    ) AS next_purchase_date,
    DATEDIFF(
        LEAD(purchase_date, 1) OVER (
            PARTITION BY product_name
            ORDER BY purchase_date
        ),
        purchase_date
    ) AS days_to_next_purchase
FROM product_purchase_dates
ORDER BY product_name, purchase_date;


-- ========== FIRST_VALUE / LAST_VALUE / NTH_VALUE ==========

-- 每个品类中最高价和最低价商品
SELECT
    product_name,
    category_id,
    price,
    FIRST_VALUE(product_name) OVER (
        PARTITION BY category_id ORDER BY price DESC
    ) AS highest_priced_product,
    FIRST_VALUE(price) OVER (
        PARTITION BY category_id ORDER BY price DESC
    ) AS highest_price,
    LAST_VALUE(product_name) OVER (
        PARTITION BY category_id ORDER BY price DESC
        ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
    ) AS lowest_priced_product
FROM products
ORDER BY category_id, price DESC;
```

### 2.4 聚合窗口函数（25分钟）

```sql
-- ========== SUM OVER：累计求和 ==========

-- 每日累计GMV
WITH daily_gmv AS (
    SELECT
        DATE(order_date) AS order_day,
        SUM(actual_amount) AS daily_amount
    FROM orders
    WHERE status = 'completed'
    GROUP BY DATE(order_date)
)
SELECT
    order_day,
    ROUND(daily_amount, 2) AS daily_gmv,
    ROUND(SUM(daily_amount) OVER (ORDER BY order_day), 2) AS cumulative_gmv
FROM daily_gmv
ORDER BY order_day;


-- ========== 移动平均 ==========

-- 7天移动平均GMV
WITH daily_gmv AS (
    SELECT
        DATE(order_date) AS order_day,
        SUM(actual_amount) AS daily_amount
    FROM orders
    WHERE status = 'completed'
    GROUP BY DATE(order_day)
)
SELECT
    order_day,
    ROUND(daily_amount, 2) AS daily_gmv,
    ROUND(
        AVG(daily_amount) OVER (
            ORDER BY order_day
            ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
        ), 2
    ) AS ma_7day
FROM daily_gmv
ORDER BY order_day;


-- ========== ROWS vs RANGE ==========

-- ROWS: 按物理行数计算窗口
-- RANGE: 按值的范围计算窗口

/*
窗口范围定义语法:
  ROWS/RANGE BETWEEN
    起点 {N PRECEDING | UNBOUNDED PRECEDING | CURRENT ROW}
    AND
    终点 {N FOLLOWING | UNBOUNDED FOLLOWING | CURRENT ROW}

常用模式:
  ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    → 从第一行到当前行（累计）

  ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
    → 当前行+前6行（7天移动平均）

  ROWS BETWEEN 1 PRECEDING AND 1 FOLLOWING
    → 当前行+前一行+后一行
*/


-- ========== 实际应用 ==========

-- 查询1：每个商品品类中，商品价格占品类总价的百分比
SELECT
    product_name,
    category_id,
    price,
    ROUND(
        price / SUM(price) OVER (PARTITION BY category_id) * 100,
        2
    ) AS pct_of_category
FROM products
ORDER BY category_id, pct_of_category DESC;


-- 查询2：查询每个用户订单的累计金额和累计订单数
SELECT
    u.username,
    o.order_id,
    o.order_date,
    o.actual_amount,
    SUM(o.actual_amount) OVER (
        PARTITION BY o.user_id
        ORDER BY o.order_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS cumulative_amount,
    COUNT(o.order_id) OVER (
        PARTITION BY o.user_id
        ORDER BY o.order_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS cumulative_orders
FROM orders o
INNER JOIN users u ON o.user_id = u.user_id
WHERE o.status = 'completed'
ORDER BY o.user_id, o.order_date;
```

### 2.5 窗口函数实战案例（20分钟）

```sql
-- ========== 案例1：用户留存分析（新用户注册后N天回访）==========

-- 找出注册后第1/3/7/30天有下单的用户数
WITH user_first_order AS (
    SELECT
        u.user_id,
        u.registered_at,
        MIN(o.order_date) AS first_order_date,
        DATEDIFF(MIN(o.order_date), u.registered_at) AS days_to_first_order
    FROM users u
    LEFT JOIN orders o ON u.user_id = o.user_id AND o.status = 'completed'
    GROUP BY u.user_id, u.registered_at
)
SELECT
    CASE
        WHEN days_to_first_order IS NULL THEN '从未下单'
        WHEN days_to_first_order <= 1 THEN '当天'
        WHEN days_to_first_order <= 3 THEN '1-3天'
        WHEN days_to_first_order <= 7 THEN '4-7天'
        WHEN days_to_first_order <= 30 THEN '8-30天'
        ELSE '30天以上'
    END AS first_order_bucket,
    COUNT(*) AS user_count,
    ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER(), 2) AS pct
FROM user_first_order
GROUP BY first_order_bucket
ORDER BY MIN(days_to_first_order);


-- ========== 案例2：用户消费节奏分析 ==========

-- 每个用户的平均下单间隔
WITH user_orders AS (
    SELECT
        user_id,
        order_date,
        LAG(order_date) OVER (
            PARTITION BY user_id ORDER BY order_date
        ) AS prev_date
    FROM orders
    WHERE status = 'completed'
)
SELECT
    u.username,
    COUNT(*) AS order_count,
    ROUND(
        AVG(DATEDIFF(order_date, prev_date)), 1
    ) AS avg_days_between_orders,
    DATEDIFF(MAX(order_date), MIN(order_date)) AS active_days
FROM user_orders uo
INNER JOIN users u ON uo.user_id = u.user_id
WHERE prev_date IS NOT NULL
GROUP BY uo.user_id, u.username
HAVING order_count >= 3
ORDER BY avg_days_between_orders
LIMIT 20;


-- ========== 案例3：销量趋势异常检测 ==========

-- 检测销量突然下降的商品
WITH product_daily_sales AS (
    SELECT
        p.product_name,
        DATE(o.order_date) AS order_day,
        COUNT(DISTINCT oi.order_id) AS sales_count
    FROM products p
    INNER JOIN order_items oi ON p.product_id = oi.product_id
    INNER JOIN orders o ON oi.order_id = o.order_id
    WHERE o.status = 'completed'
    GROUP BY p.product_id, p.product_name, DATE(o.order_date)
),
with_comparison AS (
    SELECT
        product_name,
        order_day,
        sales_count,
        LAG(sales_count, 1) OVER (
            PARTITION BY product_name ORDER BY order_day
        ) AS prev_day_sales,
        AVG(sales_count) OVER (
            PARTITION BY product_name
            ORDER BY order_day
            ROWS BETWEEN 6 PRECEDING AND 1 PRECEDING
        ) AS avg_7day_sales
    FROM product_daily_sales
)
SELECT
    product_name,
    order_day,
    sales_count,
    ROUND(prev_day_sales, 1) AS prev_day,
    ROUND(avg_7day_sales, 1) AS avg_7day,
    CASE
        WHEN sales_count < avg_7day_sales * 0.5 THEN '⚠️ 严重下降'
        WHEN sales_count < avg_7day_sales * 0.7 THEN '⚡ 轻微下降'
        ELSE '✅ 正常'
    END AS alert
FROM with_comparison
WHERE prev_day_sales IS NOT NULL
  AND avg_7day_sales IS NOT NULL
  AND sales_count < avg_7day_sales * 0.7
ORDER BY order_day DESC, product_name
LIMIT 20;


-- ========== 案例4：去重问题 ==========

-- 获取每个用户最近一笔订单（用ROW_NUMBER去重）
SELECT user_id, order_id, order_date, actual_amount
FROM (
    SELECT
        user_id,
        order_id,
        order_date,
        actual_amount,
        ROW_NUMBER() OVER (
            PARTITION BY user_id
            ORDER BY order_date DESC
        ) AS rn
    FROM orders
    WHERE status = 'completed'
) latest
WHERE rn = 1
ORDER BY user_id;


-- ========== 案例5：同比/环比计算 ==========

-- 每周GMV的周环比
WITH weekly_gmv AS (
    SELECT
        YEARWEEK(order_date, 1) AS year_week,
        SUM(actual_amount) AS gmv
    FROM orders
    WHERE status = 'completed'
    GROUP BY YEARWEEK(order_date, 1)
)
SELECT
    year_week,
    ROUND(gmv, 2) AS gmv,
    ROUND(LAG(gmv) OVER (ORDER BY year_week), 2) AS prev_week_gmv,
    ROUND(
        (gmv - LAG(gmv) OVER (ORDER BY year_week)) /
        NULLIF(LAG(gmv) OVER (ORDER BY year_week), 0) * 100,
        2
    ) AS wow_change_pct
FROM weekly_gmv
ORDER BY year_week;
```

### 练习：窗口函数综合练习

```sql
-- 练习1：给每个城市中消费最高的前3名用户排名
-- 要求：使用ROW_NUMBER + PARTITION BY

SELECT * FROM (
    SELECT
        u.city,
        u.username,
        ROUND(SUM(o.actual_amount), 2) AS total_spent,
        ROW_NUMBER() OVER (
            PARTITION BY u.city
            ORDER BY SUM(o.actual_amount) DESC
        ) AS city_rank
    FROM users u
    INNER JOIN orders o ON u.user_id = o.user_id
    WHERE o.status = 'completed'
    GROUP BY u.city, u.user_id, u.username
) ranked
WHERE city_rank <= 3
ORDER BY city, city_rank;

-- 练习2：查询每个商品品类中价格排名前3的商品（用RANK）
-- 要求：使用RANK（允许并列，但要控制只显示前3名的值）

SELECT * FROM (
    SELECT
        category_id,
        product_name,
        price,
        RANK() OVER (PARTITION BY category_id ORDER BY price DESC) AS rnk
    FROM products
) ranked
WHERE rnk <= 3
ORDER BY category_id, rnk;

-- 练习3：计算每个用户订单金额的3单移动平均
SELECT
    u.username,
    o.order_date,
    o.actual_amount,
    ROUND(
        AVG(o.actual_amount) OVER (
            PARTITION BY o.user_id
            ORDER BY o.order_date
            ROWS BETWEEN 2 PRECEDING AND CURRENT ROW
        ), 2
    ) AS ma3
FROM orders o
INNER JOIN users u ON o.user_id = u.user_id
WHERE o.status = 'completed'
ORDER BY o.user_id, o.order_date;

-- 练习4：找出每个城市第一个注册的用户
SELECT city, username, registered_at
FROM (
    SELECT
        city,
        username,
        registered_at,
        ROW_NUMBER() OVER (PARTITION BY city ORDER BY registered_at) AS rn
    FROM users
) first_users
WHERE rn = 1
ORDER BY city;

-- 练习5：计算用户连续下单天数（最难）
WITH user_dates AS (
    SELECT DISTINCT
        user_id,
        DATE(order_date) AS order_day
    FROM orders
    WHERE status = 'completed'
),
with_lag AS (
    SELECT
        user_id,
        order_day,
        LAG(order_day) OVER (PARTITION BY user_id ORDER BY order_day) AS prev_day
    FROM user_dates
),
streak_groups AS (
    SELECT
        user_id,
        order_day,
        SUM(CASE
            WHEN DATEDIFF(order_day, prev_day) = 1 OR prev_day IS NULL
            THEN 0 ELSE 1
        END) OVER (PARTITION BY user_id ORDER BY order_day) AS grp
    FROM with_lag
)
SELECT
    user_id,
    MIN(order_day) AS streak_start,
    MAX(order_day) AS streak_end,
    COUNT(*) AS streak_days
FROM streak_groups
GROUP BY user_id, grp
HAVING COUNT(*) >= 3
ORDER BY streak_days DESC
LIMIT 10;
```
