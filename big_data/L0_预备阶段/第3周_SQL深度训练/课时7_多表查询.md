### 2.1 JOIN全家桶（40分钟）

```sql
-- ========== 数据准备 ==========
-- 假设已有以下表和数据：

-- departments表
-- +----+-----------+
-- | id | name      |
-- +----+-----------+
-- | 1  | 技术部    |
-- | 2  | 市场部    |
-- | 3  | 人事部    |
-- | 4  | 财务部    |
-- +----+-----------+

-- employees表
-- +----+--------+--------+
-- | id | name   | dept_id|
-- +----+--------+--------+
-- | 1  | 张三   | 1      |
-- | 2  | 李四   | 1      |
-- | 3  | 王五   | 2      |
-- | 4  | 赵六   | NULL   |
-- | 5  | 孙七   | 5      |
-- +----+--------+--------+


-- ========== INNER JOIN（内连接）==========
-- 只返回两表中匹配的行

SELECT e.name AS 员工, d.name AS 部门
FROM employees e
INNER JOIN departments d ON e.dept_id = d.id;

-- 结果: 张三/技术部, 李四/技术部, 王五/市场部
-- 赵六(dept_id=NULL) 和 孙七(dept_id=5) 不出现


-- ========== LEFT JOIN（左连接）==========
-- 返回左表所有行 + 右表匹配行（不匹配则为NULL）

SELECT e.name AS 员工, d.name AS 部门
FROM employees e
LEFT JOIN departments d ON e.dept_id = d.id;

-- 结果:
-- 张三/技术部, 李四/技术部, 王五/市场部
-- 赵六/NULL, 孙七/NULL (左表全保留，右表无匹配为NULL)


-- ========== RIGHT JOIN（右连接）==========
-- 返回右表所有行 + 左表匹配行（不匹配则为NULL）

SELECT e.name AS 员工, d.name AS 部门
FROM employees e
RIGHT JOIN departments d ON e.dept_id = d.id;

-- 结果:
-- 张三/技术部, 李四/技术部, 王五/市场部
-- NULL/人事部, NULL/财务部 (右表全保留，左表无匹配为NULL)


-- ========== FULL JOIN（全连接）==========
-- MySQL不支持FULL JOIN，用LEFT JOIN + UNION + RIGHT JOIN模拟

SELECT e.name AS 员工, d.name AS 部门
FROM employees e
LEFT JOIN departments d ON e.dept_id = d.id

UNION

SELECT e.name AS 员工, d.name AS 部门
FROM employees e
RIGHT JOIN departments d ON e.dept_id = d.id;

-- 结果: 所有员工和所有部门都出现


-- ========== CROSS JOIN（交叉连接）==========
-- 笛卡尔积：左表每行 × 右表每行

SELECT e.name, d.name
FROM employees e
CROSS JOIN departments d;
-- 结果: 5个员工 × 4个部门 = 20行


-- ========== 多表JOIN ==========
-- 查询订单详情，包含用户信息、商品信息

SELECT
    o.order_id,
    o.order_date,
    u.username,
    p.product_name,
    oi.quantity,
    oi.unit_price,
    oi.quantity * oi.unit_price AS subtotal
FROM orders o
INNER JOIN users u ON o.user_id = u.user_id
INNER JOIN order_items oi ON o.order_id = oi.order_id
INNER JOIN products p ON oi.product_id = p.product_id
WHERE o.status = 'completed'
ORDER BY o.order_date DESC
LIMIT 10;
```

### 2.2 自连接（Self JOIN）（15分钟）

```sql
-- ========== 自连接概念 ==========
-- 自连接就是一张表跟自己JOIN

-- 场景1：查找同一城市的用户
SELECT
    u1.username AS 用户1,
    u2.username AS 用户2,
    u1.city
FROM users u1
INNER JOIN users u2
    ON u1.city = u2.city
    AND u1.user_id < u2.user_id   -- 避免重复配对和自己配自己
ORDER BY u1.city;


-- 场景2：商品的连带购买分析（买了A也买了B）
SELECT
    oi1.product_id AS product_a,
    oi2.product_id AS product_b,
    COUNT(DISTINCT oi1.order_id) AS co_purchase_count
FROM order_items oi1
INNER JOIN order_items oi2
    ON oi1.order_id = oi2.order_id       -- 同一订单
    AND oi1.product_id < oi2.product_id   -- 避免重复组合
GROUP BY oi1.product_id, oi2.product_id
HAVING co_purchase_count >= 3              -- 至少3次连带购买
ORDER BY co_purchase_count DESC
LIMIT 20;


-- 场景3：查找同一分类下价格高于平均价的商品
SELECT
    p1.product_name,
    p1.price,
    ROUND(avg_p.avg_price, 2) AS category_avg,
    ROUND(p1.price - avg_p.avg_price, 2) AS diff
FROM products p1
INNER JOIN (
    SELECT category_id, AVG(price) AS avg_price
    FROM products
    GROUP BY category_id
) avg_p ON p1.category_id = avg_p.category_id
WHERE p1.price > avg_p.avg_price
ORDER BY diff DESC
LIMIT 10;
```

### 2.3 子查询（30分钟）

```sql
-- ========== 标量子查询（返回单个值）==========

-- 查询价格高于平均价的商品
SELECT product_name, price
FROM products
WHERE price > (SELECT AVG(price) FROM products);

-- 查询消费最高的用户
SELECT username, email
FROM users
WHERE user_id = (
    SELECT user_id
    FROM orders
    WHERE status = 'completed'
    GROUP BY user_id
    ORDER BY SUM(actual_amount) DESC
    LIMIT 1
);


-- ========== 行子查询（返回一行）==========

-- 查询与"用户0001"同城的其他用户
SELECT username, city
FROM users
WHERE (city) = (
    SELECT city FROM users WHERE username = '用户0001'
) AND username != '用户0001';


-- ========== 表子查询（返回多行多列）==========

-- 查询有过消费记录的用户（使用IN）
SELECT username, city
FROM users
WHERE user_id IN (
    SELECT DISTINCT user_id FROM orders
);

-- 查询没有消费记录的用户（使用NOT IN）
SELECT username, city
FROM users
WHERE user_id NOT IN (
    SELECT DISTINCT user_id FROM orders
);


-- ========== EXISTS子查询 ==========

-- EXISTS: 检查子查询是否返回任何行
-- 查询至少有一笔已完成订单的用户
SELECT username
FROM users u
WHERE EXISTS (
    SELECT 1 FROM orders o
    WHERE o.user_id = u.user_id AND o.status = 'completed'
);

-- NOT EXISTS: 查询从未下单的用户
SELECT username
FROM users u
WHERE NOT EXISTS (
    SELECT 1 FROM orders o
    WHERE o.user_id = u.user_id
);


-- ========== 派生表（FROM子句中的子查询）==========

-- 查询各用户的最近一笔订单
SELECT u.username, latest.*
FROM users u
INNER JOIN (
    SELECT user_id, MAX(order_date) AS last_order_date
    FROM orders
    GROUP BY user_id
) latest ON u.user_id = latest.user_id;


-- ========== 关联子查询 vs 非关联子查询 ==========

-- 非关联子查询：子查询独立执行
SELECT * FROM products
WHERE price > (SELECT AVG(price) FROM products);
-- 子查询只需执行一次

-- 关联子查询：子查询引用外部查询的列
SELECT product_name, price, category_id
FROM products p1
WHERE price > (
    SELECT AVG(price)
    FROM products p2
    WHERE p2.category_id = p1.category_id  -- 引用外部列
);
-- 子查询对外部每一行都要执行一次（性能较低）
```

### 2.5 GROUP BY与聚合函数深入（25分钟）

```sql
-- ========== GROUP BY 进阶 ==========

-- 多字段分组
SELECT
    DATE_FORMAT(order_date, '%Y-%m') AS month,
    status,
    COUNT(*) AS cnt,
    SUM(actual_amount) AS total
FROM orders
GROUP BY month, status
ORDER BY month, status;


-- ========== HAVING vs WHERE ==========
-- WHERE: 在分组前过滤行（不能使用聚合函数）
-- HAVING: 在分组后过滤组（可以使用聚合函数）

-- 错误写法：
-- SELECT city, COUNT(*) FROM users WHERE COUNT(*) > 5 GROUP BY city;  -- 报错！

-- 正确写法：
SELECT city, COUNT(*) AS user_count
FROM users
WHERE is_active = TRUE           -- 分组前过滤
GROUP BY city
HAVING COUNT(*) > 5              -- 分组后过滤
ORDER BY user_count DESC;


-- ========== ROLLUP 汇总 ==========
-- 生成多级汇总行

SELECT
    COALESCE(city, '全部城市') AS city,
    COUNT(*) AS user_count
FROM users
GROUP BY city WITH ROLLUP;

-- 结果会在最后多一行"全部城市"的总计


-- ========== GROUP_CONCAT 字符串聚合 ==========
-- 将分组内的值拼接为字符串

SELECT
    city,
    COUNT(*) AS user_count,
    GROUP_CONCAT(username ORDER BY username SEPARATOR ', ') AS users_list
FROM users
GROUP BY city
ORDER BY user_count DESC
LIMIT 5;
-- 结果示例:
-- 北京 | 15 | 用户0001, 用户0005, 用户0012, ...


-- ========== 常用聚合函数总结 ==========
-- COUNT(*):    行数（含NULL）
-- COUNT(col):  非NULL值的数量
-- COUNT(DISTINCT col): 唯一值数量
-- SUM(col):    总和
-- AVG(col):    平均值
-- MAX(col):    最大值
-- MIN(col):    最小值
-- GROUP_CONCAT(col): 字符串拼接
-- STDDEV(col): 标准差
-- VARIANCE(col): 方差
```

### 2.6 电商分析核心查询（30分钟）

```sql
-- ========================================
-- 电商分析核心查询集（面试高频题）
-- ========================================

-- ========== 用户分析 ==========

-- 查询1：各城市的用户消费TOP5
SELECT
    u.city,
    COUNT(DISTINCT o.user_id) AS user_count,
    COUNT(o.order_id) AS order_count,
    ROUND(SUM(o.actual_amount), 2) AS total_revenue,
    ROUND(AVG(o.actual_amount), 2) AS avg_order_value,
    ROUND(SUM(o.actual_amount) / COUNT(DISTINCT o.user_id), 2) AS arpu
FROM users u
INNER JOIN orders o ON u.user_id = o.user_id
WHERE o.status = 'completed'
GROUP BY u.city
ORDER BY total_revenue DESC
LIMIT 5;


-- 查询2：找出"沉默用户"（注册超过30天但从未下单）
SELECT
    u.user_id,
    u.username,
    u.registered_at,
    DATEDIFF(NOW(), u.registered_at) AS days_since_register
FROM users u
LEFT JOIN orders o ON u.user_id = o.user_id
WHERE o.order_id IS NULL
  AND u.registered_at < DATE_SUB(NOW(), INTERVAL 30 DAY)
ORDER BY days_since_register DESC
LIMIT 20;


-- 查询3：高价值用户画像（消费TOP 10%）
WITH user_spending AS (
    SELECT
        user_id,
        SUM(actual_amount) AS total_spent,
        COUNT(order_id) AS order_count
    FROM orders
    WHERE status = 'completed'
    GROUP BY user_id
),
percentile AS (
    SELECT
        PERCENT_RANK() OVER (ORDER BY total_spent) AS pct_rank,
        user_id,
        total_spent,
        order_count
    FROM user_spending
)
SELECT
    u.username,
    u.city,
    u.age,
    ROUND(p.total_spent, 2) AS total_spent,
    p.order_count
FROM percentile p
INNER JOIN users u ON p.user_id = u.user_id
WHERE p.pct_rank >= 0.9
ORDER BY p.total_spent DESC;


-- ========== 商品分析 ==========

-- 查询4：各品类销售额排名及占比
SELECT
    c.category_name,
    COUNT(DISTINCT oi.product_id) AS product_count,
    SUM(oi.quantity) AS total_qty,
    ROUND(SUM(oi.quantity * oi.unit_price), 2) AS total_revenue,
    ROUND(
        SUM(oi.quantity * oi.unit_price) /
        (SELECT SUM(quantity * unit_price) FROM order_items
         INNER JOIN orders ON order_items.order_id = orders.order_id
         WHERE orders.status = 'completed') * 100, 2
    ) AS revenue_pct
FROM categories c
INNER JOIN products p ON c.category_id = p.category_id
INNER JOIN order_items oi ON p.product_id = oi.product_id
INNER JOIN orders o ON oi.order_id = o.order_id
WHERE o.status = 'completed'
GROUP BY c.category_id, c.category_name
ORDER BY total_revenue DESC;


-- 查询5：商品连带购买TOP组合
SELECT
    p1.product_name AS product_a,
    p2.product_name AS product_b,
    combo.co_purchase_count
FROM (
    SELECT
        oi1.product_id AS pid1,
        oi2.product_id AS pid2,
        COUNT(DISTINCT oi1.order_id) AS co_purchase_count
    FROM order_items oi1
    INNER JOIN order_items oi2
        ON oi1.order_id = oi2.order_id
        AND oi1.product_id < oi2.product_id
    INNER JOIN orders o ON oi1.order_id = o.order_id
    WHERE o.status = 'completed'
    GROUP BY oi1.product_id, oi2.product_id
    HAVING co_purchase_count >= 5
) combo
INNER JOIN products p1 ON combo.pid1 = p1.product_id
INNER JOIN products p2 ON combo.pid2 = p2.product_id
ORDER BY combo.co_purchase_count DESC
LIMIT 10;


-- 查询6：库存积压分析（库存高但销量低的商品）
SELECT
    p.product_name,
    p.stock,
    p.sales,
    p.price,
    ROUND(p.stock * p.price, 2) AS stock_value,
    CASE
        WHEN p.sales = 0 THEN '无销量'
        WHEN p.stock / NULLIF(p.sales, 0) > 10 THEN '严重积压'
        WHEN p.stock / NULLIF(p.sales, 0) > 5 THEN '轻度积压'
        ELSE '正常'
    END AS stock_status
FROM products p
WHERE p.is_on_sale = TRUE
ORDER BY stock_value DESC
LIMIT 10;


-- ========== 订单分析 ==========

-- 查询7：每日GMV趋势（最近30天）
SELECT
    DATE(order_date) AS order_day,
    COUNT(order_id) AS order_count,
    ROUND(SUM(actual_amount), 2) AS gmv,
    ROUND(AVG(actual_amount), 2) AS avg_order_value
FROM orders
WHERE status = 'completed'
  AND order_date >= DATE_SUB(CURDATE(), INTERVAL 30 DAY)
GROUP BY DATE(order_date)
ORDER BY order_day;


-- 查询8：客单价分布分析
SELECT
    CASE
        WHEN actual_amount < 50 THEN '50以下'
        WHEN actual_amount BETWEEN 50 AND 100 THEN '50-100'
        WHEN actual_amount BETWEEN 101 AND 200 THEN '101-200'
        WHEN actual_amount BETWEEN 201 AND 500 THEN '201-500'
        WHEN actual_amount BETWEEN 501 AND 1000 THEN '501-1000'
        ELSE '1000以上'
    END AS amount_range,
    COUNT(*) AS order_count,
    ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER(), 2) AS pct
FROM orders
WHERE status = 'completed'
GROUP BY amount_range
ORDER BY MIN(actual_amount);


-- 查询9：复购率月度趋势
WITH user_monthly AS (
    SELECT
        user_id,
        DATE_FORMAT(order_date, '%Y-%m') AS month,
        COUNT(order_id) AS order_count
    FROM orders
    WHERE status = 'completed'
    GROUP BY user_id, DATE_FORMAT(order_date, '%Y-%m')
)
SELECT
    month,
    COUNT(DISTINCT user_id) AS total_users,
    COUNT(DISTINCT CASE WHEN order_count >= 2 THEN user_id END) AS repeat_users,
    ROUND(
        COUNT(DISTINCT CASE WHEN order_count >= 2 THEN user_id END) * 100.0 /
        COUNT(DISTINCT user_id), 2
    ) AS repeat_rate
FROM user_monthly
GROUP BY month
ORDER BY month;


-- 查询10：各城市/时段订单特征
SELECT
    u.city,
    HOUR(o.order_date) AS order_hour,
    COUNT(o.order_id) AS order_count,
    ROUND(AVG(o.actual_amount), 2) AS avg_amount
FROM orders o
INNER JOIN users u ON o.user_id = u.user_id
WHERE o.status = 'completed'
GROUP BY u.city, HOUR(o.order_date)
ORDER BY order_count DESC
LIMIT 20;
```

```sql
-- 题目1：查询每个用户的总消费金额，按降序排列（使用JOIN + GROUP BY）
-- 答案：
SELECT u.username, ROUND(SUM(o.actual_amount), 2) AS total_spent
FROM users u
INNER JOIN orders o ON u.user_id = o.user_id
WHERE o.status = 'completed'
GROUP BY u.user_id, u.username
ORDER BY total_spent DESC;

-- 题目2：找出评价平均分最高的前10个商品（JOIN + AVG + GROUP BY）
-- 答案：
SELECT p.product_name, ROUND(AVG(r.rating), 2) AS avg_rating, COUNT(r.review_id) AS review_count
FROM products p
INNER JOIN reviews r ON p.product_id = r.product_id
GROUP BY p.product_id, p.product_name
ORDER BY avg_rating DESC, review_count DESC
LIMIT 10;

-- 题目3：查询每个品类中价格最高的商品（子查询）
-- 答案：
SELECT p.product_name, c.category_name, p.price
FROM products p
INNER JOIN categories c ON p.category_id = c.category_id
WHERE (p.category_id, p.price) IN (
    SELECT category_id, MAX(price)
    FROM products
    GROUP BY category_id
)
ORDER BY p.price DESC;

-- 题目4：找出从未被评价的商品（LEFT JOIN + IS NULL）
-- 答案：
SELECT p.product_name, p.price
FROM products p
LEFT JOIN reviews r ON p.product_id = r.product_id
WHERE r.review_id IS NULL;

-- 题目5：查询连续3个月（30天内）没有下单的用户
-- 答案：
SELECT u.username, u.email
FROM users u
LEFT JOIN orders o ON u.user_id = o.user_id
    AND o.order_date >= DATE_SUB(NOW(), INTERVAL 30 DAY)
WHERE o.order_id IS NULL;

-- 题目6：计算每个商品类别的平均客单价
-- 答案：
SELECT
    c.category_name,
    ROUND(AVG(o.actual_amount), 2) AS avg_order_value,
    COUNT(DISTINCT o.order_id) AS order_count
FROM categories c
INNER JOIN products p ON c.category_id = p.category_id
INNER JOIN order_items oi ON p.product_id = oi.product_id
INNER JOIN orders o ON oi.order_id = o.order_id
WHERE o.status = 'completed'
GROUP BY c.category_id, c.category_name
ORDER BY avg_order_value DESC;

-- 题目7：找出购买力最高的城市（按人均消费）
-- 答案：
SELECT
    u.city,
    COUNT(DISTINCT u.user_id) AS user_count,
    ROUND(SUM(o.actual_amount), 2) AS total_revenue,
    ROUND(SUM(o.actual_amount) / COUNT(DISTINCT u.user_id), 2) AS avg_per_user
FROM users u
INNER JOIN orders o ON u.user_id = o.user_id
WHERE o.status = 'completed'
GROUP BY u.city
HAVING user_count >= 3
ORDER BY avg_per_user DESC;

-- 题目8：查询下单数量最多的前10个商品（JOIN + COUNT + GROUP BY）
-- 答案：
SELECT p.product_name, COUNT(DISTINCT oi.order_id) AS order_count, SUM(oi.quantity) AS total_qty
FROM products p
INNER JOIN order_items oi ON p.product_id = oi.product_id
INNER JOIN orders o ON oi.order_id = o.order_id
WHERE o.status = 'completed'
GROUP BY p.product_id, p.product_name
ORDER BY order_count DESC
LIMIT 10;

-- 题目9：使用UNION合并新用户（30天内注册）和活跃用户（30天内有订单）
-- 答案：
SELECT user_id, username, 'new_user' AS user_type
FROM users
WHERE registered_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)
UNION
SELECT DISTINCT u.user_id, u.username, 'active_user' AS user_type
FROM users u
INNER JOIN orders o ON u.user_id = o.user_id
WHERE o.order_date >= DATE_SUB(NOW(), INTERVAL 30 DAY);

-- 题目10：查询每个订单包含的商品种类数（派生表子查询）
-- 答案：
SELECT o.order_id, o.order_number, o.order_date,
    (SELECT COUNT(DISTINCT product_id) FROM order_items WHERE order_id = o.order_id) AS product_types,
    (SELECT SUM(quantity) FROM order_items WHERE order_id = o.order_id) AS total_items,
    o.actual_amount
FROM orders o
WHERE o.status = 'completed'
ORDER BY product_types DESC
LIMIT 10;
```

-- 挑战题：实现RFM分析模型
-- RFM = Recency（最近一次购买时间）+ Frequency（购买频率）+ Monetary（消费金额）

WITH rfm_raw AS (
    SELECT
        user_id,
        DATEDIFF(NOW(), MAX(order_date)) AS recency,
        COUNT(order_id) AS frequency,
        SUM(actual_amount) AS monetary
    FROM orders
    WHERE status = 'completed'
    GROUP BY user_id
),
rfm_scored AS (
    SELECT
        user_id,
        NTILE(4) OVER (ORDER BY recency DESC) AS r_score,  -- 越小越好
        NTILE(4) OVER (ORDER BY frequency) AS f_score,      -- 越大越好
        NTILE(4) OVER (ORDER BY monetary) AS m_score       -- 越大越好
    FROM rfm_raw
)
SELECT
    u.username,
    r.r_score AS R,
    r.f_score AS F,
    r.m_score AS M,
    CONCAT(r.r_score, r.f_score, r.m_score) AS rfm_cell,
    CASE
        WHEN r.r_score >= 3 AND r.f_score >= 3 AND r.m_score >= 3 THEN '重要价值客户'
        WHEN r.r_score >= 3 AND r.f_score >= 3 AND r.m_score < 3 THEN '重要发展客户'
        WHEN r.r_score >= 3 AND r.f_score < 3 AND r.m_score >= 3 THEN '重要保持客户'
        WHEN r.r_score < 3 AND r.f_score >= 3 AND r.m_score >= 3 THEN '重要挽留客户'
        WHEN r.r_score >= 3 AND r.f_score < 3 AND r.m_score < 3 THEN '一般价值客户'
        WHEN r.r_score < 3 AND r.f_score >= 3 AND r.m_score < 3 THEN '一般发展客户'
        ELSE '流失客户'
    END AS customer_segment
FROM rfm_scored r
INNER JOIN users u ON r.user_id = u.user_id
ORDER BY rfm_cell DESC;
