> **项目时长**：12小时（约2.5天）
> **项目类型**：独立完成（纯命令行环境）

## 一、项目概述

完全在命令行环境下完成一系列运维任务。本项目考察Linux实战能力，所有操作必须在纯命令行完成（不使用GUI工具）。

#### 2.1 安装配置Nginx

```bash
# 安装Nginx
sudo apt install nginx -y
sudo systemctl enable nginx
sudo systemctl start nginx

# 验证
curl http://localhost
# 应该看到 "Welcome to nginx!"

# 配置反向代理
sudo vim /etc/nginx/sites-available/reverse-proxy

# 添加以下配置:
server {
    listen 80;
    server_name localhost;

    # 静态文件
    location / {
        root /var/www/html;
        index index.html;
    }

    # 反向代理到Python应用
    location /api/ {
        proxy_pass http://127.0.0.1:5000/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # 访问日志
    access_log /var/log/nginx/reverse-proxy-access.log;
    error_log /var/log/nginx/reverse-proxy-error.log;
}

# 启用站点
sudo ln -s /etc/nginx/sites-available/reverse-proxy /etc/nginx/sites-enabled/
sudo nginx -t                    # 测试配置
sudo systemctl reload nginx      # 重载配置
```

#### 3.2 系统监控脚本

```bash
# 创建监控脚本
vim /opt/scripts/system_monitor.sh

#!/bin/bash
# system_monitor.sh - 系统资源监控脚本

LOG_DIR="/var/log/system_monitor"
mkdir -p $LOG_DIR

LOG_FILE="$LOG_DIR/monitor_$(date +%Y%m%d).log"
ALERT_FILE="$LOG_DIR/alerts_$(date +%Y%m%d).log"

# 配置阈值
CPU_THRESHOLD=80
MEM_THRESHOLD=85
DISK_THRESHOLD=85

timestamp() {
    date '+%Y-%m-%d %H:%M:%S'
}

# CPU使用率
CPU_USAGE=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1)

# 内存使用率
MEM_INFO=$(free | grep Mem)
MEM_TOTAL=$(echo $MEM_INFO | awk '{print $2}')
MEM_USED=$(echo $MEM_INFO | awk '{print $3}')
MEM_USAGE=$(echo "scale=2; $MEM_USED / $MEM_TOTAL * 100" | bc)

# 磁盘使用率
DISK_USAGE=$(df -h / | awk 'NR==2 {print $5}' | sed 's/%//')

# 系统负载
LOAD=$(uptime | awk -F'load average:' '{print $2}' | cut -d',' -f1 | xargs)

# 写入日志
echo "[$(timestamp)] CPU:${CPU_USAGE}% MEM:${MEM_USAGE}% DISK:${DISK_USAGE}% LOAD:${LOAD}" >> $LOG_FILE

# 告警检查
if (( $(echo "$CPU_USAGE > $CPU_THRESHOLD" | bc -l) )); then
    echo "[$(timestamp)] ⚠️ ALERT: CPU使用率 ${CPU_USAGE}% (阈值: ${CPU_THRESHOLD}%)" >> $ALERT_FILE
fi

if (( $(echo "$MEM_USAGE > $MEM_THRESHOLD" | bc -l) )); then
    echo "[$(timestamp)] ⚠️ ALERT: 内存使用率 ${MEM_USAGE}% (阈值: ${MEM_THRESHOLD}%)" >> $ALERT_FILE
fi

if (( DISK_USAGE > DISK_THRESHOLD )); then
    echo "[$(timestamp)] ⚠️ ALERT: 磁盘使用率 ${DISK_USAGE}% (阈值: ${DISK_THRESHOLD}%)" >> $ALERT_FILE
fi

# 内存详细信息（每30分钟记录一次）
if [ $(date +%M) -eq 0 ] || [ $(date +%M) -eq 30 ]; then
    echo "=== 内存详情 ($(timestamp)) ===" >> $LOG_FILE
    free -h >> $LOG_FILE
    echo "=== TOP 5 进程 ===" >> $LOG_FILE
    ps aux --sort=-%mem | head -6 >> $LOG_FILE
fi

# 给脚本执行权限
chmod +x /opt/scripts/system_monitor.sh
```

#### 4.1 backup.sh：数据库备份

```bash
#!/bin/bash
# backup.sh - 每日自动备份MySQL数据库

BACKUP_DIR="/backup/mysql"
DB_NAME="ecommerce"
DB_USER="datauser"
DB_PASS="SecurePass123!"
RETENTION_DAYS=30

mkdir -p $BACKUP_DIR

TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="$BACKUP_DIR/${DB_NAME}_${TIMESTAMP}.sql.gz"

# 执行备份
mysqldump -u $DB_USER -p$DB_PASS \
    --single-transaction \
    --routines \
    --triggers \
    $DB_NAME | gzip > $BACKUP_FILE

# 检查备份是否成功
if [ $? -eq 0 ]; then
    echo "[$(date)] 备份成功: $BACKUP_FILE ($(du -h $BACKUP_FILE | cut -f1))"
else
    echo "[$(date)] 备份失败!" >&2
    exit 1
fi

# 清理超过保留期的备份
find $BACKUP_DIR -name "${DB_NAME}_*.sql.gz" -mtime +$RETENTION_DAYS -delete

echo "[$(date)] 已清理 ${RETENTION_DAYS} 天前的备份"
```

#### 4.3 health_check.sh：服务健康检查

```bash
#!/bin/bash
# health_check.sh - 检查所有服务状态

LOG_FILE="/var/log/health_check.log"

check_service() {
    local service=$1
    local display=$2

    if systemctl is-active --quiet $service; then
        echo "✅ $display: 运行中"
    else
        echo "❌ $display: 未运行"
        echo "[$(date)] $display 未运行" >> $LOG_FILE
    fi
}

check_port() {
    local port=$1
    local display=$2

    if ss -tlnp | grep -q ":$port "; then
        echo "✅ $display (端口 $port): 监听中"
    else
        echo "❌ $display (端口 $port): 未监听"
        echo "[$(date)] $display 端口 $port 未监听" >> $LOG_FILE
    fi
}

echo "=== 服务健康检查 ($(date)) ==="

echo ""
echo "【系统服务】"
check_service "nginx" "Nginx"
check_service "mysql" "MySQL"
check_service "log-analyzer" "Log Analyzer"

echo ""
echo "【端口检查】"
check_port 80 "HTTP"
check_port 3306 "MySQL"
check_port 5000 "Python App"

echo ""
echo "【系统资源】"
echo "  磁盘: $(df -h / | awk 'NR==2 {print $5 " used (" $3 "/" $2 ")"}')"
echo "  内存: $(free -h | awk '/Mem:/ {print $3 "/" $2 " (" $3/$2*100 "%)"}')"
echo "  负载: $(uptime | awk -F'load average:' '{print $2}')"

echo ""
echo "=== 检查完成 ==="
```

### 6.1 backup.sh：数据库全量备份脚本

```bash
#!/bin/bash
# ============================================================
# backup.sh - 每日自动备份MySQL数据库
# 功能: 全量备份、压缩归档、过期清理、钉钉告警
# 用法: ./backup.sh [database_name]
# ============================================================
set -euo pipefail

# ─── 配置区 ─────────────────────────────────────────
DB_NAME="${1:-ecommerce}"
DB_USER="datauser"
DB_PASS="SecurePass123!"       # 生产环境应从配置文件或环境变量读取
DB_HOST="localhost"
DB_PORT="3306"

BACKUP_BASE_DIR="/backup/mysql"
BACKUP_DIR="${BACKUP_BASE_DIR}/$(date +%Y/%m)"
RETENTION_DAYS=30

LOG_DIR="/var/log/backup"
LOG_FILE="${LOG_DIR}/backup_$(date +%Y%m%d).log"

# 钉钉Webhook告警（可选,如不需要则留空）
DINGTALK_WEBHOOK=""

# ─── 初始化 ─────────────────────────────────────────
mkdir -p "$BACKUP_DIR" "$LOG_DIR"

exec 1> >(tee -a "$LOG_FILE")
exec 2> >(tee -a "$LOG_FILE" >&2)

log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $*"
}

send_alert() {
    local msg="$1"
    if [[ -n "$DINGTALK_WEBHOOK" ]]; then
        curl -s -H "Content-Type: application/json" \
            -d "{\"msgtype\":\"text\",\"text\":{\"content\":\"[MySQL备份] $msg\"}}" \
            "$DINGTALK_WEBHOOK" > /dev/null 2>&1 || true
    fi
}

# ─── 备份前检查 ─────────────────────────────────────
log "===== 开始数据库备份 ====="
log "目标数据库: $DB_NAME"

# 检查MySQL连接
if ! mysql -u "$DB_USER" -p"$DB_PASS" -h "$DB_HOST" -P "$DB_PORT" \
    -e "SELECT 1;" > /dev/null 2>&1; then
    log "错误: 无法连接到MySQL数据库"
    send_alert "备份失败：无法连接到 $DB_HOST:$DB_PORT/$DB_NAME"
    exit 1
fi

# 获取数据库大小
DB_SIZE=$(mysql -u "$DB_USER" -p"$DB_PASS" -h "$DB_HOST" -P "$DB_PORT" \
    -e "SELECT ROUND(SUM(data_length + index_length)/1024/1024,2)
        FROM information_schema.tables WHERE table_schema='$DB_NAME';" \
    -N -B 2>/dev/null || echo "N/A")
log "数据库大小: ${DB_SIZE} MB"

# ─── 执行备份 ───────────────────────────────────────
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="${BACKUP_DIR}/${DB_NAME}_${TIMESTAMP}.sql.gz"
CHECKSUM_FILE="${BACKUP_FILE}.md5"

log "开始导出数据..."

# 全量备份（包含存储过程、触发器、事件）
if mysqldump \
    -u "$DB_USER" \
    -p"$DB_PASS" \
    -h "$DB_HOST" \
    -P "$DB_PORT" \
    --single-transaction \
    --routines \
    --triggers \
    --events \
    --hex-blob \
    --default-character-set=utf8mb4 \
    --skip-lock-tables \
    --quick \
    "$DB_NAME" 2>> "$LOG_FILE" | gzip -9 > "$BACKUP_FILE"; then

    # ─── 生成校验和 ──────────────────────────────
    if command -v md5sum &> /dev/null; then
        md5sum "$BACKUP_FILE" | cut -d' ' -f1 > "$CHECKSUM_FILE"
    elif command -v md5 &> /dev/null; then
        md5 -q "$BACKUP_FILE" > "$CHECKSUM_FILE"
    fi

    BACKUP_SIZE=$(du -h "$BACKUP_FILE" | cut -f1)
    COMPRESSION_RATIO=$(python3 -c "
        raw=float('${DB_SIZE}' or 0)
        compressed=$(du -b '$BACKUP_FILE' | cut -f1)
        compressed_mb=compressed/1024/1024
        if raw > 0:
            print(f'{compressed_mb/raw*100:.1f}%')
        else:
            print('N/A')
    " 2>/dev/null || echo "N/A")

    log "备份成功: $BACKUP_FILE"
    log "文件大小: $BACKUP_SIZE"
    log "压缩率: $COMPRESSION_RATIO"
    log "MD5校验: $(cat "$CHECKSUM_FILE" 2>/dev/null || echo 'N/A')"

    send_alert "✅ ${DB_NAME} 备份成功 | 大小:${BACKUP_SIZE} | 压缩率:${COMPRESSION_RATIO}"
else
    log "错误: 备份失败!"
    send_alert "❌ ${DB_NAME} 备份失败,请立即检查!"
    rm -f "$BACKUP_FILE"
    exit 1
fi

# ─── 创建副本到异地备份目录（可选） ────────────────
REMOTE_BACKUP="/mnt/remote_backup/mysql"
```

```bash
#!/bin/bash
# ============================================================
# health_check.sh - 全方位服务健康检查脚本
# 功能: 检查服务状态、端口、磁盘、内存、进程、数据库连接
# 用法: ./health_check.sh [--alert] [--json]
# ============================================================
set -euo pipefail

# ─── 配置区 ─────────────────────────────────────────
LOG_DIR="/var/log/health_check"
LOG_FILE="${LOG_DIR}/health_$(date +%Y%m%d).log"
ALERT_LOG="${LOG_DIR}/alerts.log"

# 阈值配置
CPU_WARN=80
CPU_CRITICAL=95
MEM_WARN=85
MEM_CRITICAL=95
DISK_WARN=80
DISK_CRITICAL=90
LOAD_WARN=4.0
LOAD_CRITICAL=8.0

# 钉钉/企业微信告警URL
ALERT_WEBHOOK=""

mkdir -p "$LOG_DIR"

# ─── 工具函数 ───────────────────────────────────────
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

log() { echo "[$(date '+%H:%M:%S')] $*" | tee -a "$LOG_FILE"; }
ok() { log "${GREEN}✅ $*${NC}"; }
warn() { log "${YELLOW}⚠️  $*${NC}"; }
err() { log "${RED}❌ $*${NC}"; }

send_alert() {
    local level="$1" msg="$2"
    local emoji=""
    case "$level" in
        CRITICAL) emoji="🔴" ;;
        WARNING)  emoji="🟡" ;;
        INFO)     emoji="🟢" ;;
    esac
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] ${emoji} [${level}] ${msg}" >> "$ALERT_LOG"
    if [[ -n "$ALERT_WEBHOOK" ]]; then
        curl -s -H "Content-Type: application/json" \
            -d "{\"msgtype\":\"text\",\"text\":{\"content\":\"${emoji} [健康检查-${level}] ${msg}\"}}" \
            "$ALERT_WEBHOOK" > /dev/null 2>&1 || true
    fi
}

# ─── 检查函数 ───────────────────────────────────────

check_service() {
    local service="$1"
    local label="${2:-$service}"
    local status

    status=$(systemctl is-active "$service" 2>/dev/null || echo "not-found")

    case "$status" in
        active)
            ok "$label 运行正常"
            return 0
            ;;
        inactive)
            warn "$label 未运行 (已停止)"
            send_alert "WARNING" "$label 服务已停止"
            return 1
            ;;
        failed)
            err "$label 运行失败"
            send_alert "CRITICAL" "$label 服务运行失败"
            return 2
            ;;
        not-found)
            warn "$label 未安装"
            return 3
            ;;
        *)
            warn "$label 状态未知: $status"
            return 4
            ;;
    esac
}

check_port() {
    local port="$1"
    local label="${2:-端口$port}"

    if netstat -tlnp 2>/dev/null | grep -q ":${port} "; then
        ok "$label (端口:$port) 监听中"
        return 0
    elif ss -tlnp 2>/dev/null | grep -q ":${port} "; then
        ok "$label (端口:$port) 监听中"
        return 0
    else
        err "$label (端口:$port) 未监听"
        send_alert "WARNING" "端口 $port ($label) 未监听"
        return 1
    fi
}

check_url() {
    local url="$1"
    local label="${2:-$url}"
    local expected="${3:-200}"
    local http_code
    local response_time

    response_time=$(curl -s -o /dev/null -w "%{time_total}" \
        --connect-timeout 5 --max-time 10 "$url" 2>/dev/null || echo "timeout")
    http_code=$(curl -s -o /dev/null -w "%{http_code}" \
        --connect-timeout 5 --max-time 10 "$url" 2>/dev/null || echo "000")

    if [[ "$http_code" == "$expected" ]]; then
        ok "$label → HTTP $http_code (${response_time}s)"
        return 0
    else
        err "$label → HTTP $http_code (预期 $expected)"
        send_alert "WARNING" "URL $url 返回 $http_code (预期 $expected)"
        return 1
    fi
}

check_mysql_connection() {
    local host="${1:-localhost}"
    local user="${2:-datauser}"
    local pass="${3:-SecurePass123!}"
    local db="${4:-ecommerce}"

    if mysql -u "$user" -p"$pass" -h "$host" "$db" \
        -e "SELECT 1;" > /dev/null 2>&1; then
        # 检查连接数
        local conn_count=$(mysql -u "$user" -p"$pass" -h "$host" \
            -e "SHOW STATUS LIKE 'Threads_connected';" -N -B 2>/dev/null | awk '{print $2}')
        ok "MySQL连接正常 (当前连接: ${conn_count:-N/A})"
        return 0
    else
        err "MySQL连接失败"
        send_alert "CRITICAL" "MySQL ($host/$db) 连接失败"
        return 1
    fi
}

# ─── 主检查流程 ─────────────────────────────────────
ALERT_FLAG=false
if [[ "${1:-}" == "--alert" ]]; then
    ALERT_FLAG=true
fi
OUTPUT_JSON="${2:-}"

echo ""
echo "╔══════════════════════════════════════════════════════╗"
echo "║           系统健康检查报告                            ║"
echo "║           $(date '+%Y-%m-%d %H:%M:%S')                        ║"
echo "╚══════════════════════════════════════════════════════╝"
echo ""

EXIT_CODE=0

        EXIT_CODE=1
    fi

if check_mysql_connection; then

else
    EXIT_CODE=1
fi

echo ""

# ─── 5. 系统资源 ──────────────────────────────────
echo "━━━ 【系统资源】 ━━━"

# CPU
CPU_USAGE=$(top -bn1 | grep "Cpu(s)" | awk '{print $2}' | cut -d'%' -f1 2>/dev/null || \
    grep 'cpu ' /proc/stat | awk '{usage=($2+$4)*100/($2+$4+$5)} END {print usage}')
CPU_USAGE=${CPU_USAGE:-0}

if (( CPU_INT >= CPU_CRITICAL )); then

MEM_USAGE=$(awk "BEGIN {printf \"%.1f\", ($MEM_USED/$MEM_TOTAL)*100}")

    err "磁盘使用率: ${DISK_USAGE}% (严重!)"

echo "━━━ 【内存占用TOP5】 ━━━"
ps aux --sort=-%mem | head -6 | tail -5 | awk '{printf "  PID:%-7s MEM:%-6s CPU:%-6s CMD: %s\n", $2, $4"%", $3"%", $11}'

if [[ $EXIT_CODE -eq 0 ]]; then
    echo "║  ✅ 所有检查通过!  ($PASSED_CHECKS/$TOTAL_CHECKS, ${PASS_RATE}%)"
else
    echo "║  ⚠️  存在问题! 通过: $PASSED_CHECKS/$TOTAL_CHECKS (${PASS_RATE}%)"
fi
echo "║  检查时间: $(date '+%Y-%m-%d %H:%M:%S')"
echo "║  主机名: $(hostname)"
echo "║  运行时间: $(uptime -p)"
echo "╚══════════════════════════════════════════════════════╝"

exit $EXIT_CODE
```

### 6.4 cleanup.sh：日志清理与磁盘空间回收脚本

```bash
#!/bin/bash
# ============================================================
# cleanup.sh - 智能日志清理与磁盘空间回收脚本
# 功能: 清理过期日志、临时文件、Docker残留、APT缓存、旧内核
# 用法: ./cleanup.sh [--dry-run] [--aggressive]
# ============================================================
set -euo pipefail

# ─── 配置区 ─────────────────────────────────────────
LOG_RETENTION_DAYS=30
TEMP_RETENTION_DAYS=7
TRUNCATE_LARGE_LOGS_MB=500      # 超过此大小的日志文件截断
MIN_FREE_SPACE_PCT=10           # 磁盘剩余空间低于此百分比时触发强力清理

DRY_RUN=false
AGGRESSIVE=false

for arg in "$@"; do
    case "$arg" in
        --dry-run) DRY_RUN=true ;;
        --aggressive) AGGRESSIVE=true ;;
    esac
done

CLEANUP_LOG="/var/log/cleanup/cleanup_$(date +%Y%m%d).log"
mkdir -p "$(dirname "$CLEANUP_LOG")"

exec 1> >(tee -a "$CLEANUP_LOG")
exec 2> >(tee -a "$CLEANUP_LOG" >&2)

# ─── 工具函数 ───────────────────────────────────────
TOTAL_FREED=0

log() { echo "[$(date '+%H:%M:%S')] $*"; }

safe_delete() {
    local target="$1"
    local desc="$2"

    if [[ "$DRY_RUN" == "true" ]]; then
        local size=$(du -sh "$target" 2>/dev/null | cut -f1 || echo "0")
        log "[DRY-RUN] 将删除: $target ($desc, $size)"
        return
    fi

    if [[ -e "$target" ]]; then
        local size_before=$(df -k / | awk 'NR==2 {print $3}')
        rm -rf "$target" 2>/dev/null || sudo rm -rf "$target" 2>/dev/null || true
        local size_after=$(df -k / | awk 'NR==2 {print $3}')
        local freed=$((size_before - size_after))
        TOTAL_FREED=$((TOTAL_FREED + freed))
        if (( freed > 0 )); then
            log "  已删除: $desc (释放 ${freed}KB)"
        fi
    fi
}

# ─── 检查磁盘空间 ──────────────────────────────────
DISK_USAGE=$(df -h / | awk 'NR==2 {print $5}' | sed 's/%//')
DISK_FREE=$(df -h / | awk 'NR==2 {print $4}')
DISK_TOTAL=$(df -h / | awk 'NR==2 {print $2}')

log "===== 开始清理 $(date '+%Y-%m-%d %H:%M:%S') ====="
log "磁盘状态: 已用 ${DISK_USAGE}% | 可用 ${DISK_FREE} | 总计 ${DISK_TOTAL}"

if (( DISK_USAGE > (100 - MIN_FREE_SPACE_PCT) )); then
    log "磁盘空间不足,启用强力清理模式!"
    AGGRESSIVE=true
fi

log "清理模式: DRY_RUN=$DRY_RUN | AGGRESSIVE=$AGGRESSIVE"

# ─── 阶段1: Nginx日志清理 ───────────────────────────
log ""
log "━━━ [1/9] Nginx日志清理 ━━━"

# 清理Nginx压缩日志
log "清理 ${LOG_RETENTION_DAYS} 天前的Nginx压缩日志..."
find /var/log/nginx -name "*.gz" -mtime +$LOG_RETENTION_DAYS -type f 2>/dev/null | while read f; do
    safe_delete "$f" "Nginx压缩日志: $(basename "$f")"
done

# 截断大型access日志
if [[ "$AGGRESSIVE" == "true" ]]; then
    for logfile in /var/log/nginx/*.log; do
        if [[ -f "$logfile" ]]; then
            LOG_SIZE=$(stat -c%s "$logfile" 2>/dev/null || echo 0)
            LOG_SIZE_MB=$((LOG_SIZE / 1024 / 1024))
            if (( LOG_SIZE_MB > TRUNCATE_LARGE_LOGS_MB )); then
                if [[ "$DRY_RUN" == "true" ]]; then
                    log "[DRY-RUN] 将截断: $logfile (${LOG_SIZE_MB}MB)"
                else
                    sudo truncate -s 0 "$logfile" 2>/dev/null || true
                    log "  已截断: $(basename "$logfile") (原 ${LOG_SIZE_MB}MB)"
                fi
            fi
        fi
    done
fi

# ─── 阶段2: 应用日志清理 ───────────────────────────
log ""
log "━━━ [2/9] 应用日志清理 ━━━"
find /var/log/log-analyzer -name "*.log.*" -mtime +$LOG_RETENTION_DAYS -delete 2>/dev/null
log "已清理应用轮转日志 (保留 ${LOG_RETENTION_DAYS} 天)"

# ─── 阶段3: 系统监控日志清理 ────────────────────────
log ""
log "━━━ [3/9] 监控日志清理 ━━━"
find /var/log/system_monitor -name "*.log" -mtime +$LOG_RETENTION_DAYS -delete 2>/dev/null
log "已清理监控日志 (保留 ${LOG_RETENTION_DAYS} 天)"

# ─── 阶段4: systemd journal清理 ────────────────────
log ""
log "━━━ [4/9] systemd journal清理 ━━━"

if [[ "$DRY_RUN" == "true" ]]; then
    log "[DRY-RUN] 将清理 journal 日志"
else
    if [[ "$AGGRESSIVE" == "true" ]]; then
        sudo journalctl --vacuum-time=7d --vacuum-size=500M 2>/dev/null || true
        log "强力清理: journal保留7天/500MB"
    else
        sudo journalctl --vacuum-time=30d 2>/dev/null || true
        log "常规清理: journal保留30天"
    fi
fi

# ─── 阶段5: 临时文件清理 ───────────────────────────
log ""
log "━━━ [5/9] 临时文件清理 ━━━"

# /tmp目录
find /tmp -type f -atime +$TEMP_RETENTION_DAYS ! -name ".*" 2>/dev/null | while read f; do
    safe_delete "$f" "临时文件: $(basename "$f")"
done

# /var/tmp 目录
find /var/tmp -type f -atime +$((TEMP_RETENTION_DAYS * 2)) 2>/dev/null | while read f; do
    safe_delete "$f" "系统临时文件: $(basename "$f")"
done

# Python字节码缓存
find /opt -name "__pycache__" -type d 2>/dev/null | while read d; do
    if [[ "$DRY_RUN" == "true" ]]; then
        log "[DRY-RUN] 将清理: $d"
    else
        sudo rm -rf "$d" 2>/dev/null || rm -rf "$d" 2>/dev/null || true
    fi
done
log "已清理 __pycache__ 目录"

find /opt -name "*.pyc" -type f -delete 2>/dev/null || true

# ─── 阶段6: APT缓存清理 ────────────────────────────
log ""
log "━━━ [6/9] APT缓存清理 ━━━"
if [[ "$DRY_RUN" == "true" ]]; then
    log "[DRY-RUN] 将执行 apt-get clean/autoremove"
else
    sudo apt-get clean -y 2>/dev/null && log "  apt-get clean 完成" || true
    sudo apt-get autoclean -y 2>/dev/null && log "  apt-get autoclean 完成" || true
    if [[ "$AGGRESSIVE" == "true" ]]; then
        sudo apt-get autoremove --purge -y 2>/dev/null && log "  apt-get autoremove 完成" || true
    fi
fi

# ─── 阶段7: Docker清理 ─────────────────────────────
log ""
log "━━━ [7/9] Docker清理 ━━━"
if command -v docker &> /dev/null; then
    if [[ "$DRY_RUN" == "true" ]]; then
        log "[DRY-RUN] 将执行 docker system prune"
    else
        docker system prune -f 2>/dev/null && log "  Docker无用资源已清理" || true
        if [[ "$AGGRESSIVE" == "true" ]]; then
            docker system prune -a -f --volumes 2>/dev/null && \
                log "  Docker全量清理完成(含未使用镜像和卷)" || true
        fi
    fi
else
    log "Docker未安装,跳过"
fi

# ─── 阶段8: 旧内核清理(仅强力模式) ─────────────────
log ""
log "━━━ [8/9] 内核清理 ━━━"
if [[ "$AGGRESSIVE" == "true" ]]; then
    if [[ "$DRY_RUN" == "true" ]]; then
        log "[DRY-RUN] 将清理旧内核"
    else
        CURRENT_KERNEL=$(uname -r)
        INSTALLED_KERNELS=$(dpkg -l 'linux-image-*' 2>/dev/null | \
            grep '^ii' | awk '{print $2}' | grep -v "$CURRENT_KERNEL" || true)
        if [[ -n "$INSTALLED_KERNELS" ]]; then
            log "当前内核: $CURRENT_KERNEL"
            for kernel in $INSTALLED_KERNELS; do
                sudo apt-get purge -y "$kernel" 2>/dev/null && \
                    log "  已移除旧内核: $kernel" || true
            done
        else
            log "无旧内核需要清理"
        fi
    fi
else
    log "非常规模式,跳过旧内核清理"
fi

# ─── 阶段9: 备份清理(仅强力模式) ───────────────────
log ""
log "━━━ [9/9] 备份清理 ━━━"
if [[ "$AGGRESSIVE" == "true" ]]; then
    BACKUP_RETENTION=7
    find /backup/mysql -name "*.sql.gz" -mtime +$BACKUP_RETENTION -type f 2>/dev/null | while read f; do
        safe_delete "$f" "过期备份: $(basename "$f")"
    done
    log "强力清除超过${BACKUP_RETENTION}天的数据库备份"
fi

# ─── 磁盘空间报告 ──────────────────────────────────
DISK_AFTER=$(df -h / | awk 'NR==2 {print $5" used ("$3"/"$2") - avail:"$4}')
TOTAL_FREED_MB=$((TOTAL_FREED / 1024))

echo ""
echo "╔══════════════════════════════════════════════════════╗"
echo "║              清理任务完成                             ║"
echo "╠══════════════════════════════════════════════════════╣"
if (( TOTAL_FREED_MB > 0 )); then
    echo "║  释放空间: ${TOTAL_FREED_MB} MB"
else
    echo "║  释放空间: 无需清理或DRY-RUN模式"
fi
echo "║  磁盘状态: $DISK_AFTER"
echo "║  清理时间: $(date '+%Y-%m-%d %H:%M:%S')"
echo "╚══════════════════════════════════════════════════════╝"

exit 0
```


**service_guard.sh 服务守护脚本**：

```bash
#!/bin/bash
# 检查服务状态并自动重启

SERVICES=("nginx" "mysql" "log-analyzer")

for svc in "${SERVICES[@]}"; do
    if systemctl is-active --quiet "$svc"; then
        :
    else
        echo "[$(date)] $svc 已停止,尝试重启..." >> /var/log/service_guard.log
        systemctl restart "$svc"
        sleep 3
        if systemctl is-active --quiet "$svc"; then
            echo "[$(date)] $svc 重启成功" >> /var/log/service_guard.log
        else
            echo "[$(date)] ❌ $svc 重启失败!" >> /var/log/service_guard.log
            # 发送告警
            curl -s "https://your-alert-api.com/alert?svc=$svc&status=failed" > /dev/null 2>&1
        fi
    fi
done
```

## 十、常见问题排查

### 问题1：Nginx启动失败报"port already in use"

**现象**：
```
nginx: [emerg] bind() to 0.0.0.0:80 failed (98: Address already in use)
```

**排查步骤**：
```bash
# 1. 查看哪个进程占用了80端口
sudo lsof -i :80
# 或
sudo netstat -tlnp | grep :80

# 2. 如果是其他Web服务器(Apache等),停掉它
sudo systemctl stop apache2

# 3. 确保Nginx配置中listen 80只有一个
sudo grep -r "listen 80" /etc/nginx/

# 4. 杀掉顽固进程后重启
sudo fuser -k 80/tcp
sudo systemctl restart nginx
```

### 问题2：MySQL连接报"Too many connections"

**现象**：
```
ERROR 1040 (HY000): Too many connections
```

**排查与解决**：
```bash
# 1. 查看当前最大连接数
mysql -u root -p -e "SHOW VARIABLES LIKE 'max_connections';"

# 2. 查看当前连接数
mysql -u root -p -e "SHOW STATUS LIKE 'Threads_connected';"

# 3. 查看各用户连接数
mysql -u root -p -e "SELECT USER, HOST, COUNT(*) FROM information_schema.PROCESSLIST GROUP BY USER, HOST;"

# 4. 临时增大最大连接数(重启失效)
mysql -u root -p -e "SET GLOBAL max_connections = 500;"

# 5. 永久修改: 编辑 /etc/mysql/mysql.conf.d/mysqld.cnf
# [mysqld]
# max_connections = 500
# wait_timeout = 300

# 6. 杀掉空闲连接
mysql -u root -p -e "SELECT CONCAT('KILL ',id,';') FROM information_schema.PROCESSLIST WHERE Command='Sleep' AND Time>300;" | grep KILL | mysql -u root -p
```

### 问题3：Python Flask应用返回503

**现象**：
```
curl http://localhost/api/health → 503 Service Unavailable
```

**排查步骤**：
```bash
# 1. 检查Python应用是否在运行
ps aux | grep gunicorn
ps aux | grep "python app.py"

# 2. 检查端口是否在监听
ss -tlnp | grep 5000

# 3. 查看应用日志
sudo journalctl -u log-analyzer --since "10 minutes ago"
tail -100 /var/log/log-analyzer/gunicorn-error.log

# 4. 测试直接访问Python端口(绕过Nginx)
curl http://127.0.0.1:5000/api/health

# 5. 手动启动调试
cd /opt/log-analyzer/current
source venv/bin/activate
python app.py  # 查看启动错误

# 6. 常见原因
# - Python依赖未安装: pip install -r requirements.txt
# - 端口被占用: fuser -k 5000/tcp
# - 数据库连接失败: 检查MySQL是否运行
# - 文件权限: chown -R www-data:www-data /opt/log-analyzer
```

```bash
sudo /opt/scripts/cleanup.sh --aggressive
```

### 问题5：cron任务不执行

**排查步骤**：
```bash

# 2. 查看cron日志

# 3. 检查脚本是否有执行权限
ls -la /opt/scripts/

# 4. 检查脚本中的环境变量
# cron执行时PATH与交互式shell不同,需要绝对路径或设置PATH

# 5. 常见原因
# - 脚本没有执行权限: chmod +x script.sh

# 6. 测试cron命令
```

### 问题6：SSH无法连接

**排查步骤**：
```bash

systemctl status sshd

# 2. 检查端口监听
ss -tlnp | grep :22

# 3. 检查防火墙
sudo ufw status

# 4. 检查SSH配置
sudo cat /etc/ssh/sshd_config | grep -E "Port|PermitRootLogin|PasswordAuthentication|PubkeyAuthentication"

# 5. 查看SSH日志
sudo tail -50 /var/log/auth.log

# 6. 常见原因:
# - SSH服务未启动: systemctl restart sshd
# - 防火墙阻挡: ufw allow 22
# - 密钥权限: chmod 600 ~/.ssh/id_rsa
# - known_hosts冲突: ssh-keygen -R 目标IP
```

### 问题8：gunicorn workers自动退出(内存溢出)

### 问题9：Nginx upstream超时

**现象**：
```
upstream timed out (110: Connection timed out)
```

**解决**：
```nginx
# 在location /api/块中增加:
proxy_connect_timeout 30s;
proxy_send_timeout 120s;
proxy_read_timeout 120s;

# 增加upstream keepalive
upstream backend {
    server 127.0.0.1:5000;
    keepalive 32;
}
```
