```python
class DataRecord:
    """数据处理记录类——最简单的类示例"""

    # 类属性（所有实例共享）
    record_count = 0

    def __init__(self, name, value, timestamp=None):
        """初始化方法——创建对象时自动调用"""
        from datetime import datetime
        self.name = name          # 实例属性
        self.value = value
        self.timestamp = timestamp or datetime.now()
        DataRecord.record_count += 1

    def display(self):
        """实例方法——展示记录信息"""
        return f"[{self.timestamp}] {self.name}: {self.value}"

    def update_value(self, new_value):
        """更新值"""
        old = self.value
        self.value = new_value
        return f"值已更新: {old} → {new_value}"

    @classmethod
    def get_count(cls):
        """类方法——操作类属性"""
        return cls.record_count

    @staticmethod
    def validate_name(name):
        """静态方法——与类和实例无关的工具函数"""
        return isinstance(name, str) and len(name) > 0

    def __str__(self):
        """魔术方法——print()时调用"""
        return self.display()

    def __repr__(self):
        """魔术方法——在交互式环境中显示"""
        return f"DataRecord(name='{self.name}', value={self.value})"

class MetricTracker:
    """指标追踪器——展示属性访问控制"""

    def __init__(self, metric_name):
        self._name = metric_name        # 约定：_开头表示"受保护的"
        self._values = []               # 内部数据
        self.__min_threshold = 0        # __开头：名称改编（name mangling）

    @property
    def name(self):
        """只读属性——像属性一样访问，但有方法逻辑"""
        return self._name.upper()

    @property
    def count(self):
        return len(self._values)

    @property
    def average(self):
        if not self._values:
            return 0
        return sum(self._values) / len(self._values)

    @property
    def max_value(self):
        return max(self._values) if self._values else None

    @property
    def min_value(self):
        return min(self._values) if self._values else None

    def add(self, value):
        """添加一个数据点"""
        if value < self.__min_threshold:
            print(f"警告: 值 {value} 低于阈值 {self.__min_threshold}")
        self._values.append(value)

    def add_batch(self, values):
        """批量添加"""
        self._values.extend(values)

    def get_stats(self):
        """获取统计摘要"""
        return {
            "metric": self._name,
            "count": self.count,
            "avg": self.average,
            "max": self.max_value,
            "min": self.min_value,
        }

# 使用示例
tracker = MetricTracker("response_time")
tracker.add_batch([120, 150, 200, 180, 250, 90, 300])
print(f"指标: {tracker.name}")
print(f"数据点数: {tracker.count}")
print(f"平均: {tracker.average:.1f}ms")
print(f"最大: {tracker.max_value}ms")
```

```python
import logging

from datetime import datetime

# ========== 抽象基类 ==========

class DataProcessor(ABC):
    """数据处理基类——所有ETL任务的模板"""

    def __init__(self, name):
        self.name = name
        self.logger = self._setup_logger()

    def _setup_logger(self):
        """设置日志记录器"""
        logger = logging.getLogger(self.name)
        logger.setLevel(logging.INFO)

        # 控制台输出
        console = logging.StreamHandler()
        console.setFormatter(
            logging.Formatter(
                '%(asctime)s [%(name)s] %(levelname)s: %(message)s',
                datefmt='%Y-%m-%d %H:%M:%S'
            )
        )

        # 文件输出
        file_handler = logging.FileHandler(f"{self.name}.log", encoding="utf-8")
        file_handler.setFormatter(
            logging.Formatter(
                '%(asctime)s [%(levelname)s] %(message)s'
            )
        )

        logger.addHandler(console)
        logger.addHandler(file_handler)
        return logger

    @abstractmethod
    def extract(self, source):
        """从数据源提取数据（子类必须实现）"""
        pass

    @abstractmethod
    def transform(self, data):
        """转换数据（子类必须实现）"""
        pass

    @abstractmethod
    def load(self, data, target):
        """加载数据到目标（子类必须实现）"""
        pass

    def run(self, source, target):
        """执行完整ETL流程（模板方法模式）"""
        self.logger.info(f"开始ETL任务: {self.name}")

        try:
            raw = self.extract(source)
            self.logger.info(f"提取完成: {len(raw)} 行")

            cleaned = self.transform(raw)
            self.logger.info(f"转换完成: {len(cleaned)} 行")

            self.load(cleaned, target)
            self.logger.info(f"加载完成: {target}")

            return True
        except Exception as e:
            self.logger.error(f"ETL失败: {e}", exc_info=True)
            return False


# ========== 具体实现类 ==========

class CSVToJSONProcessor(DataProcessor):
    """CSV转JSON的ETL处理器"""

    def extract(self, source):
        """从CSV文件提取数据"""
        import csv
        self.logger.info(f"正在读取: {source}")

        records = []
        with open(source, "r", encoding="utf-8") as f:
            reader = csv.DictReader(f)
            for row in reader:
                if any(row.values()):  # 跳过空行
                    records.append(row)

        return records

    def transform(self, data):
        """数据清洗和转换"""
        # 去除空行和重复行
        seen = set()
        cleaned = []
        for row in data:
            key = row.get("id", "").strip()
            if key and key not in seen:
                seen.add(key)
                cleaned.append({k: v.strip() for k, v in row.items()})

        self.logger.info(f"清洗前后: {len(data)} → {len(cleaned)} 行")
        return cleaned

    def load(self, data, target):
        """写入JSON文件"""
        import json
        with open(target, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)

        self.logger.info(f"已写入 {len(data)} 条记录")


# ========== 继承的另一个示例 ==========

class CSVToJSONLProcessor(CSVToJSONProcessor):
    """CSV转JSONL（每行一个JSON对象）的处理器"""

    def load(self, data, target):
        """重写load方法——改为JSON Lines格式"""
        import json
        with open(target, "w", encoding="utf-8") as f:
            for record in data:
                f.write(json.dumps(record, ensure_ascii=False) + "\n")

        self.logger.info(f"已写入 {len(data)} 条记录（JSONL格式）")


# ========== 多态的体现 ==========

class DataPipeline:
    """数据管道——可以运行任意DataProcessor子类"""

    def __init__(self, *processors):
        self.processors = list(processors)

    def add_processor(self, processor):
        self.processors.append(processor)

    def run_all(self, source_dir, target_dir):
        """运行所有处理器"""
        results = {}
        for p in self.processors:
            source = f"{source_dir}/input.csv"
            target = f"{target_dir}/{p.name}_output"
            success = p.run(source, target)
            results[p.name] = success
        return results

# 使用示例
pipeline = DataPipeline(
    CSVToJSONProcessor("csv_to_json"),
    CSVToJSONLProcessor("csv_to_jsonl"),
)
```

### 2.4 实战：构建数据处理ETL基类（30分钟）

```python
"""
完整的ETL处理框架实战代码
"""
import csv
import json
import logging
import time
from pathlib import Path
from abc import ABC, abstractmethod
from datetime import datetime, timedelta


# ========== 日志配置 ==========
def setup_logging(log_dir="logs"):
    """统一日志配置"""
    log_dir = Path(log_dir)
    log_dir.mkdir(exist_ok=True)

    log_file = log_dir / f"etl_{datetime.now().strftime('%Y%m%d')}.log"

    logging.basicConfig(
        level=logging.INFO,
        format='%(asctime)s [%(levelname)-8s] %(name)s: %(message)s',
        datefmt='%Y-%m-%d %H:%M:%S',
        handlers=[
            logging.StreamHandler(),
            logging.FileHandler(log_file, encoding='utf-8'),
        ]
    )


# ========== 抽象基类（增强版）==========
class ETLProcessor(ABC):
    """
    增强版ETL处理器基类
    新增功能：耗时记录、进度跟踪、统计报告
    """

    def __init__(self, name):
        self.name = name
        self.logger = logging.getLogger(name)
        self.stats = {
            "name": name,
            "start_time": None,
            "end_time": None,
            "extract_count": 0,
            "transform_count": 0,
            "load_count": 0,
            "errors": [],
        }

    @abstractmethod
    def extract(self, source):
        pass

    @abstractmethod
    def transform(self, data):
        pass

    @abstractmethod
    def load(self, data, target):
        pass

    def run(self, source, target):
        """执行完整ETL流程"""
        self.stats["start_time"] = datetime.now()
        self.logger.info(f"{'='*40}")
        self.logger.info(f"开始ETL任务: {self.name}")
        self.logger.info(f"数据源: {source}")
        self.logger.info(f"目标: {target}")

        try:
            start = time.perf_counter()
            raw = self.extract(source)
            extract_time = time.perf_counter() - start
            self.stats["extract_count"] = len(raw) if raw else 0
            self.logger.info(f"Extract完成: {self.stats['extract_count']} 行 "
                           f"({extract_time:.2f}s)")

            start = time.perf_counter()
            cleaned = self.transform(raw)
            transform_time = time.perf_counter() - start
            self.stats["transform_count"] = len(cleaned) if cleaned else 0
            self.logger.info(f"Transform完成: {self.stats['transform_count']} 行 "
                           f"({transform_time:.2f}s)")

            start = time.perf_counter()
            self.load(cleaned, target)
            load_time = time.perf_counter() - start
            self.stats["load_count"] = self.stats["transform_count"]
            self.logger.info(f"Load完成 ({load_time:.2f}s)")

            self.stats["end_time"] = datetime.now()
            duration = (self.stats["end_time"] -
                       self.stats["start_time"]).total_seconds()
            self.logger.info(f"ETL任务完成，总耗时: {duration:.2f}s")
            self._print_report()
            return True

        except Exception as e:
            self.stats["errors"].append(str(e))
            self.logger.error(f"ETL失败: {e}", exc_info=True)
            return False

    def _print_report(self):
        """打印统计报告"""
        self.logger.info(f"{'─'*40}")
        self.logger.info(f"任务报告: {self.name}")
        self.logger.info(f"  提取行数: {self.stats['extract_count']}")
        self.logger.info(f"  转换后行数: {self.stats['transform_count']}")
        self.logger.info(f"  数据过滤率: "
                       f"{(1 - self.stats['transform_count']/max(self.stats['extract_count'], 1))*100:.1f}%")
        self.logger.info(f"  错误数: {len(self.stats['errors'])}")
        duration = (self.stats["end_time"] - self.stats["start_time"]).total_seconds()
        self.logger.info(f"  总耗时: {duration:.2f}s")
        self.logger.info(f"{'─'*40}")


# ========== 具体实现：订单数据ETL处理器 ==========
class OrderETLProcessor(ETLProcessor):
    """订单数据ETL处理器"""

    VALID_STATUSES = {"completed", "pending", "cancelled", "refunded"}

    def extract(self, source):
        """从CSV提取订单数据"""
        records = []
        with open(source, "r", encoding="utf-8") as f:
            reader = csv.DictReader(f)
            for row in reader:
                records.append(row)
        return records

    def transform(self, data):
        """清洗和转换订单数据"""
        cleaned = []
        seen_ids = set()

        for row in data:
            # 跳过空行
            if not any(row.values()):
                continue

            # 标准化字段
            row = {k: v.strip() for k, v in row.items()}

            # 去重
            order_id = row.get("order_id", "")
            if not order_id or order_id in seen_ids:
                continue
            seen_ids.add(order_id)

            # 数据验证
            try:
                amount = float(row.get("amount", 0))
                if amount <= 0:
                    self.logger.warning(f"无效金额: {amount}, 订单={order_id}")
                    continue

                status = row.get("status", "unknown")
                if status not in self.VALID_STATUSES:
                    self.logger.warning(f"无效状态: {status}, 订单={order_id}")

                # 添加转换后的字段
                row["amount_verified"] = amount
                row["is_high_value"] = amount > 1000

            except (ValueError, TypeError) as e:
                self.logger.warning(f"数据验证失败: {e}, 订单={order_id}")
                continue

            cleaned.append(row)

        return cleaned

    def load(self, data, target):
        """输出到CSV和JSON"""
        target_path = Path(target)
        target_path.parent.mkdir(parents=True, exist_ok=True)

        # 输出CSV
        csv_file = str(target_path) + ".csv"
        if data:
            with open(csv_file, "w", encoding="utf-8", newline="") as f:
                writer = csv.DictWriter(f, fieldnames=data[0].keys())
                writer.writeheader()
                writer.writerows(data)

        # 输出JSON
        json_file = str(target_path) + ".json"
        with open(json_file, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
```

### 练习：继承ETLProcessor实现CSV清洗器

```python
"""
练习要求：
1. 继承ETLProcessor基类
2. 实现一个日志文件清洗器
3. 添加数据过滤功能（过滤指定状态码的日志）
"""
import re
from collections import defaultdict

class LogCleaner(ETLProcessor):
    """日志文件清洗处理器"""

    LOG_PATTERN = re.compile(
        r'(?P<ip>\S+) \S+ \S+ '
        r'\[(?P<time>[^\]]+)\] '
        r'"(?P<method>\S+) (?P<url>\S+) \S+" '
        r'(?P<status>\d+) (?P<size>\d+)'
    )

    def __init__(self, name, exclude_statuses=None):
        super().__init__(name)
        self.exclude_statuses = exclude_statuses or []
        self.error_count = 0

    def extract(self, source):
        """读取日志文件"""
        lines = []
        with open(source, "r", encoding="utf-8") as f:
            for line in f:
                lines.append(line.strip())
        return lines

    def transform(self, data):
        """解析并过滤日志"""
        cleaned = []
        self.error_count = 0

        for line in data:
            if not line:
                continue

            match = self.LOG_PATTERN.match(line)
            if not match:
                self.error_count += 1
                continue

            record = match.groupdict()

            # 过滤排除的状态码
            if record["status"] in self.exclude_statuses:
                continue

            # 添加衍生字段
            record["size_kb"] = round(int(record["size"]) / 1024, 2)
            record["is_error"] = record["status"] != "200"

            cleaned.append(record)

        self.logger.info(f"解析失败行数: {self.error_count}")
        return cleaned

    def load(self, data, target):
        """输出清洗后的日志"""
        import json

        # 汇总统计
        summary = {
            "total": len(data),
            "unique_ips": len(set(r["ip"] for r in data)),
            "status_dist": defaultdict(int),
            "method_dist": defaultdict(int),
            "total_size_kb": sum(r["size_kb"] for r in data),
        }

        for r in data:
            summary["status_dist"][r["status"]] += 1
            summary["method_dist"][r["method"]] += 1

        summary["status_dist"] = dict(summary["status_dist"])
```

```python
"""
作业：增强版CSV到JSONL的ETL处理器
"""
import csv
import json
from pathlib import Path

# pip install tqdm
try:
    from tqdm import tqdm
    HAS_TQDM = True
except ImportError:
    HAS_TQDM = False
    print("建议安装tqdm: pip install tqdm")


class EnhancedCSVToJSONL(ETLProcessor):
    """增强版CSV转JSONL处理器"""

    def extract(self, source):
        records = []
        with open(source, "r", encoding="utf-8") as f:
            reader = csv.DictReader(f)
            lines = f.readlines()  # 获取行数用于进度条

            # TODO: 使用tqdm显示进度
            iterator = tqdm(reader, total=len(lines), desc="提取数据") \
                       if HAS_TQDM else reader

            for row in iterator:
                if any(row.values()):
                    records.append({k: v.strip() for k, v in row.items()})

        return records

    def transform(self, data):
        # TODO: 实现数据去重和验证
        seen = set()
        cleaned = []

        iterator = tqdm(data, desc="清洗数据") if HAS_TQDM else data

        for row in iterator:
            row_id = row.get("id", "")
            if row_id and row_id not in seen:
                seen.add(row_id)
                cleaned.append(row)

        return cleaned

    def load(self, data, target):
```

编写一个 `ConfigManager` 类，支持从多种格式（JSON、YAML、INI）读取配置：

```python
"""
配置管理器——支持多种配置格式
"""
import json
from pathlib import Path

class ConfigManager:
    def __init__(self, config_path=None):
        self._config = {}
        if config_path:
            self.load(config_path)

    def load(self, path):
        path = Path(path)
        suffix = path.suffix.lower()

        if suffix == ".json":
            self._load_json(path)
        elif suffix in (".yaml", ".yml"):
            self._load_yaml(path)
        elif suffix in (".ini", ".cfg"):
            self._load_ini(path)
        else:
            raise ValueError(f"不支持的配置格式: {suffix}")

    def _load_json(self, path):
        with open(path, "r", encoding="utf-8") as f:
            self._config = json.load(f)

    def get(self, key, default=None):
        """支持点号分隔的嵌套键: 'database.host'"""
        keys = key.split(".")
        value = self._config
        for k in keys:
            if isinstance(value, dict):
                value = value.get(k)
            else:
                return default
        return value if value is not None else default

    def set(self, key, value):
        keys = key.split(".")
        target = self._config
        for k in keys[:-1]:
            target = target.setdefault(k, {})
        target[keys[-1]] = value

    def save(self, path):
        with open(path, "w", encoding="utf-8") as f:
            json.dump(self._config, f, ensure_ascii=False, indent=2)


# 使用示例
config = ConfigManager("config.json")
config.set("database.host", "localhost")
config.set("database.port", 3306)
config.save("config.json")
```
